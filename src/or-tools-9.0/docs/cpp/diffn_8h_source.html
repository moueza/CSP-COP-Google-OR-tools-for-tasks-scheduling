151 // Enforces that the boxes with corners in (x, y), (x + dx, y), (x, y + dy)
152 // and (x + dx, y + dy) do not overlap.
153 // If strict is true, and if one box has a zero dimension, it still cannot
154 // intersect another box.
155 inline std::function<void(Model*)> NonOverlappingRectangles(
156  const std::vector<IntervalVariable>& x,
157  const std::vector<IntervalVariable>& y, bool is_strict,
158  bool add_cumulative_relaxation = true) {
159  return [=](Model* model) {
160  SchedulingConstraintHelper* x_helper =
161  new SchedulingConstraintHelper(x, model);
162  SchedulingConstraintHelper* y_helper =
163  new SchedulingConstraintHelper(y, model);
164  model->TakeOwnership(x_helper);
165  model->TakeOwnership(y_helper);
166 
167  NonOverlappingRectanglesEnergyPropagator* energy_constraint =
168  new NonOverlappingRectanglesEnergyPropagator(x_helper, y_helper);
169  GenericLiteralWatcher* const watcher =
170  model->GetOrCreate<GenericLiteralWatcher>();
171  watcher->SetPropagatorPriority(energy_constraint->RegisterWith(watcher), 3);
172  model->TakeOwnership(energy_constraint);
173 
174  NonOverlappingRectanglesDisjunctivePropagator* constraint =
175  new NonOverlappingRectanglesDisjunctivePropagator(is_strict, x_helper,
176  y_helper, model);
177  constraint->Register(/*fast_priority=*/3, /*slow_priority=*/4);
178  model->TakeOwnership(constraint);
179 
180  if (add_cumulative_relaxation) {
181  AddCumulativeRelaxation(x, x_helper, y_helper, model);
182  AddCumulativeRelaxation(y, y_helper, x_helper, model);
183  }
184  };
185 }
