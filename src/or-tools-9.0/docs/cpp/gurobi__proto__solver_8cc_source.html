41 inline absl::Status GurobiCodeToUtilStatus(int error_code,
42  const char* source_file,
43  int source_line,
44  const char* statement,
45  GRBenv* const env) {
46  if (error_code == GRB_OK) return absl::OkStatus();
47  return absl::InvalidArgumentError(absl::StrFormat(
48  "Gurobi error code %d (file '%s', line %d) on '%s': %s", error_code,
49  source_file, source_line, statement, GRBgeterrormsg(env)));

52 int AddIndicatorConstraint(const MPGeneralConstraintProto& gen_cst,
53  GRBmodel* gurobi_model,
54  std::vector<int>* tmp_variables,
55  std::vector<double>* tmp_coefficients) {
56  CHECK(gurobi_model != nullptr);
57  CHECK(tmp_variables != nullptr);
58  CHECK(tmp_coefficients != nullptr);
59 
60  const auto& ind_cst = gen_cst.indicator_constraint();
61  MPConstraintProto cst = ind_cst.constraint();
62  if (cst.lower_bound() > -std::numeric_limits<double>::infinity()) {
63  int status = GRBaddgenconstrIndicator(
64  gurobi_model, gen_cst.name().c_str(), ind_cst.var_index(),
65  ind_cst.var_value(), cst.var_index_size(),
66  cst.mutable_var_index()->mutable_data(),
67  cst.mutable_coefficient()->mutable_data(),
68  cst.upper_bound() == cst.lower_bound() ? GRB_EQUAL : GRB_GREATER_EQUAL,
69  cst.lower_bound());
70  if (status != GRB_OK) return status;
71  }
72  if (cst.upper_bound() < std::numeric_limits<double>::infinity() &&
73  cst.lower_bound() != cst.upper_bound()) {
74  return GRBaddgenconstrIndicator(gurobi_model, gen_cst.name().c_str(),
75  ind_cst.var_index(), ind_cst.var_value(),
76  cst.var_index_size(),
77  cst.mutable_var_index()->mutable_data(),
78  cst.mutable_coefficient()->mutable_data(),
79  GRB_LESS_EQUAL, cst.upper_bound());
80  }
81 
82  return GRB_OK;
83 }

85 int AddSosConstraint(const MPSosConstraint& sos_cst, GRBmodel* gurobi_model,
86  std::vector<int>* tmp_variables,
87  std::vector<double>* tmp_weights) {
88  CHECK(gurobi_model != nullptr);
89  CHECK(tmp_variables != nullptr);
90  CHECK(tmp_weights != nullptr);
91 
92  tmp_variables->resize(sos_cst.var_index_size(), 0);
93  for (int v = 0; v < sos_cst.var_index_size(); ++v) {
94  (*tmp_variables)[v] = sos_cst.var_index(v);
95  }
96  tmp_weights->resize(sos_cst.var_index_size(), 0);
97  if (sos_cst.weight_size() == sos_cst.var_index_size()) {
98  for (int w = 0; w < sos_cst.weight_size(); ++w) {
99  (*tmp_weights)[w] = sos_cst.weight(w);
100  }
101  } else {
102  DCHECK_EQ(sos_cst.weight_size(), 0);
103  // Gurobi requires variable weights in their SOS constraints.
104  std::iota(tmp_weights->begin(), tmp_weights->end(), 1);
105  }
106 
107  std::vector<int> types = {sos_cst.type() == MPSosConstraint::SOS1_DEFAULT
108  ? GRB_SOS_TYPE1
109  : GRB_SOS_TYPE2};
110  std::vector<int> begins = {0};
111  return GRBaddsos(gurobi_model, /*numsos=*/1,
112  /*nummembers=*/sos_cst.var_index_size(),
113  /*types=*/types.data(),
114  /*beg=*/begins.data(), /*ind=*/tmp_variables->data(),
115  /*weight*/ tmp_weights->data());
116 }

118 int AddQuadraticConstraint(const MPGeneralConstraintProto& gen_cst,
119  GRBmodel* gurobi_model) {
120  CHECK(gurobi_model != nullptr);
121  constexpr double kInfinity = std::numeric_limits<double>::infinity();
122 
123  CHECK(gen_cst.has_quadratic_constraint());
124  const MPQuadraticConstraint& quad_cst = gen_cst.quadratic_constraint();
125 
126  auto addqconstr = [](GRBmodel* gurobi_model, MPQuadraticConstraint quad_cst,
127  char sense, double rhs, const std::string& name) {
128  return GRBaddqconstr(
129  gurobi_model,
130  /*numlnz=*/quad_cst.var_index_size(),
131  /*lind=*/quad_cst.mutable_var_index()->mutable_data(),
132  /*lval=*/quad_cst.mutable_coefficient()->mutable_data(),
133  /*numqnz=*/quad_cst.qvar1_index_size(),
134  /*qrow=*/quad_cst.mutable_qvar1_index()->mutable_data(),
135  /*qcol=*/quad_cst.mutable_qvar2_index()->mutable_data(),
136  /*qval=*/quad_cst.mutable_qcoefficient()->mutable_data(),
137  /*sense=*/sense,
138  /*rhs=*/rhs,
139  /*QCname=*/name.c_str());
140  };
141 
142  if (quad_cst.has_lower_bound() && quad_cst.lower_bound() > -kInfinity) {
143  const int grb_status =
144  addqconstr(gurobi_model, gen_cst.quadratic_constraint(),
145  GRB_GREATER_EQUAL, quad_cst.lower_bound(),
146  gen_cst.has_name() ? gen_cst.name() + "_lb" : "");
147  if (grb_status != GRB_OK) return grb_status;
148  }
149  if (quad_cst.has_upper_bound() && quad_cst.upper_bound() < kInfinity) {
150  const int grb_status =
151  addqconstr(gurobi_model, gen_cst.quadratic_constraint(), GRB_LESS_EQUAL,
152  quad_cst.upper_bound(),
153  gen_cst.has_name() ? gen_cst.name() + "_ub" : "");
154  if (grb_status != GRB_OK) return grb_status;
155  }
156 
157  return GRB_OK;
158 }

222 absl::Status SetSolverSpecificParameters(const std::string& parameters,
223  GRBenv* gurobi) {
224  if (parameters.empty()) return absl::OkStatus();
225  std::vector<std::string> error_messages;
226  for (absl::string_view line : absl::StrSplit(parameters, '\n')) {
227  // Comment tokens end at the next new-line, or the end of the string.
228  // The first character must be '#'
229  if (line[0] == '#') continue;
230  for (absl::string_view token :
231  absl::StrSplit(line, ',', absl::SkipWhitespace())) {
232  if (token.empty()) continue;
233  std::vector<std::string> key_value =
234  absl::StrSplit(token, absl::ByAnyChar(" ="), absl::SkipWhitespace());
235  // If one parameter fails, we keep processing the list of parameters.
236  if (key_value.size() != 2) {
237  const std::string current_message =
238  absl::StrCat("Cannot parse parameter '", token,
239  "'. Expected format is 'ParameterName value' or "
240  "'ParameterName=value'");
241  error_messages.push_back(current_message);
242  continue;
243  }
244  const int gurobi_code =
245  GRBsetparam(gurobi, key_value[0].c_str(), key_value[1].c_str());
246  if (gurobi_code != GRB_OK) {
247  const std::string current_message = absl::StrCat(
248  "Error setting parameter '", key_value[0], "' to value '",
249  key_value[1], "': ", GRBgeterrormsg(gurobi));
250  error_messages.push_back(current_message);
251  continue;
252  }
253  VLOG(2) << absl::StrCat("Set parameter '", key_value[0], "' to value '",
254  key_value[1]);
255  }
256  }
257 
258  if (error_messages.empty()) return absl::OkStatus();
259  return absl::InvalidArgumentError(absl::StrJoin(error_messages, "\n"));
260 }

262 absl::StatusOr<MPSolutionResponse> GurobiSolveProto(
263  const MPModelRequest& request, GRBenv* gurobi_env) {
264  MPSolutionResponse response;
265  const absl::optional<LazyMutableCopy<MPModelProto>> optional_model =
266  ExtractValidMPModelOrPopulateResponseStatus(request, &response);
267  if (!optional_model) return response;
268  const MPModelProto& model = optional_model->get();
269 
270  // We set `gurobi_env` to point to a new environment if no existing one is
271  // provided. We must make sure that we free this environment when we exit this
272  // function.
273  bool gurobi_env_was_created = false;
274  auto gurobi_env_deleter = absl::MakeCleanup([&]() {
275  if (gurobi_env_was_created && gurobi_env != nullptr) {
276  GRBfreeenv(gurobi_env);
277  }
278  });
279  if (gurobi_env == nullptr) {
280  ASSIGN_OR_RETURN(gurobi_env, GetGurobiEnv());
281  gurobi_env_was_created = true;
282  }
283 
284  GRBmodel* gurobi_model = nullptr;
285  auto gurobi_model_deleter = absl::MakeCleanup([&]() {
286  const int error_code = GRBfreemodel(gurobi_model);
287  LOG_IF(DFATAL, error_code != GRB_OK)
288  << "GRBfreemodel failed with error " << error_code << ": "
289  << GRBgeterrormsg(gurobi_env);
290  });
291 
292 // `gurobi_env` references ther GRBenv argument.
293 #define RETURN_IF_GUROBI_ERROR(x) \
294  RETURN_IF_ERROR( \
295  GurobiCodeToUtilStatus(x, __FILE__, __LINE__, #x, gurobi_env));
296 
297  RETURN_IF_GUROBI_ERROR(GRBnewmodel(gurobi_env, &gurobi_model,
298  model.name().c_str(),
299  /*numvars=*/0,
300  /*obj=*/nullptr,
301  /*lb=*/nullptr,
302  /*ub=*/nullptr,
303  /*vtype=*/nullptr,
304  /*varnames=*/nullptr));
305  GRBenv* const model_env = GRBgetenv(gurobi_model);
306 
307  if (request.has_solver_specific_parameters()) {
308  const auto parameters_status = SetSolverSpecificParameters(
309  request.solver_specific_parameters(), model_env);
310  if (!parameters_status.ok()) {
311  response.set_status(MPSOLVER_MODEL_INVALID_SOLVER_PARAMETERS);
312  response.set_status_str(
313  std::string(parameters_status.message())); // NOLINT
314  return response;
315  }
316  }
317  if (request.solver_time_limit_seconds() > 0) {
318  RETURN_IF_GUROBI_ERROR(GRBsetdblparam(model_env, GRB_DBL_PAR_TIMELIMIT,
319  request.solver_time_limit_seconds()));
320  }
321  RETURN_IF_GUROBI_ERROR(
322  GRBsetintparam(model_env, GRB_INT_PAR_OUTPUTFLAG,
323  request.enable_internal_solver_output()));
324 
325  const int variable_size = model.variable_size();
326  bool has_integer_variables = false;
327  {
328  std::vector<double> obj_coeffs(variable_size, 0);
329  std::vector<double> lb(variable_size);
330  std::vector<double> ub(variable_size);
331  std::vector<char> ctype(variable_size);
332  std::vector<const char*> varnames(variable_size);
333  for (int v = 0; v < variable_size; ++v) {
334  const MPVariableProto& variable = model.variable(v);
335  obj_coeffs[v] = variable.objective_coefficient();
336  lb[v] = variable.lower_bound();
337  ub[v] = variable.upper_bound();
338  ctype[v] = variable.is_integer() ? GRB_INTEGER : GRB_CONTINUOUS;
339  if (variable.is_integer()) has_integer_variables = true;
340  if (!variable.name().empty()) varnames[v] = variable.name().c_str();
341  }
342 
343  RETURN_IF_GUROBI_ERROR(
344  GRBaddvars(gurobi_model, variable_size, 0, nullptr, nullptr, nullptr,
345  /*obj=*/obj_coeffs.data(),
346  /*lb=*/lb.data(), /*ub=*/ub.data(), /*vtype=*/ctype.data(),
347  /*varnames=*/const_cast<char**>(varnames.data())));
348 
349  // Set solution hints if any.
350  for (int i = 0; i < model.solution_hint().var_index_size(); ++i) {
351  RETURN_IF_GUROBI_ERROR(GRBsetdblattrelement(
352  gurobi_model, GRB_DBL_ATTR_START, model.solution_hint().var_index(i),
353  model.solution_hint().var_value(i)));
354  }
355  }
356 
357  {
358  std::vector<int> ct_variables;
359  std::vector<double> ct_coefficients;
360  for (int c = 0; c < model.constraint_size(); ++c) {
361  const MPConstraintProto& constraint = model.constraint(c);
362  const int size = constraint.var_index_size();
363  ct_variables.resize(size, 0);
364  ct_coefficients.resize(size, 0);
365  for (int i = 0; i < size; ++i) {
366  ct_variables[i] = constraint.var_index(i);
367  ct_coefficients[i] = constraint.coefficient(i);
368  }
369  // Using GRBaddrangeconstr for constraints that don't require it adds
370  // a slack which is not always removed by presolve.
371  if (constraint.lower_bound() == constraint.upper_bound()) {
372  RETURN_IF_GUROBI_ERROR(GRBaddconstr(
373  gurobi_model, /*numnz=*/size, /*cind=*/ct_variables.data(),
374  /*cval=*/ct_coefficients.data(),
375  /*sense=*/GRB_EQUAL, /*rhs=*/constraint.lower_bound(),
376  /*constrname=*/constraint.name().c_str()));
377  } else if (constraint.lower_bound() ==
378  -std::numeric_limits<double>::infinity()) {
379  RETURN_IF_GUROBI_ERROR(GRBaddconstr(
380  gurobi_model, /*numnz=*/size, /*cind=*/ct_variables.data(),
381  /*cval=*/ct_coefficients.data(),
382  /*sense=*/GRB_LESS_EQUAL, /*rhs=*/constraint.upper_bound(),
383  /*constrname=*/constraint.name().c_str()));
384  } else if (constraint.upper_bound() ==
385  std::numeric_limits<double>::infinity()) {
386  RETURN_IF_GUROBI_ERROR(GRBaddconstr(
387  gurobi_model, /*numnz=*/size, /*cind=*/ct_variables.data(),
388  /*cval=*/ct_coefficients.data(),
389  /*sense=*/GRB_GREATER_EQUAL, /*rhs=*/constraint.lower_bound(),
390  /*constrname=*/constraint.name().c_str()));
391  } else {
392  RETURN_IF_GUROBI_ERROR(GRBaddrangeconstr(
393  gurobi_model, /*numnz=*/size, /*cind=*/ct_variables.data(),
394  /*cval=*/ct_coefficients.data(),
395  /*lower=*/constraint.lower_bound(),
396  /*upper=*/constraint.upper_bound(),
397  /*constrname=*/constraint.name().c_str()));
398  }
399  }
400 
401  for (const auto& gen_cst : model.general_constraint()) {
402  switch (gen_cst.general_constraint_case()) {
403  case MPGeneralConstraintProto::kIndicatorConstraint: {
404  RETURN_IF_GUROBI_ERROR(AddIndicatorConstraint(
405  gen_cst, gurobi_model, &ct_variables, &ct_coefficients));
406  break;
407  }
408  case MPGeneralConstraintProto::kSosConstraint: {
409  RETURN_IF_GUROBI_ERROR(AddSosConstraint(gen_cst.sos_constraint(),
410  gurobi_model, &ct_variables,
411  &ct_coefficients));
412  break;
413  }
414  case MPGeneralConstraintProto::kQuadraticConstraint: {
415  RETURN_IF_GUROBI_ERROR(AddQuadraticConstraint(gen_cst, gurobi_model));
416  break;
417  }
418  case MPGeneralConstraintProto::kAbsConstraint: {
419  RETURN_IF_GUROBI_ERROR(GRBaddgenconstrAbs(
420  gurobi_model,
421  /*name=*/gen_cst.name().c_str(),
422  /*resvar=*/gen_cst.abs_constraint().resultant_var_index(),
423  /*argvar=*/gen_cst.abs_constraint().var_index()));
424  break;
425  }
426  case MPGeneralConstraintProto::kAndConstraint: {
427  RETURN_IF_GUROBI_ERROR(
428  AddAndConstraint(gen_cst, gurobi_model, &ct_variables));
429  break;
430  }
431  case MPGeneralConstraintProto::kOrConstraint: {
432  RETURN_IF_GUROBI_ERROR(
433  AddOrConstraint(gen_cst, gurobi_model, &ct_variables));
434  break;
435  }
436  case MPGeneralConstraintProto::kMinConstraint: {
437  RETURN_IF_GUROBI_ERROR(
438  AddMinConstraint(gen_cst, gurobi_model, &ct_variables));
439  break;
440  }
441  case MPGeneralConstraintProto::kMaxConstraint: {
442  RETURN_IF_GUROBI_ERROR(
443  AddMaxConstraint(gen_cst, gurobi_model, &ct_variables));
444  break;
445  }
446  default:
447  return absl::UnimplementedError(
448  absl::StrFormat("General constraints of type %i not supported.",
449  gen_cst.general_constraint_case()));
450  }
451  }
452  }
453 
454  RETURN_IF_GUROBI_ERROR(GRBsetintattr(gurobi_model, GRB_INT_ATTR_MODELSENSE,
455  model.maximize() ? -1 : 1));
456  RETURN_IF_GUROBI_ERROR(GRBsetdblattr(gurobi_model, GRB_DBL_ATTR_OBJCON,
457  model.objective_offset()));
458  if (model.has_quadratic_objective()) {
459  MPQuadraticObjective qobj = model.quadratic_objective();
460  if (qobj.coefficient_size() > 0) {
461  RETURN_IF_GUROBI_ERROR(
462  GRBaddqpterms(gurobi_model, /*numqnz=*/qobj.coefficient_size(),
463  /*qrow=*/qobj.mutable_qvar1_index()->mutable_data(),
464  /*qcol=*/qobj.mutable_qvar2_index()->mutable_data(),
465  /*qval=*/qobj.mutable_coefficient()->mutable_data()));
466  }
467  }
468 
469  RETURN_IF_GUROBI_ERROR(GRBupdatemodel(gurobi_model));
470  RETURN_IF_GUROBI_ERROR(GRBoptimize(gurobi_model));
471 
472  int optimization_status = 0;
473  RETURN_IF_GUROBI_ERROR(
474  GRBgetintattr(gurobi_model, GRB_INT_ATTR_STATUS, &optimization_status));
475  int solution_count = 0;
476  RETURN_IF_GUROBI_ERROR(
477  GRBgetintattr(gurobi_model, GRB_INT_ATTR_SOLCOUNT, &solution_count));
478  switch (optimization_status) {
479  case GRB_OPTIMAL:
480  response.set_status(MPSOLVER_OPTIMAL);
481  break;
482  case GRB_INF_OR_UNBD:
483  DLOG(INFO) << "Gurobi solve returned GRB_INF_OR_UNBD, which we treat as "
484  "INFEASIBLE even though it may mean UNBOUNDED.";
485  response.set_status_str(
486  "The model may actually be unbounded: Gurobi returned "
487  "GRB_INF_OR_UNBD");
488  ABSL_FALLTHROUGH_INTENDED;
489  case GRB_INFEASIBLE:
490  response.set_status(MPSOLVER_INFEASIBLE);
491  break;
492  case GRB_UNBOUNDED:
493  response.set_status(MPSOLVER_UNBOUNDED);
494  break;
495  default: {
496  if (solution_count > 0) {
497  response.set_status(MPSOLVER_FEASIBLE);
498  } else {
499  response.set_status(MPSOLVER_NOT_SOLVED);
500  response.set_status_str(
501  absl::StrFormat("Gurobi status code %d", optimization_status));
502  }
503  break;
504  }
505  }
506 
507  if (solution_count > 0 && (response.status() == MPSOLVER_FEASIBLE ||
508  response.status() == MPSOLVER_OPTIMAL)) {
509  double objective_value = 0;
510  RETURN_IF_GUROBI_ERROR(
511  GRBgetdblattr(gurobi_model, GRB_DBL_ATTR_OBJVAL, &objective_value));
512  response.set_objective_value(objective_value);
513  double best_objective_bound = 0;
514  const int error = GRBgetdblattr(gurobi_model, GRB_DBL_ATTR_OBJBOUND,
515  &best_objective_bound);
516  if (response.status() == MPSOLVER_OPTIMAL &&
517  error == GRB_ERROR_DATA_NOT_AVAILABLE) {
518  // If the presolve deletes all variables, there's no best bound.
