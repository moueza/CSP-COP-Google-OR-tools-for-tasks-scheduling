80  static constexpr double kInf = std::numeric_limits<double>::infinity();
81  // Data associated with each constraint. With it we know if the underlying
82  // representation is either:
83  // linear_terms <= upper_bound (if lower bound <= -GRB_INFINITY)
84  // linear_terms >= lower_bound (if upper bound >= GRB_INFINTY)
85  // linear_terms == xxxxx_bound (if upper_bound == lower_bound)
86  // linear_term - slack == 0 (with slack bounds equal to xxxxx_bound)
87  struct ConstraintData {
88  GurobiLinearConstraintIndex constraint_index = kUnspecifiedConstraint;
89  // only valid for true ranged constraints.
90  GurobiVariableIndex slack_index = kUnspecifiedIndex;
91  double lower_bound = -kInf;
92  double upper_bound = kInf;
93  };

176  // Note: Gurobi environments CAN be shared across several models, however
177  // there are some caveats:
178  // - Environments are not thread-safe.
179  // - Once a gurobi_model_ is created, it makes an internal copy of the
180  // "master" environment, so, later changes to that environment will not
181  // be reflected in the gurobi_model_, for that reason we also keep
182  // around a pointer to the gurobi_model_ environment in the
183  // `active_env_` (which should not be freed).
184  // - Every "master" environment counts as a "use" of a Gurobi License.
185  // This means that if you have a limited usage count of licenses, this
186  // implementation will be consuming more licenses. On the other hand, if
187  // you have a machine license, a site license, or an academic license,
188  // this disadvantage goes away.
189  //
190  // TODO(user) implement a sharing master Gurobi environment mode.
191  // This would be akin to the `default environment` of Gurobi in python.
192  GRBenv* master_env_ = nullptr;
193  GRBenv* active_env_ = nullptr;
194  GRBmodel* gurobi_model_ = nullptr;
195  // Note that we use linked_hash_map because the index of the gurobi_model_
196  // variables/constraints is exactly the order in which they are added to the
197  // model.
198  // Internal correspondence from variable proto IDs to Gurobi-numbered
199  // variables.
200  gtl::linked_hash_map<VariableId, GurobiVariableIndex> variables_map_;
201  // Internal correspondence from linear constraint proto IDs to
202  // Gurobi-numbered linear constraint and extra information.
203  gtl::linked_hash_map<LinearConstraintId, ConstraintData>
204  linear_constraints_map_;
205  // For those constraints that need a slack in the gurobi_model_
206  // representation (i.e. those with
207  // ConstraintData.slack_index != kUnspecifiedIndex), we keep a hash_map of
208  // those LinearConstraintId's to a reference of the ConstraintData stored in
209  // the linear_constraints_map_. This means that we only have one location
210  // were we store information about constraints, but two places from where we
211  // can access it.
212  // Furthermore, the internal slack_index associated with each
213  // actual slack variable is increasing in the order of them in this
214  // slack_map_, and is the main reason why we choose to use
215  // gtl::linked_hash_map, as the order of insertion (of the remaining
216  // elements) is preserved after removals.
217  // Note that after deletions, the renumbering of variables must be done
218  // simultaneously for variables and slacks following the merged-order of the
219  // variable_index and slack_map_.slack_index.
220  gtl::linked_hash_map<LinearConstraintId, ConstraintData&> slack_map_;
221  // Number of Gurobi variables, this internal quantity is updated when we
222  // actually add or remove variables or constraints in the underlying Gurobi
223  // model. Furthermore, since 'ModelUpdate' can trigger both creation and
224  // deletion of variables and constraints, we batch those changes in two
225  // steps: first add all new variables and constraints, and then delete all
226  // variables and constraints that need deletion. Finally flush changes at
227  // the gurobi model level (if any deletion was performed).
228  int num_gurobi_variables_ = 0;
229 
230  static constexpr int kGrbBasicConstraint = 0;
231  static constexpr int kGrbNonBasicConstraint = -1;
232 };
