64 // Helper class to convert a flatzinc model to a CpModelProto.

70  // Note that we always encode a constant argument with a constant variable.
71  int LookupVar(const fz::Argument& argument);

73 
74  // Create and return the indices of the IntervalConstraint corresponding
75  // to the flatzinc "interval" specified by a start var and a duration var.

81  // to the flatzinc "interval" specified by a start var and a size var.
82  // This method will cache intervals with the key <start_var, size_var>.

86  // corresponding to the flatzinc "interval" specified by a start var, the
87  // size_var, and the Boolean opt_var. This method will cache intervals with
88  // the key <start, duration, opt_var>.
89  int GetOrCreateOptionalInterval(int start_var, int size_var, int opt_var);

103  void TranslateSearchAnnotations(
104  const std::vector<fz::Annotation>& search_annotations);

107  CpModelProto proto;
108  SatParameters parameters;

117 int CpModelProtoWithMapping::LookupConstant(int64_t value) {
118  if (gtl::ContainsKey(constant_value_to_index, value)) {
119  return constant_value_to_index[value];
120  }
121 
122  // Create the constant on the fly.
123  const int index = proto.variables_size();
124  IntegerVariableProto* var_proto = proto.add_variables();
125  var_proto->add_domain(value);
126  var_proto->add_domain(value);
127  constant_value_to_index[value] = index;
128  return index;
129 }

131 int CpModelProtoWithMapping::LookupVar(const fz::Argument& argument) {
132  if (argument.HasOneValue()) return LookupConstant(argument.Value());
133  CHECK_EQ(argument.type, fz::Argument::INT_VAR_REF);
134  return fz_var_to_index[argument.Var()];

137 std::vector<int> CpModelProtoWithMapping::LookupVars(
138  const fz::Argument& argument) {
139  std::vector<int> result;
140  if (argument.type == fz::Argument::VOID_ARGUMENT) return result;
141  if (argument.type == fz::Argument::INT_LIST) {
142  for (int64_t value : argument.values) {
143  result.push_back(LookupConstant(value));
144  }
145  } else if (argument.type == fz::Argument::INT_VALUE) {
146  result.push_back(LookupConstant(argument.Value()));
147  } else {
148  CHECK_EQ(argument.type, fz::Argument::INT_VAR_REF_ARRAY);
149  for (fz::IntegerVariable* var : argument.variables) {
150  CHECK(var != nullptr);
151  result.push_back(fz_var_to_index[var]);
152  }
153  }
154  return result;
155 }

162 int CpModelProtoWithMapping::GetOrCreateOptionalInterval(int start_var,
163  int size_var,
164  int opt_var) {
165  const std::tuple<int, int, int> key =
166  std::make_tuple(start_var, size_var, opt_var);
167  if (gtl::ContainsKey(start_size_opt_tuple_to_interval, key)) {
168  return start_size_opt_tuple_to_interval[key];
169  }
170  const int interval_index = proto.constraints_size();
171 
172  auto* ct = proto.add_constraints();
173  if (opt_var != std::numeric_limits<int32_t>::max()) {
174  ct->add_enforcement_literal(opt_var);
175  }
176  auto* interval = ct->mutable_interval();
177  interval->set_start(start_var);
178  interval->set_size(size_var);
179 
180  interval->set_end(proto.variables_size());
181  auto* end_var = proto.add_variables();
182  const auto start_proto = proto.variables(start_var);
183  const auto size_proto = proto.variables(size_var);
184  end_var->add_domain(start_proto.domain(0) + size_proto.domain(0));
185  end_var->add_domain(start_proto.domain(start_proto.domain_size() - 1) +
186  size_proto.domain(size_proto.domain_size() - 1));
187  start_size_opt_tuple_to_interval[key] = interval_index;
188  return interval_index;
189 }

200 void CpModelProtoWithMapping::FillAMinusBInDomain(
201  const std::vector<int64_t>& domain, const fz::Constraint& fz_ct,
202  ConstraintProto* ct) {
203  auto* arg = ct->mutable_linear();
204  if (fz_ct.arguments[1].type == fz::Argument::INT_VALUE) {
205  const int64_t value = fz_ct.arguments[1].Value();
206  const int var_a = LookupVar(fz_ct.arguments[0]);
207  for (const int64_t domain_bound : domain) {
208  if (domain_bound == std::numeric_limits<int64_t>::min() ||
209  domain_bound == std::numeric_limits<int64_t>::max()) {
210  arg->add_domain(domain_bound);
211  } else {
212  arg->add_domain(domain_bound + value);
213  }
214  }
215  arg->add_vars(var_a);
216  arg->add_coeffs(1);
217  } else if (fz_ct.arguments[0].type == fz::Argument::INT_VALUE) {
218  const int64_t value = fz_ct.arguments[0].Value();
219  const int var_b = LookupVar(fz_ct.arguments[1]);
220  for (int64_t domain_bound : gtl::reversed_view(domain)) {
221  if (domain_bound == std::numeric_limits<int64_t>::min()) {
222  arg->add_domain(std::numeric_limits<int64_t>::max());
223  } else if (domain_bound == std::numeric_limits<int64_t>::max()) {
224  arg->add_domain(std::numeric_limits<int64_t>::min());
225  } else {
226  arg->add_domain(value - domain_bound);
227  }
228  }
229  arg->add_vars(var_b);
230  arg->add_coeffs(1);
231  } else {
232  for (const int64_t domain_bound : domain) arg->add_domain(domain_bound);
233  arg->add_vars(LookupVar(fz_ct.arguments[0]));
234  arg->add_coeffs(1);
235  arg->add_vars(LookupVar(fz_ct.arguments[1]));
236  arg->add_coeffs(-1);
237  }
238 }

240 void CpModelProtoWithMapping::FillLinearConstraintWithGivenDomain(
241  const std::vector<int64_t>& domain, const fz::Constraint& fz_ct,
242  ConstraintProto* ct) {
243  auto* arg = ct->mutable_linear();
244  for (const int64_t domain_bound : domain) arg->add_domain(domain_bound);
245  std::vector<int> vars = LookupVars(fz_ct.arguments[1]);
246  for (int i = 0; i < vars.size(); ++i) {
247  arg->add_vars(vars[i]);
248  arg->add_coeffs(fz_ct.arguments[0].values[i]);
249  }

252 void CpModelProtoWithMapping::FillConstraint(const fz::Constraint& fz_ct,
253  ConstraintProto* ct) {
254  if (fz_ct.type == "false_constraint") {
255  // An empty clause is always false.
256  ct->mutable_bool_or();
257  } else if (fz_ct.type == "bool_clause") {
258  auto* arg = ct->mutable_bool_or();
259  for (const int var : LookupVars(fz_ct.arguments[0])) {
260  arg->add_literals(TrueLiteral(var));
261  }
262  for (const int var : LookupVars(fz_ct.arguments[1])) {
263  arg->add_literals(FalseLiteral(var));
264  }
265  } else if (fz_ct.type == "bool_xor") {
266  // This is not the same semantics as the array_bool_xor as this constraint
267  // is actually a fully reified xor(a, b) <==> x.
268  const int a = LookupVar(fz_ct.arguments[0]);
269  const int b = LookupVar(fz_ct.arguments[1]);
270  const int x = LookupVar(fz_ct.arguments[2]);
271 
272  // not(x) => a == b
273  ct->add_enforcement_literal(NegatedRef(x));
274  auto* const refute = ct->mutable_linear();
275  refute->add_vars(a);
276  refute->add_coeffs(1);
277  refute->add_vars(b);
278  refute->add_coeffs(-1);
279  refute->add_domain(0);
280  refute->add_domain(0);
281 
282  // x => a + b == 1
283  auto* ct2 = proto.add_constraints();
284  ct2->add_enforcement_literal(x);
285  auto* const enforce = ct2->mutable_linear();
286  enforce->add_vars(a);
287  enforce->add_coeffs(1);
288  enforce->add_vars(b);
289  enforce->add_coeffs(1);
290  enforce->add_domain(1);
291  enforce->add_domain(1);
292  } else if (fz_ct.type == "array_bool_or") {
293  auto* arg = ct->mutable_bool_or();
294  for (const int var : LookupVars(fz_ct.arguments[0])) {
295  arg->add_literals(TrueLiteral(var));
296  }
297  } else if (fz_ct.type == "array_bool_or_negated") {
298  auto* arg = ct->mutable_bool_and();
299  for (const int var : LookupVars(fz_ct.arguments[0])) {
300  arg->add_literals(FalseLiteral(var));
301  }
302  } else if (fz_ct.type == "array_bool_and") {
303  auto* arg = ct->mutable_bool_and();
304  for (const int var : LookupVars(fz_ct.arguments[0])) {
305  arg->add_literals(TrueLiteral(var));
306  }
307  } else if (fz_ct.type == "array_bool_and_negated") {
308  auto* arg = ct->mutable_bool_or();
309  for (const int var : LookupVars(fz_ct.arguments[0])) {
310  arg->add_literals(FalseLiteral(var));
311  }
312  } else if (fz_ct.type == "array_bool_xor") {
313  auto* arg = ct->mutable_bool_xor();
314  for (const int var : LookupVars(fz_ct.arguments[0])) {
315  arg->add_literals(TrueLiteral(var));
316  }
317  } else if (fz_ct.type == "bool_le" || fz_ct.type == "int_le") {
318  FillAMinusBInDomain({std::numeric_limits<int64_t>::min(), 0}, fz_ct, ct);
319  } else if (fz_ct.type == "bool_ge" || fz_ct.type == "int_ge") {
320  FillAMinusBInDomain({0, std::numeric_limits<int64_t>::max()}, fz_ct, ct);
321  } else if (fz_ct.type == "bool_lt" || fz_ct.type == "int_lt") {
322  FillAMinusBInDomain({std::numeric_limits<int64_t>::min(), -1}, fz_ct, ct);
323  } else if (fz_ct.type == "bool_gt" || fz_ct.type == "int_gt") {
324  FillAMinusBInDomain({1, std::numeric_limits<int64_t>::max()}, fz_ct, ct);
325  } else if (fz_ct.type == "bool_eq" || fz_ct.type == "int_eq" ||
326  fz_ct.type == "bool2int") {
327  FillAMinusBInDomain({0, 0}, fz_ct, ct);
328  } else if (fz_ct.type == "bool_ne" || fz_ct.type == "bool_not") {
329  auto* arg = ct->mutable_linear();
330  arg->add_vars(LookupVar(fz_ct.arguments[0]));
331  arg->add_coeffs(1);
332  arg->add_vars(LookupVar(fz_ct.arguments[1]));
333  arg->add_coeffs(1);
334  arg->add_domain(1);
335  arg->add_domain(1);
336  } else if (fz_ct.type == "int_ne") {
337  FillAMinusBInDomain({std::numeric_limits<int64_t>::min(), -1, 1,
338  std::numeric_limits<int64_t>::max()},
339  fz_ct, ct);
340  } else if (fz_ct.type == "int_lin_eq") {
341  const int64_t rhs = fz_ct.arguments[2].values[0];
342  FillLinearConstraintWithGivenDomain({rhs, rhs}, fz_ct, ct);
343  } else if (fz_ct.type == "bool_lin_eq") {
344  auto* arg = ct->mutable_linear();
345  const std::vector<int> vars = LookupVars(fz_ct.arguments[1]);
346  for (int i = 0; i < vars.size(); ++i) {
347  arg->add_vars(vars[i]);
348  arg->add_coeffs(fz_ct.arguments[0].values[i]);
349  }
350  if (fz_ct.arguments[2].IsVariable()) {
351  arg->add_vars(LookupVar(fz_ct.arguments[2]));
352  arg->add_coeffs(-1);
353  arg->add_domain(0);
354  arg->add_domain(0);
355  } else {
356  const int64_t v = fz_ct.arguments[2].Value();
357  arg->add_domain(v);
358  arg->add_domain(v);
359  }
360  } else if (fz_ct.type == "int_lin_le" || fz_ct.type == "bool_lin_le") {
361  const int64_t rhs = fz_ct.arguments[2].values[0];
362  FillLinearConstraintWithGivenDomain(
363  {std::numeric_limits<int64_t>::min(), rhs}, fz_ct, ct);
364  } else if (fz_ct.type == "int_lin_lt") {
365  const int64_t rhs = fz_ct.arguments[2].values[0];
366  FillLinearConstraintWithGivenDomain(
367  {std::numeric_limits<int64_t>::min(), rhs - 1}, fz_ct, ct);
368  } else if (fz_ct.type == "int_lin_ge") {
369  const int64_t rhs = fz_ct.arguments[2].values[0];
370  FillLinearConstraintWithGivenDomain(
371  {rhs, std::numeric_limits<int64_t>::max()}, fz_ct, ct);
372  } else if (fz_ct.type == "int_lin_gt") {
373  const int64_t rhs = fz_ct.arguments[2].values[0];
374  FillLinearConstraintWithGivenDomain(
375  {rhs + 1, std::numeric_limits<int64_t>::max()}, fz_ct, ct);
376  } else if (fz_ct.type == "int_lin_ne") {
377  const int64_t rhs = fz_ct.arguments[2].values[0];
378  FillLinearConstraintWithGivenDomain(
379  {std::numeric_limits<int64_t>::min(), rhs - 1, rhs + 1,
380  std::numeric_limits<int64_t>::max()},
381  fz_ct, ct);
382  } else if (fz_ct.type == "set_in") {
383  auto* arg = ct->mutable_linear();
384  arg->add_vars(LookupVar(fz_ct.arguments[0]));
385  arg->add_coeffs(1);
386  if (fz_ct.arguments[1].type == fz::Argument::INT_LIST) {
387  FillDomainInProto(Domain::FromValues(std::vector<int64_t>{
388  fz_ct.arguments[1].values.begin(),
389  fz_ct.arguments[1].values.end()}),
390  arg);
391  } else if (fz_ct.arguments[1].type == fz::Argument::INT_INTERVAL) {
392  FillDomainInProto(
393  Domain(fz_ct.arguments[1].values[0], fz_ct.arguments[1].values[1]),
394  arg);
395  } else {
396  LOG(FATAL) << "Wrong format";
397  }
398  } else if (fz_ct.type == "set_in_negated") {
399  auto* arg = ct->mutable_linear();
400  arg->add_vars(LookupVar(fz_ct.arguments[0]));
401  arg->add_coeffs(1);
402  if (fz_ct.arguments[1].type == fz::Argument::INT_LIST) {
403  FillDomainInProto(
404  Domain::FromValues(
405  std::vector<int64_t>{fz_ct.arguments[1].values.begin(),
406  fz_ct.arguments[1].values.end()})
407  .Complement(),
408  arg);
409  } else if (fz_ct.arguments[1].type == fz::Argument::INT_INTERVAL) {
410  FillDomainInProto(
411  Domain(fz_ct.arguments[1].values[0], fz_ct.arguments[1].values[1])
412  .Complement(),
413  arg);
414  } else {
415  LOG(FATAL) << "Wrong format";
416  }
417  } else if (fz_ct.type == "int_min") {
418  auto* arg = ct->mutable_int_min();
419  arg->set_target(LookupVar(fz_ct.arguments[2]));
420  arg->add_vars(LookupVar(fz_ct.arguments[0]));
421  arg->add_vars(LookupVar(fz_ct.arguments[1]));
422  } else if (fz_ct.type == "array_int_minimum" || fz_ct.type == "minimum_int") {
423  auto* arg = ct->mutable_int_min();
424  arg->set_target(LookupVar(fz_ct.arguments[0]));
425  for (const int var : LookupVars(fz_ct.arguments[1])) arg->add_vars(var);
426  } else if (fz_ct.type == "int_max") {
427  auto* arg = ct->mutable_int_max();
428  arg->set_target(LookupVar(fz_ct.arguments[2]));
429  arg->add_vars(LookupVar(fz_ct.arguments[0]));
430  arg->add_vars(LookupVar(fz_ct.arguments[1]));
431  } else if (fz_ct.type == "array_int_maximum" || fz_ct.type == "maximum_int") {
432  auto* arg = ct->mutable_int_max();
433  arg->set_target(LookupVar(fz_ct.arguments[0]));
434  for (const int var : LookupVars(fz_ct.arguments[1])) arg->add_vars(var);
435  } else if (fz_ct.type == "int_times") {
436  auto* arg = ct->mutable_int_prod();
437  arg->set_target(LookupVar(fz_ct.arguments[2]));
438  arg->add_vars(LookupVar(fz_ct.arguments[0]));
439  arg->add_vars(LookupVar(fz_ct.arguments[1]));
440  } else if (fz_ct.type == "int_abs") {
441  auto* arg = ct->mutable_int_max();
442  arg->set_target(LookupVar(fz_ct.arguments[1]));
443  arg->add_vars(LookupVar(fz_ct.arguments[0]));
444  arg->add_vars(-LookupVar(fz_ct.arguments[0]) - 1);
445  } else if (fz_ct.type == "int_plus") {
446  auto* arg = ct->mutable_linear();
447  FillDomainInProto(Domain(0, 0), arg);
448  arg->add_vars(LookupVar(fz_ct.arguments[0]));
449  arg->add_coeffs(1);
450  arg->add_vars(LookupVar(fz_ct.arguments[1]));
451  arg->add_coeffs(1);
452  arg->add_vars(LookupVar(fz_ct.arguments[2]));
453  arg->add_coeffs(-1);
454  } else if (fz_ct.type == "int_div") {
455  auto* arg = ct->mutable_int_div();
456  arg->add_vars(LookupVar(fz_ct.arguments[0]));
457  arg->add_vars(LookupVar(fz_ct.arguments[1]));
458  arg->set_target(LookupVar(fz_ct.arguments[2]));
459  } else if (fz_ct.type == "int_mod") {
460  auto* arg = ct->mutable_int_mod();
461  arg->set_target(LookupVar(fz_ct.arguments[2]));
462  arg->add_vars(LookupVar(fz_ct.arguments[0]));
463  arg->add_vars(LookupVar(fz_ct.arguments[1]));
464  } else if (fz_ct.type == "array_int_element" ||
465  fz_ct.type == "array_bool_element" ||
466  fz_ct.type == "array_var_int_element" ||
467  fz_ct.type == "array_var_bool_element" ||
468  fz_ct.type == "array_int_element_nonshifted") {
469  if (fz_ct.arguments[0].type == fz::Argument::INT_VAR_REF ||
470  fz_ct.arguments[0].type == fz::Argument::INT_VALUE) {
471  auto* arg = ct->mutable_element();
472  arg->set_index(LookupVar(fz_ct.arguments[0]));
473  arg->set_target(LookupVar(fz_ct.arguments[2]));
474 
475  if (!absl::EndsWith(fz_ct.type, "_nonshifted")) {
476  // Add a dummy variable at position zero because flatzinc index start
477  // at 1.
478  // TODO(user): Make sure that zero is not in the index domain...
479  arg->add_vars(arg->target());
480  }
481  for (const int var : LookupVars(fz_ct.arguments[1])) arg->add_vars(var);
482  } else {
483  // Special case added by the presolve or in flatzinc. We encode this
484  // as a table constraint.
485  CHECK(!absl::EndsWith(fz_ct.type, "_nonshifted"));
486  auto* arg = ct->mutable_table();
487 
488  // the constraint is:
489  // values[coeff1 * vars[0] + coeff2 * vars[1] + offset] == target.
490  for (const int var : LookupVars(fz_ct.arguments[0])) arg->add_vars(var);
491  arg->add_vars(LookupVar(fz_ct.arguments[2])); // the target
492 
493  const std::vector<int64_t>& values = fz_ct.arguments[1].values;
494  const int64_t coeff1 = fz_ct.arguments[3].values[0];
495  const int64_t coeff2 = fz_ct.arguments[3].values[1];
496  const int64_t offset = fz_ct.arguments[4].values[0] - 1;
497 
498  for (const int64_t a : AllValuesInDomain(proto.variables(arg->vars(0)))) {
499  for (const int64_t b :
500  AllValuesInDomain(proto.variables(arg->vars(1)))) {
501  const int index = coeff1 * a + coeff2 * b + offset;
502  CHECK_GE(index, 0);
503  CHECK_LT(index, values.size());
504  arg->add_values(a);
505  arg->add_values(b);
506  arg->add_values(values[index]);
507  }
508  }
509  }
510  } else if (fz_ct.type == "ortools_table_int") {
511  auto* arg = ct->mutable_table();
512  for (const int var : LookupVars(fz_ct.arguments[0])) arg->add_vars(var);
513  for (const int64_t value : fz_ct.arguments[1].values)
514  arg->add_values(value);
515  } else if (fz_ct.type == "ortools_regular") {
516  auto* arg = ct->mutable_automaton();
517  for (const int var : LookupVars(fz_ct.arguments[0])) arg->add_vars(var);
518 
519  int count = 0;
520  const int num_states = fz_ct.arguments[1].Value();
521  const int num_values = fz_ct.arguments[2].Value();
522  for (int i = 1; i <= num_states; ++i) {
523  for (int j = 1; j <= num_values; ++j) {
524  CHECK_LT(count, fz_ct.arguments[3].values.size());
525  const int next = fz_ct.arguments[3].values[count++];
526  if (next == 0) continue; // 0 is a failing state.
527  arg->add_transition_tail(i);
528  arg->add_transition_label(j);
529  arg->add_transition_head(next);
530  }
531  }
532 
533  arg->set_starting_state(fz_ct.arguments[4].Value());
534  switch (fz_ct.arguments[5].type) {
535  case fz::Argument::INT_VALUE: {
536  arg->add_final_states(fz_ct.arguments[5].values[0]);
537  break;
538  }
539  case fz::Argument::INT_INTERVAL: {
540  for (int v = fz_ct.arguments[5].values[0];
541  v <= fz_ct.arguments[5].values[1]; ++v) {
542  arg->add_final_states(v);
543  }
544  break;
545  }
546  case fz::Argument::INT_LIST: {
547  for (const int v : fz_ct.arguments[5].values) {
548  arg->add_final_states(v);
549  }
550  break;
551  }
552  default: {
553  LOG(FATAL) << "Wrong constraint " << fz_ct.DebugString();
554  }
555  }
556  } else if (fz_ct.type == "fzn_all_different_int") {
557  auto* arg = ct->mutable_all_diff();
558  for (const int var : LookupVars(fz_ct.arguments[0])) arg->add_vars(var);
559  } else if (fz_ct.type == "fzn_circuit" || fz_ct.type == "fzn_subcircuit") {
560  // Try to auto-detect if it is zero or one based.
561  bool found_zero = false;
562  bool found_size = false;
563  int64_t size = 0;
564  if (fz_ct.arguments[0].variables.empty() &&
565  !fz_ct.arguments[0].values.empty()) {
566  // Fully instantiated (sub)circuit constraints.
567  size = fz_ct.arguments[0].values.size();
568  for (const int64_t value : fz_ct.arguments[0].values) {
569  if (value == 0) found_zero = true;
570  if (value == size) found_size = true;
571  }
572  } else {
573  size = fz_ct.arguments[0].variables.size();
574  for (fz::IntegerVariable* const var : fz_ct.arguments[0].variables) {
575  if (var->domain.Min() == 0) found_zero = true;
576  if (var->domain.Max() == size) found_size = true;
577  }
578  }
579 
580  const bool is_one_based = !found_zero || found_size;
581  const int64_t min_index = is_one_based ? 1 : 0;
582  const int64_t max_index = min_index + size - 1;
583  // The arc-based mutable circuit.
584  auto* circuit_arg = ct->mutable_circuit();
585 
586  // We fully encode all variables so we can use the literal based circuit.
587  // TODO(user): avoid fully encoding more than once?
588  int64_t index = min_index;
589  const bool is_circuit = (fz_ct.type == "fzn_circuit");
590  for (const int var : LookupVars(fz_ct.arguments[0])) {
591  Domain domain = ReadDomainFromProto(proto.variables(var));
592 
593  // Restrict the domain of var to [min_index, max_index]
594  domain = domain.IntersectionWith(Domain(min_index, max_index));
595  if (is_circuit) {
596  // We simply make sure that the variable cannot take the value index.
597  domain = domain.IntersectionWith(Domain::FromIntervals(
598  {{std::numeric_limits<int64_t>::min(), index - 1},
599  {index + 1, std::numeric_limits<int64_t>::max()}}));
600  }
601  FillDomainInProto(domain, proto.mutable_variables(var));
602 
603  for (const ClosedInterval interval : domain.intervals()) {
604  for (int64_t value = interval.start; value <= interval.end; ++value) {
605  // Create one Boolean variable for this arc.
606  const int literal = proto.variables_size();
607  {
608  auto* new_var = proto.add_variables();
609  new_var->add_domain(0);
610  new_var->add_domain(1);
611  }
612 
613  // Add the arc.
614  circuit_arg->add_tails(index);
615  circuit_arg->add_heads(value);
616  circuit_arg->add_literals(literal);
617 
618  // literal => var == value.
619  {
620  auto* ct = proto.add_constraints();
621  ct->add_enforcement_literal(literal);
622  ct->mutable_linear()->add_coeffs(1);
623  ct->mutable_linear()->add_vars(var);
624  ct->mutable_linear()->add_domain(value);
625  ct->mutable_linear()->add_domain(value);
626  }
627 
628  // not(literal) => var != value
629  {
630  auto* ct = proto.add_constraints();
631  ct->add_enforcement_literal(NegatedRef(literal));
632  ct->mutable_linear()->add_coeffs(1);
633  ct->mutable_linear()->add_vars(var);
634  ct->mutable_linear()->add_domain(
635  std::numeric_limits<int64_t>::min());
636  ct->mutable_linear()->add_domain(value - 1);
637  ct->mutable_linear()->add_domain(value + 1);
638  ct->mutable_linear()->add_domain(
639  std::numeric_limits<int64_t>::max());
640  }
641  }
642  }
643 
644  ++index;
645  }
646  } else if (fz_ct.type == "fzn_inverse") {
647  auto* arg = ct->mutable_inverse();
648 
649  const auto direct_variables = LookupVars(fz_ct.arguments[0]);
650  const auto inverse_variables = LookupVars(fz_ct.arguments[1]);
651 
652  const int num_variables =
653  std::min(direct_variables.size(), inverse_variables.size());
654 
655  // Try to auto-detect if it is zero or one based.
656  bool found_zero = false;
657  bool found_size = false;
658  for (fz::IntegerVariable* const var : fz_ct.arguments[0].variables) {
659  if (var->domain.Min() == 0) found_zero = true;
660  if (var->domain.Max() == num_variables) found_size = true;
661  }
662  for (fz::IntegerVariable* const var : fz_ct.arguments[1].variables) {
663  if (var->domain.Min() == 0) found_zero = true;
664  if (var->domain.Max() == num_variables) found_size = true;
665  }
666 
667  // Add a dummy constant variable at zero if the indexing is one based.
668  const bool is_one_based = !found_zero || found_size;
669  const int offset = is_one_based ? 1 : 0;
670 
671  if (is_one_based) arg->add_f_direct(LookupConstant(0));
672  for (const int var : direct_variables) {
673  arg->add_f_direct(var);
674  // Intersect domains with offset + [0, num_variables).
675  FillDomainInProto(
676  ReadDomainFromProto(proto.variables(var))
677  .IntersectionWith(Domain(offset, num_variables - 1 + offset)),
678  proto.mutable_variables(var));
679  }
680 
681  if (is_one_based) arg->add_f_inverse(LookupConstant(0));
682  for (const int var : inverse_variables) {
683  arg->add_f_inverse(var);
684  // Intersect domains with offset + [0, num_variables).
685  FillDomainInProto(
686  ReadDomainFromProto(proto.variables(var))
687  .IntersectionWith(Domain(offset, num_variables - 1 + offset)),
688  proto.mutable_variables(var));
689  }
690  } else if (fz_ct.type == "fzn_cumulative") {
691  const std::vector<int> starts = LookupVars(fz_ct.arguments[0]);
692  const std::vector<int> durations = LookupVars(fz_ct.arguments[1]);
693  const std::vector<int> demands = LookupVars(fz_ct.arguments[2]);
694  const int capacity = LookupVar(fz_ct.arguments[3]);
695 
696  auto* arg = ct->mutable_cumulative();
697  arg->set_capacity(capacity);
698  for (int i = 0; i < starts.size(); ++i) {
699  // Special case for a 0-1 demand, we mark the interval as optional
700  // instead and fix the demand to 1.
701  if (proto.variables(demands[i]).domain().size() == 2 &&
702  proto.variables(demands[i]).domain(0) == 0 &&
703  proto.variables(demands[i]).domain(1) == 1 &&
704  proto.variables(capacity).domain(1) == 1) {
705  arg->add_intervals(
706  GetOrCreateOptionalInterval(starts[i], durations[i], demands[i]));
707  arg->add_demands(LookupConstant(1));
708  } else {
709  arg->add_intervals(GetOrCreateInterval(starts[i], durations[i]));
710  arg->add_demands(demands[i]);
711  }
712  }
713  } else if (fz_ct.type == "fzn_diffn" || fz_ct.type == "fzn_diffn_nonstrict") {
714  const std::vector<int> x = LookupVars(fz_ct.arguments[0]);
715  const std::vector<int> y = LookupVars(fz_ct.arguments[1]);
716  const std::vector<int> dx = LookupVars(fz_ct.arguments[2]);
717  const std::vector<int> dy = LookupVars(fz_ct.arguments[3]);
718  const std::vector<int> x_intervals = CreateIntervals(x, dx);
719  const std::vector<int> y_intervals = CreateIntervals(y, dy);
720  auto* arg = ct->mutable_no_overlap_2d();
721  for (int i = 0; i < x.size(); ++i) {
722  arg->add_x_intervals(x_intervals[i]);
723  arg->add_y_intervals(y_intervals[i]);
724  }
725  arg->set_boxes_with_null_area_can_overlap(fz_ct.type ==
726  "fzn_diffn_nonstrict");
727  } else if (fz_ct.type == "ortools_network_flow" ||
728  fz_ct.type == "ortools_network_flow_cost") {
729  // Note that we leave ct empty here (with just the name set).
730  // We simply do a linear encoding of this constraint.
731  const bool has_cost = fz_ct.type == "ortools_network_flow_cost";
732  const std::vector<int> flow = LookupVars(fz_ct.arguments[has_cost ? 3 : 2]);
733 
734  // Flow conservation constraints.
735  const int num_nodes = fz_ct.arguments[1].values.size();
736  std::vector<std::vector<int>> flows_per_node(num_nodes);
737  std::vector<std::vector<int>> coeffs_per_node(num_nodes);
738  const int num_arcs = fz_ct.arguments[0].values.size() / 2;
739  for (int arc = 0; arc < num_arcs; arc++) {
740  const int tail = fz_ct.arguments[0].values[2 * arc] - 1;
741  const int head = fz_ct.arguments[0].values[2 * arc + 1] - 1;
742  if (tail == head) continue;
743 
744  flows_per_node[tail].push_back(flow[arc]);
745  coeffs_per_node[tail].push_back(1);
746  flows_per_node[head].push_back(flow[arc]);
747  coeffs_per_node[head].push_back(-1);
748  }
749  for (int node = 0; node < num_nodes; node++) {
750  auto* arg = proto.add_constraints()->mutable_linear();
751  arg->add_domain(fz_ct.arguments[1].values[node]);
752  arg->add_domain(fz_ct.arguments[1].values[node]);
753  for (int i = 0; i < flows_per_node[node].size(); ++i) {
754  arg->add_vars(flows_per_node[node][i]);
755  arg->add_coeffs(coeffs_per_node[node][i]);
756  }
757  }
758 
759  if (has_cost) {
760  auto* arg = proto.add_constraints()->mutable_linear();
761  arg->add_domain(0);
762  arg->add_domain(0);
763  for (int arc = 0; arc < num_arcs; arc++) {
764  const int64_t weight = fz_ct.arguments[2].values[arc];
765  if (weight != 0) {
766  arg->add_vars(flow[arc]);
767  arg->add_coeffs(weight);
768  }
769  }
770  arg->add_vars(LookupVar(fz_ct.arguments[4]));
771  arg->add_coeffs(-1);
772  }
773  } else {
774  LOG(FATAL) << " Not supported " << fz_ct.type;
775  }
776 }

778 void CpModelProtoWithMapping::FillReifOrImpliedConstraint(
779  const fz::Constraint& fz_ct, ConstraintProto* ct) {
780  // Start by adding a non-reified version of the same constraint.
781  std::string simplified_type;
782  if (absl::EndsWith(fz_ct.type, "_reif")) {
783  // Remove _reif.
784  simplified_type = fz_ct.type.substr(0, fz_ct.type.size() - 5);
785  } else if (absl::EndsWith(fz_ct.type, "_imp")) {
786  // Remove _imp.
787  simplified_type = fz_ct.type.substr(0, fz_ct.type.size() - 4);
788  } else {
789  // Keep name as it is an implicit reified constraint.
790  simplified_type = fz_ct.type;
791  }
792 
793  // We need a copy to be able to change the type of the constraint.
794  fz::Constraint copy = fz_ct;
795  copy.type = simplified_type;
796 
797  // Create the CP-SAT constraint.
798  FillConstraint(copy, ct);
799 
800  // In case of reified constraints, the type of the opposite constraint.
801  std::string negated_type;
802 
803  // Fill enforcement_literal and set copy.type to the negated constraint.
804  if (simplified_type == "array_bool_or") {
805  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[1])));
806  negated_type = "array_bool_or_negated";
807  } else if (simplified_type == "array_bool_and") {
808  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[1])));
809  negated_type = "array_bool_and_negated";
810  } else if (simplified_type == "set_in") {
811  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
812  negated_type = "set_in_negated";
813  } else if (simplified_type == "bool_eq" || simplified_type == "int_eq") {
814  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
815  negated_type = "int_ne";
816  } else if (simplified_type == "bool_ne" || simplified_type == "int_ne") {
817  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
818  negated_type = "int_eq";
819  } else if (simplified_type == "bool_le" || simplified_type == "int_le") {
820  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
821  negated_type = "int_gt";
822  } else if (simplified_type == "bool_lt" || simplified_type == "int_lt") {
823  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
824  negated_type = "int_ge";
825  } else if (simplified_type == "bool_ge" || simplified_type == "int_ge") {
826  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
827  negated_type = "int_lt";
828  } else if (simplified_type == "bool_gt" || simplified_type == "int_gt") {
829  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[2])));
830  negated_type = "int_le";
831  } else if (simplified_type == "int_lin_eq") {
832  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
833  negated_type = "int_lin_ne";
834  } else if (simplified_type == "int_lin_ne") {
835  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
836  negated_type = "int_lin_eq";
837  } else if (simplified_type == "int_lin_le") {
838  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
839  negated_type = "int_lin_gt";
840  } else if (simplified_type == "int_lin_ge") {
841  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
842  negated_type = "int_lin_lt";
843  } else if (simplified_type == "int_lin_lt") {
844  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
845  negated_type = "int_lin_ge";
846  } else if (simplified_type == "int_lin_gt") {
847  ct->add_enforcement_literal(TrueLiteral(LookupVar(fz_ct.arguments[3])));
848  negated_type = "int_lin_le";
849  } else {
850  LOG(FATAL) << "Unsupported " << simplified_type;
851  }
852 
853  // One way implication. We can stop here.
854  if (absl::EndsWith(fz_ct.type, "_imp")) return;
855 
856  // Add the other side of the reification because CpModelProto only support
857  // half reification.
858  ConstraintProto* negated_ct = proto.add_constraints();
859  negated_ct->set_name(fz_ct.type + " (negated)");
860  negated_ct->add_enforcement_literal(
861  sat::NegatedRef(ct->enforcement_literal(0)));
862  copy.type = negated_type;
863  FillConstraint(copy, negated_ct);
864 }

929 std::string SolutionString(
930  const fz::SolutionOutputSpecs& output,
931  const std::function<int64_t(fz::IntegerVariable*)>& value_func) {
932  if (output.variable != nullptr) {
933  const int64_t value = value_func(output.variable);
934  if (output.display_as_boolean) {

936  ";");
937  } else {
938  return absl::StrCat(output.name, " = ", value, ";");
939  }
940  } else {
941  const int bound_size = output.bounds.size();
942  std::string result =
943  absl::StrCat(output.name, " = array", bound_size, "d(");

945  if (output.bounds[i].max_value >= output.bounds[i].min_value) {
946  absl::StrAppend(&result, output.bounds[i].min_value, "..",
947  output.bounds[i].max_value, ", ");
948  } else {
949  result.append("{},");
950  }
951  }
952  result.append("[");
953  for (int i = 0; i < output.flat_variables.size(); ++i) {
954  const int64_t value = value_func(output.flat_variables[i]);
955  if (output.display_as_boolean) {
956  result.append(value ? "true" : "false");
957  } else {
958  absl::StrAppend(&result, value);
959  }
960  if (i != output.flat_variables.size() - 1) {
961  result.append(", ");
962  }
963  }
964  result.append("]);");
965  return result;
966  }
967  return "";
968 }

999 void SolveFzWithCpModelProto(const fz::Model& fz_model,
1000  const fz::FlatzincSatParameters& p,
1001  const std::string& sat_params,
1002  SolverLogger* logger,
1003  SolverLogger* solution_logger) {
1004  CpModelProtoWithMapping m;
1005  m.proto.set_name(fz_model.name());
1006 
1007  // The translation is easy, we create one variable per flatzinc variable,
1008  // plus eventually a bunch of constant variables that will be created
1009  // lazily.
1010  int num_variables = 0;
1011  for (fz::IntegerVariable* fz_var : fz_model.variables()) {
1012  if (!fz_var->active) continue;
1013  m.fz_var_to_index[fz_var] = num_variables++;
1014  IntegerVariableProto* var = m.proto.add_variables();
1015  var->set_name(fz_var->name);
1016  if (fz_var->domain.is_interval) {
1017  if (fz_var->domain.values.empty()) {
1018  // The CP-SAT solver checks that constraints cannot overflow during
1019  // their propagation. Because of that, we trim undefined variable
1020  // domains (i.e. int in minizinc) to something hopefully large enough.
1021  LOG_FIRST_N(WARNING, 1)
1022  << "Using flag --fz_int_max for unbounded integer variables.";
1023  LOG_FIRST_N(WARNING, 1)
1024  << " actual domain is [" << -absl::GetFlag(FLAGS_fz_int_max)
1025  << ".." << absl::GetFlag(FLAGS_fz_int_max) << "]";
1026  var->add_domain(-absl::GetFlag(FLAGS_fz_int_max));
1027  var->add_domain(absl::GetFlag(FLAGS_fz_int_max));
1028  } else {
1029  var->add_domain(fz_var->domain.values[0]);
1030  var->add_domain(fz_var->domain.values[1]);
1031  }
1032  } else {
1033  FillDomainInProto(Domain::FromValues(fz_var->domain.values), var);
1034  }
1035  }
1036 
1037  // Translate the constraints.
1038  for (fz::Constraint* fz_ct : fz_model.constraints()) {
1039  if (fz_ct == nullptr || !fz_ct->active) continue;
1040  ConstraintProto* ct = m.proto.add_constraints();
1041  ct->set_name(fz_ct->type);
1042  if (absl::EndsWith(fz_ct->type, "_reif") ||
1043  absl::EndsWith(fz_ct->type, "_imp") || fz_ct->type == "array_bool_or" ||
1044  fz_ct->type == "array_bool_and") {
1045  m.FillReifOrImpliedConstraint(*fz_ct, ct);
1046  } else {
1047  m.FillConstraint(*fz_ct, ct);
1048  }
1049  }
1050 
1051  // Fill the objective.
1052  if (fz_model.objective() != nullptr) {
1053  CpObjectiveProto* objective = m.proto.mutable_objective();
1054  objective->add_coeffs(1);
1055  if (fz_model.maximize()) {
1056  objective->set_scaling_factor(-1);
1057  objective->add_vars(
1058  NegatedCpModelVariable(m.fz_var_to_index[fz_model.objective()]));
1059  } else {
1060  objective->add_vars(m.fz_var_to_index[fz_model.objective()]);
1061  }
1062  }
1063 
1064  // Fill the search order.
1065  m.TranslateSearchAnnotations(fz_model.search_annotations());
1066 
1067  if (p.display_all_solutions && !m.proto.has_objective()) {
1068  // Enumerate all sat solutions.
1069  m.parameters.set_enumerate_all_solutions(true);
1070  }
1071 
1072  m.parameters.set_log_search_progress(p.log_search_progress);
1073 
1074  if (p.use_free_search) {
1075  m.parameters.set_search_branching(SatParameters::AUTOMATIC_SEARCH);
1076  if (p.number_of_threads <= 1) {
1077  m.parameters.set_interleave_search(true);
1078  m.parameters.set_reduce_memory_usage_in_interleave_mode(true);
1079  }
1080  } else {
1081  m.parameters.set_search_branching(SatParameters::FIXED_SEARCH);
1082  }
1083  if (p.max_time_in_seconds > 0) {
1084  m.parameters.set_max_time_in_seconds(p.max_time_in_seconds);
1085  }
1086 
1087  // We don't support enumerating all solution in parallel for a SAT problem.
1088  // But note that we do support it for an optimization problem since the
1089  // meaning of p.all_solutions is not the same in this case.
1090  if (p.display_all_solutions && fz_model.objective() == nullptr) {
1091  m.parameters.set_num_search_workers(1);
1092  } else {
1093  m.parameters.set_num_search_workers(std::max(1, p.number_of_threads));
1094  }
1095 
1096  // The order is important, we want the flag parameters to overwrite anything
1097  // set in m.parameters.
1098  sat::SatParameters flag_parameters;
1099  CHECK(google::protobuf::TextFormat::ParseFromString(sat_params,
1100  &flag_parameters))
1101  << sat_params;
1102  m.parameters.MergeFrom(flag_parameters);
1103 
1104  // We only need an observer if 'p.all_solutions' is true.
1105  std::function<void(const CpSolverResponse&)> solution_observer = nullptr;
1106  if (p.display_all_solutions) {
1107  solution_observer = [&fz_model, &m, &p,
1108  solution_logger](const CpSolverResponse& r) {
1109  const std::string solution_string =
1110  SolutionString(fz_model, [&m, &r](fz::IntegerVariable* v) {
1111  return r.solution(gtl::FindOrDie(m.fz_var_to_index, v));
1112  });
1113  SOLVER_LOG(solution_logger, solution_string);
1114  if (p.display_statistics) {
1115  OutputFlatzincStats(r, solution_logger);
1116  }
1117  SOLVER_LOG(solution_logger, "----------");
1118  };
1119  }
1120 
1121  Model sat_model;
1122  sat_model.Add(NewSatParameters(m.parameters));
1123  if (solution_observer != nullptr) {
1124  sat_model.Add(NewFeasibleSolutionObserver(solution_observer));
1125  }
1126  // Setup logging.
1127  sat_model.GetOrCreate<SatParameters>()->set_log_to_stdout(false);
1128  sat_model.Register<SolverLogger>(logger);
1129 
1130  const CpSolverResponse response = SolveCpModel(m.proto, &sat_model);
1131 
1132  // Check the returned solution with the fz model checker.
1133  if (response.status() == CpSolverStatus::FEASIBLE ||
1134  response.status() == CpSolverStatus::OPTIMAL) {
1135  CHECK(CheckSolution(
1136  fz_model,
1137  [&response, &m](fz::IntegerVariable* v) {
1138  return response.solution(gtl::FindOrDie(m.fz_var_to_index, v));
1139  },
1140  logger));
1141  }
1142 
1143  // Output the solution in the flatzinc official format.
1144  if (solution_logger->LoggingIsEnabled()) {
1145  if (response.status() == CpSolverStatus::FEASIBLE ||
1146  response.status() == CpSolverStatus::OPTIMAL) {
1147  if (!p.display_all_solutions) { // Already printed otherwise.
1148  const std::string solution_string =
1149  SolutionString(fz_model, [&response, &m](fz::IntegerVariable* v) {
1150  return response.solution(gtl::FindOrDie(m.fz_var_to_index, v));
1151  });
1152  SOLVER_LOG(solution_logger, solution_string);
1153  SOLVER_LOG(solution_logger, "----------");
1154  }
1155  if (response.status() == CpSolverStatus::OPTIMAL) {
1156  SOLVER_LOG(solution_logger, "==========");
1157  }
1158  } else if (response.status() == CpSolverStatus::INFEASIBLE) {
1159  SOLVER_LOG(solution_logger, "=====UNSATISFIABLE=====");
1160  } else if (response.status() == CpSolverStatus::MODEL_INVALID) {
1161  const std::string error_message = ValidateCpModel(m.proto);
1162  VLOG(1) << "%% Error message = '" << error_message << "'";
1163  if (absl::StrContains(error_message, "overflow")) {
1164  SOLVER_LOG(solution_logger, "=====OVERFLOW=====");
1165  } else {
1166  SOLVER_LOG(solution_logger, "=====MODEL INVALID=====");
1167  }
1168  } else {
1169  SOLVER_LOG(solution_logger, "%% TIMEOUT");
1170  }
1171  if (p.display_statistics) {
1172  OutputFlatzincStats(response, solution_logger);
1173  }
1174  }
1175 }
