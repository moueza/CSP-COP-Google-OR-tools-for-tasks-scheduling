62 inline BasisStatus ConvertVariableStatus(const int status) {
63  switch (status) {
64  case GRB_BASIC:
65  return BasisStatus::BASIC;
66  case GRB_NONBASIC_LOWER:
67  return BasisStatus::AT_LOWER_BOUND;
68  case GRB_NONBASIC_UPPER:
69  return BasisStatus::AT_UPPER_BOUND;
70  case GRB_SUPERBASIC:
71  return BasisStatus::FREE;
72  default:
73  return BasisStatus::INVALID;
74  }
75 }

77 inline int GrbVariableStatus(const BasisStatus status) {
78  switch (status) {
79  case BasisStatus::BASIC:
80  return GRB_BASIC;
81  case BasisStatus::AT_LOWER_BOUND:
82  case BasisStatus::FIXED_VALUE:
83  return GRB_NONBASIC_LOWER;
84  case BasisStatus::AT_UPPER_BOUND:
85  return GRB_NONBASIC_UPPER;
86  case BasisStatus::FREE:
87  return GRB_SUPERBASIC;
88  case BasisStatus::INVALID:
89  default:
90  LOG(FATAL) << "Unexpected invalid initial_basis.";
91  return 0;
92  }
93 }

97 GurobiParametersProto MergeParameters(
98  const CommonSolveParametersProto& common_parameters,
99  const GurobiParametersProto& gurobi_parameters) {
100  GurobiParametersProto merged_parameters;
101  if (common_parameters.has_enable_output()) {
102  // TODO(user): Gurobi allows for custom display callbacks. This
103  // allows redirecting output to a file, or parsing it and retrieve key
104  // information, or redirect it to IDE outputs (such as in Jupiter
105  // Notebooks). We should install such a callback to manage output if this is
106  // enabled.
107  const int enable_output = common_parameters.enable_output() ? 1 : 0;
108  GurobiParametersProto::Parameter* const parameter =
109  merged_parameters.add_parameters();
110  parameter->set_name(GRB_INT_PAR_LOGTOCONSOLE);
111  parameter->set_value(absl::StrCat(enable_output));
112  }
113  if (common_parameters.has_time_limit()) {
114  const double time_limit = absl::ToDoubleSeconds(
115  util_time::DecodeGoogleApiProto(common_parameters.time_limit())
116  .value());
117  GurobiParametersProto::Parameter* const parameter =
118  merged_parameters.add_parameters();
119  parameter->set_name(GRB_DBL_PAR_TIMELIMIT);
120  parameter->set_value(absl::StrCat(time_limit));
121  }
122  if (common_parameters.has_threads()) {
123  const int threads = common_parameters.threads();
124  GurobiParametersProto::Parameter* const parameter =
125  merged_parameters.add_parameters();
126  parameter->set_name(GRB_INT_PAR_THREADS);
127  parameter->set_value(absl::StrCat(threads));
128  }
129  if (common_parameters.has_random_seed()) {
130  const int random_seed =
131  std::min(GRB_MAXINT, std::max(common_parameters.random_seed(), 0));
132  GurobiParametersProto::Parameter* const parameter =
133  merged_parameters.add_parameters();
134  parameter->set_name(GRB_INT_PAR_SEED);
135  parameter->set_value(absl::StrCat(random_seed));
136  }
137  if (common_parameters.lp_algorithm() != LP_ALGORITHM_UNSPECIFIED) {
138  GurobiParametersProto::Parameter* const parameter =
139  merged_parameters.add_parameters();
140  parameter->set_name(GRB_INT_PAR_METHOD);
141  switch (common_parameters.lp_algorithm()) {
142  case LP_ALGORITHM_PRIMAL_SIMPLEX:
143  parameter->set_value(absl::StrCat(GRB_METHOD_PRIMAL));
144  break;
145  case LP_ALGORITHM_DUAL_SIMPLEX:
146  parameter->set_value(absl::StrCat(GRB_METHOD_DUAL));
147  break;
148  case LP_ALGORITHM_BARRIER:
149  parameter->set_value(absl::StrCat(GRB_METHOD_BARRIER));
150  break;
151  default:
152  LOG(FATAL) << "LPAlgorithm: "
153  << ProtoEnumToString(common_parameters.lp_algorithm())
154  << " unknown, error setting Gurobi parameters";
155  }
156  }
157  if (common_parameters.scaling() != EMPHASIS_UNSPECIFIED) {
158  GurobiParametersProto::Parameter* const parameter =
159  merged_parameters.add_parameters();
160  parameter->set_name(GRB_INT_PAR_SCALEFLAG);
161  switch (common_parameters.scaling()) {
162  case EMPHASIS_OFF:
163  parameter->set_value(absl::StrCat(0));
164  break;
165  case EMPHASIS_LOW:
166  case EMPHASIS_MEDIUM:
167  parameter->set_value(absl::StrCat(1));
168  break;
169  case EMPHASIS_HIGH:
170  parameter->set_value(absl::StrCat(2));
171  break;
172  case EMPHASIS_VERY_HIGH:
173  parameter->set_value(absl::StrCat(3));
174  break;
175  default:
176  LOG(FATAL) << "Scaling emphasis: "
177  << ProtoEnumToString(common_parameters.scaling())
178  << " unknown, error setting Gurobi parameters";
179  }
180  }
181  if (common_parameters.cuts() != EMPHASIS_UNSPECIFIED) {
182  GurobiParametersProto::Parameter* const parameter =
183  merged_parameters.add_parameters();
184  parameter->set_name(GRB_INT_PAR_CUTS);
185  switch (common_parameters.cuts()) {
186  case EMPHASIS_OFF:
187  parameter->set_value(absl::StrCat(0));
188  break;
189  case EMPHASIS_LOW:
190  case EMPHASIS_MEDIUM:
191  parameter->set_value(absl::StrCat(1));
192  break;
193  case EMPHASIS_HIGH:
194  parameter->set_value(absl::StrCat(2));
195  break;
196  case EMPHASIS_VERY_HIGH:
197  parameter->set_value(absl::StrCat(3));
198  break;
199  default:
200  LOG(FATAL) << "Cuts emphasis: "
201  << ProtoEnumToString(common_parameters.cuts())
202  << " unknown, error setting Gurobi parameters";
203  }
204  }
205  if (common_parameters.heuristics() != EMPHASIS_UNSPECIFIED) {
206  GurobiParametersProto::Parameter* const parameter =
207  merged_parameters.add_parameters();
208  parameter->set_name(GRB_DBL_PAR_HEURISTICS);
209  switch (common_parameters.heuristics()) {
210  case EMPHASIS_OFF:
211  parameter->set_value(absl::StrCat(0.0));
212  break;
213  case EMPHASIS_LOW:
214  parameter->set_value(absl::StrCat(0.025));
215  break;
216  case EMPHASIS_MEDIUM:
217  // As of Gurobi 9.1 this is the default value.
218  // https://www.gurobi.com/documentation/9.1/refman/heuristics.html
219  parameter->set_value(absl::StrCat(0.05));
220  break;
221  case EMPHASIS_HIGH:
222  parameter->set_value(absl::StrCat(0.1));
223  break;
224  case EMPHASIS_VERY_HIGH:
225  parameter->set_value(absl::StrCat(0.2));
226  break;
227  default:
228  LOG(FATAL) << "Heuristics emphasis: "
229  << ProtoEnumToString(common_parameters.heuristics())
230  << " unknown, error setting Gurobi parameters";
231  }
232  }
233 
234  if (common_parameters.presolve() != EMPHASIS_UNSPECIFIED) {
235  GurobiParametersProto::Parameter* const parameter =
236  merged_parameters.add_parameters();
237  parameter->set_name(GRB_INT_PAR_PRESOLVE);
238  switch (common_parameters.presolve()) {
239  case EMPHASIS_OFF:
240  parameter->set_value(absl::StrCat(0));
241  break;
242  case EMPHASIS_LOW:
243  case EMPHASIS_MEDIUM:
244  parameter->set_value(absl::StrCat(1));
245  break;
246  case EMPHASIS_HIGH:
247  case EMPHASIS_VERY_HIGH:
248  parameter->set_value(absl::StrCat(2));
249  break;
250  default:
251  LOG(FATAL) << "Presolve emphasis: "
252  << ProtoEnumToString(common_parameters.presolve())
253  << " unknown, error setting Gurobi parameters";
254  }
255  }
256 
257  for (const GurobiParametersProto_Parameter& parameter :
258  gurobi_parameters.parameters()) {
259  *merged_parameters.add_parameters() = parameter;
260  }
261  return merged_parameters;
262 }

284 std::string GurobiSolver::GurobiErrorMessage(int error_code) const {
285  if (error_code == kGrbOk) {
286  return {};
287  }
288  if (active_env_ == nullptr) {
289  if (error_code == GRB_ERROR_NO_LICENSE) {
290  return absl::StrCat("Gurobi error code:", error_code,
291  " (Failed to obtain a valid license)");
292  } else {
293  return absl::StrCat("Gurboi error code:", error_code,
294  " (No environment is available)");
295  }
296  }
297  return absl::StrCat("Gurobi error code: ", error_code, ", ",
298  GRBgeterrormsg(active_env_));

332  CHECK(master_env_ == nullptr);

349  int gurobi_major, gurobi_minor, gurobi_technical;
350  GRBversion(&gurobi_major, &gurobi_minor, &gurobi_technical);
351  VLOG(3) << absl::StrFormat(
352  "Successfully opened environment for Gurobi v%d.%d.%d (%s)",
353  gurobi_major, gurobi_minor, gurobi_technical, GRBplatform());

394 absl::StatusOr<std::pair<SolveResultProto::TerminationReason, std::string>>
395 GurobiSolver::ConvertTerminationReason(const int gurobi_status,
396  const bool has_feasible_solution) {
397  switch (gurobi_status) {
398  case GRB_OPTIMAL:
399  return std::make_pair(SolveResultProto::OPTIMAL, "");
400  case GRB_INFEASIBLE:
401  return std::make_pair(SolveResultProto::INFEASIBLE, "");
402  case GRB_UNBOUNDED: {
403  if (has_feasible_solution) {
404  return std::make_pair(SolveResultProto::UNBOUNDED, "");
405  } else {
406  return std::make_pair(SolveResultProto::DUAL_INFEASIBLE,
407  "Gurobi status GRB_UNBOUNDED, but no feasible "
408  "point found, only primal ray.");
409  }
410  }
411  case GRB_INF_OR_UNBD:
412  return std::make_pair(SolveResultProto::DUAL_INFEASIBLE,
413  "Gurobi status GRB_INF_OR_UNBD.");
414  case GRB_CUTOFF:
415  return std::make_pair(SolveResultProto::OBJECTIVE_LIMIT,
416  "Gurobi status GRB_CUTOFF.");
417  case GRB_ITERATION_LIMIT:
418  return std::make_pair(SolveResultProto::ITERATION_LIMIT, "");
419  case GRB_NODE_LIMIT:
420  return std::make_pair(SolveResultProto::NODE_LIMIT, "");
421  case GRB_TIME_LIMIT:
422  return std::make_pair(SolveResultProto::TIME_LIMIT, "");
423  case GRB_SOLUTION_LIMIT:
424  return std::make_pair(SolveResultProto::SOLUTION_LIMIT, "");
425  case GRB_INTERRUPTED:
426  return std::make_pair(SolveResultProto::INTERRUPTED, "");
427  case GRB_NUMERIC:
428  return std::make_pair(SolveResultProto::NUMERICAL_ERROR, "");
429  case GRB_SUBOPTIMAL:
430  return std::make_pair(SolveResultProto::IMPRECISE, "");
431  case GRB_USER_OBJ_LIMIT:
432  return std::make_pair(SolveResultProto::OBJECTIVE_LIMIT,
433  "Gurobi status GRB_USR_OBJ_LIMIT.");
434  case GRB_LOADED:
435  return absl::InternalError(
436  "Error creating termination reason, unexpected gurobi status code "
437  "GRB_LOADED.");
438  case GRB_INPROGRESS:
439  return absl::InternalError(
440  "Error creating termination reason, unexpected gurobi status code "
441  "GRB_INPROGRESS.");
442  default:
443  return absl::InternalError(absl::StrCat(
444  "Missing Gurobi optimization status code case: ", gurobi_status));
445  }

477 absl::Status GurobiSolver::SetGurobiBasis(const BasisProto& basis) {
478  std::vector<int> gurobi_variable_basis_status(num_gurobi_variables_);
479  for (const auto [id, value] : MakeView(basis.variable_status())) {
480  gurobi_variable_basis_status[variables_map_.at(id)] =
481  GrbVariableStatus(static_cast<BasisStatus>(value));
482  }
483 
484  std::vector<int> gurobi_constraint_basis_status;
485  gurobi_constraint_basis_status.reserve(num_gurobi_constraints());
486  for (const auto [id, value] : MakeView(basis.constraint_status())) {
487  const ConstraintData& constraint_data = linear_constraints_map_.at(id);
488  // Non-ranged constraints
489  if (constraint_data.slack_index == kUnspecifiedIndex) {
490  if (value == BasisStatus::BASIC) {
491  gurobi_constraint_basis_status.push_back(kGrbBasicConstraint);
492  } else {
493  gurobi_constraint_basis_status.push_back(kGrbNonBasicConstraint);
494  }
495  // Ranged constraints
496  } else if (value == BasisStatus::BASIC) {
497  // Either constraint or MathOpt slack is basic, but not both (because
498  // columns for MathOpt slack and internal Gurobi slack are linearly
499  // dependent). We choose the MathOpt slack to be basic.
500  gurobi_variable_basis_status[constraint_data.slack_index] = GRB_BASIC;
501  gurobi_constraint_basis_status.push_back(kGrbNonBasicConstraint);
502  } else {
503  gurobi_variable_basis_status[constraint_data.slack_index] =
504  GrbVariableStatus(static_cast<BasisStatus>(value));
505  gurobi_constraint_basis_status.push_back(kGrbNonBasicConstraint);
506  }
507  }
508  RETURN_IF_GUROBI_ERROR(GRBsetintattrarray(
509  /*model=*/gurobi_model_, /*attrname=*/GRB_INT_ATTR_VBASIS,
510  /*first=*/0, /*len=*/gurobi_variable_basis_status.size(),
511  /*newvalues=*/gurobi_variable_basis_status.data()));
512  RETURN_IF_GUROBI_ERROR(GRBsetintattrarray(
513  /*model=*/gurobi_model_, /*attrname=*/GRB_INT_ATTR_CBASIS,
514  /*first=*/0, /*len=*/gurobi_constraint_basis_status.size(),
515  /*newvalues=*/gurobi_constraint_basis_status.data()));
516 
517  return absl::OkStatus();
518 }

520 absl::StatusOr<BasisProto> GurobiSolver::GetGurobiBasis() {
521  BasisProto basis;
522 
523  std::vector<int> gurobi_variable_basis_status(num_gurobi_variables_);
524  RETURN_IF_ERROR(GetIntAttrArray(
525  GRB_INT_ATTR_VBASIS, absl::MakeSpan(gurobi_variable_basis_status)));
526 
527  for (auto [variable_id, gurobi_variable_index] : variables_map_) {
528  basis.mutable_variable_status()->add_ids(variable_id);
529  const BasisStatus variable_status = ConvertVariableStatus(
530  gurobi_variable_basis_status[gurobi_variable_index]);
531  if (variable_status == BasisStatus::INVALID) {
532  return absl::InternalError(
533  absl::StrCat("Invalid Gurobi variable basis status: ",
534  gurobi_variable_basis_status[gurobi_variable_index]));
535  }
536  basis.mutable_variable_status()->add_values(variable_status);
537  }
538 
539  std::vector<int> gurobi_constraint_basis_status(num_gurobi_constraints());
540  RETURN_IF_ERROR(GetIntAttrArray(
541  GRB_INT_ATTR_CBASIS, absl::MakeSpan(gurobi_constraint_basis_status)));
542  for (auto [constraint_id, gurobi_data] : linear_constraints_map_) {
543  basis.mutable_constraint_status()->add_ids(constraint_id);
544  const int gurobi_constraint_status =
545  gurobi_constraint_basis_status[gurobi_data.constraint_index];
546  if ((gurobi_constraint_status != kGrbBasicConstraint) &&
547  (gurobi_constraint_status != kGrbNonBasicConstraint)) {
548  return absl::InternalError(
549  absl::StrCat("Invalid Gurobi constraint basis status: ",
550  gurobi_constraint_status));
551  }
552  // linear_terms <= upper_bound
553  if (gurobi_data.lower_bound <= -GRB_INFINITY &&
554  gurobi_data.upper_bound < GRB_INFINITY) {
555  if (gurobi_constraint_status == kGrbBasicConstraint) {
556  basis.mutable_constraint_status()->add_values(BasisStatus::BASIC);
557  } else {
558  basis.mutable_constraint_status()->add_values(
559  BasisStatus::AT_UPPER_BOUND);
560  }
561  // linear_terms >= lower_bound
562  } else if (gurobi_data.lower_bound > -GRB_INFINITY &&
563  gurobi_data.upper_bound >= GRB_INFINITY) {
564  if (gurobi_constraint_status == kGrbBasicConstraint) {
565  basis.mutable_constraint_status()->add_values(BasisStatus::BASIC);
566  } else {
567  basis.mutable_constraint_status()->add_values(
568  BasisStatus::AT_LOWER_BOUND);
569  }
570  // linear_terms == xxxxx_bound
571  } else if (gurobi_data.lower_bound == gurobi_data.upper_bound) {
572  if (gurobi_constraint_status == kGrbBasicConstraint) {
573  basis.mutable_constraint_status()->add_values(BasisStatus::BASIC);
574  } else {
575  // TODO(user): consider refining this to
576  // AT_LOWER_BOUND/AT_UPPER_BOUND using the sign of the dual variable.
577  basis.mutable_constraint_status()->add_values(BasisStatus::FIXED_VALUE);
578  }
579  // linear_term - slack == 0 (ranged constraint)
580  } else {
581  const BasisStatus slack_status = ConvertVariableStatus(
582  gurobi_variable_basis_status[gurobi_data.slack_index]);
583  if (slack_status == BasisStatus::INVALID) {
584  return absl::InternalError(absl::StrCat(
585  "Invalid Gurobi slack variable basis status: ", slack_status));
586  }
587  if ((gurobi_constraint_status == kGrbBasicConstraint) ||
588  (slack_status == BasisStatus::BASIC)) {
589  basis.mutable_constraint_status()->add_values(BasisStatus::BASIC);
590  } else {
591  basis.mutable_constraint_status()->add_values(slack_status);
592  }
593  }
594  }
595  return basis;
596 }

598 // See go/mathopt-dev-transformations#gurobi-inf for details of this
599 // transformation, comments inside the function refer to the notation there.
600 absl::StatusOr<DualRayProto> GurobiSolver::GetGurobiDualRay(
601  const SparseVectorFilterProto& linear_constraints_filter,
602  const SparseVectorFilterProto& variables_filter, const bool is_maximize) {
603  // farkas_dual = lambda
604  std::vector<double> farkas_dual(num_gurobi_constraints());
605  RETURN_IF_ERROR(
606  GetDoubleAttrArray(GRB_DBL_ATTR_FARKASDUAL, absl::MakeSpan(farkas_dual)));
607  DualRayProto dual_ray;
608 
609  // Compute y = -lambda
610  {
611  SparseVectorFilterPredicate predicate(linear_constraints_filter);
612  for (auto [constraint_id, gurobi_data] : linear_constraints_map_) {
613  // constraint_dual_value = y[constraint_id]
614  const double value = -farkas_dual[gurobi_data.constraint_index];
615  if (predicate.AcceptsAndUpdate(constraint_id, value)) {
616  dual_ray.mutable_dual_values()->add_ids(constraint_id);
617  if (is_maximize) {
618  dual_ray.mutable_dual_values()->add_values(-value);
619  } else {
620  dual_ray.mutable_dual_values()->add_values(value);
621  }
622  }
623  }
624  }
625 
626  // Compute r = \bar{a} = A^T lambda
627  std::vector<int> constraint_indices(num_gurobi_constraints());
628  std::vector<double> coefficients(num_gurobi_constraints());
629  {
630  SparseVectorFilterPredicate predicate(variables_filter);
631  for (auto [var_id, gurobi_variable_index] : variables_map_) {
632  // reduced_cost_value = r[gurobi_variable_index]
633  // = \bar{a}[gurobi_variable_index]
634  double reduced_cost_value = 0.0;
635  int result_size;
636  int vbeg;
637  // coefficients = column gurobi_variable_index of A
638  RETURN_IF_GUROBI_ERROR(GRBgetvars(
639  gurobi_model_, &result_size, &vbeg, constraint_indices.data(),
640  coefficients.data(), gurobi_variable_index, 1));
641  for (int i = 0; i < result_size; ++i) {
642  reduced_cost_value +=
643  farkas_dual[constraint_indices[i]] * coefficients[i];
644  }
645  if (predicate.AcceptsAndUpdate(var_id, reduced_cost_value)) {
646  dual_ray.mutable_reduced_costs()->add_ids(var_id);
647  if (is_maximize) {
648  dual_ray.mutable_reduced_costs()->add_values(-reduced_cost_value);
649  } else {
650  dual_ray.mutable_reduced_costs()->add_values(reduced_cost_value);
651  }
652  }
653  }
654  }
655  return dual_ray;
656 }

658 absl::Status GurobiSolver::ExtractSolveResultProto(
659  const bool is_maximize, SolveResultProto& result,
660  const ModelSolveParametersProto& model_parameters) {
661  int num_solutions = 0;
662  if (GRBisattravailable(gurobi_model_, GRB_INT_ATTR_SOLCOUNT)) {
663  ASSIGN_OR_RETURN(num_solutions, GetIntAttr(GRB_INT_ATTR_SOLCOUNT));
664  }
665  if (GRBisattravailable(gurobi_model_, GRB_INT_ATTR_STATUS)) {
666  ASSIGN_OR_RETURN(const int grb_termination,
667  GetIntAttr(GRB_INT_ATTR_STATUS));
668 
669  const bool has_feasible_solution = num_solutions > 0;
670 
671  ASSIGN_OR_RETURN(
672  const auto reason_and_detail,
673  ConvertTerminationReason(grb_termination, has_feasible_solution));
674  result.set_termination_reason(reason_and_detail.first);
675  result.set_termination_detail(reason_and_detail.second);
676  }
677  if (GRBisattravailable(gurobi_model_, GRB_DBL_ATTR_OBJVAL)) {
678  ASSIGN_OR_RETURN(const double obj_val, GetDoubleAttr(GRB_DBL_ATTR_OBJVAL));
679  result.mutable_solve_stats()->set_best_primal_bound(obj_val);
680  } else {
681  result.mutable_solve_stats()->set_best_primal_bound(is_maximize ? -kInf
682  : kInf);
683  }
684  if (GRBisattravailable(gurobi_model_, GRB_DBL_ATTR_OBJBOUND)) {
685  ASSIGN_OR_RETURN(const double obj_bound,
686  GetDoubleAttr(GRB_DBL_ATTR_OBJBOUND));
687  result.mutable_solve_stats()->set_best_dual_bound(obj_bound);
688  } else {
689  result.mutable_solve_stats()->set_best_dual_bound(is_maximize ? kInf
690  : -kInf);
691  }
692  if (GRBisattravailable(gurobi_model_, GRB_DBL_ATTR_ITERCOUNT)) {
693  ASSIGN_OR_RETURN(const double simplex_iters_double,
694  GetDoubleAttr(GRB_DBL_ATTR_ITERCOUNT));
695  ASSIGN_OR_RETURN(const int64_t simplex_iters,
696  SafeInt64FromDouble(simplex_iters_double));
697  result.mutable_solve_stats()->set_simplex_iterations(simplex_iters);
698  }
699  if (GRBisattravailable(gurobi_model_, GRB_INT_ATTR_BARITERCOUNT)) {
700  ASSIGN_OR_RETURN(const int barrier_iters,
701  GetIntAttr(GRB_INT_ATTR_BARITERCOUNT));
702  result.mutable_solve_stats()->set_barrier_iterations(barrier_iters);
703  }
704  if (GRBisattravailable(gurobi_model_, GRB_DBL_ATTR_NODECOUNT)) {
705  ASSIGN_OR_RETURN(const double nodes_double,
706  GetDoubleAttr(GRB_DBL_ATTR_NODECOUNT));
707  ASSIGN_OR_RETURN(const int64_t nodes, SafeInt64FromDouble(nodes_double));
708  result.mutable_solve_stats()->set_node_count(nodes);
709  }
710 
711  for (int i = 0; i < num_solutions; ++i) {
712  PrimalSolutionProto* const primal_solution = result.add_primal_solutions();
713  double sol_val;
714  std::vector<double> grb_var_values(num_gurobi_variables_);
715  // TODO(user): there seems to be some kind of issue with Gurobi where
716  // GRB_DBL_ATTR_POOLOBJVAL is not always filled in when when there is
717  // one, solution, probably a bug.
718  if (i == 0) {
719  ASSIGN_OR_RETURN(sol_val, GetDoubleAttr(GRB_DBL_ATTR_OBJVAL));
720  RETURN_IF_ERROR(
721  GetDoubleAttrArray(GRB_DBL_ATTR_X, absl::MakeSpan(grb_var_values)));
722  } else {
723  RETURN_IF_GUROBI_ERROR(
724  GRBsetintparam(active_env_, GRB_INT_PAR_SOLUTIONNUMBER, i))
725  << "Error setting solution to " << i;
726  ASSIGN_OR_RETURN(sol_val, GetDoubleAttr(GRB_DBL_ATTR_POOLOBJVAL));
727  RETURN_IF_ERROR(
728  GetDoubleAttrArray(GRB_DBL_ATTR_XN, absl::MakeSpan(grb_var_values)));
729  }
730  primal_solution->set_objective_value(sol_val);
731  GurobiVectorToSparseDoubleVector(
732  grb_var_values, variables_map_,
733  *primal_solution->mutable_variable_values(),
734  model_parameters.primal_variables_filter());
735  }
736  // TODO(user): support getting infeasibility proofs, and basis.
737  // If model is LP, get dual solutions
738  // Note that we can ignore the reduced costs of the slack variables for ranged
739  // constraints because of
740  // go/mathopt-dev-transformations#slack-var-range-constraint
741  ASSIGN_OR_RETURN(const bool is_lp, IsLP());
742  if (is_lp && GRBisattravailable(gurobi_model_, GRB_DBL_ATTR_PI) &&
743  GRBisattravailable(gurobi_model_, GRB_DBL_ATTR_RC)) {
744  std::vector<double> grb_constraint_duals(num_gurobi_constraints());
745  RETURN_IF_ERROR(GetDoubleAttrArray(GRB_DBL_ATTR_PI,
746  absl::MakeSpan(grb_constraint_duals)));
747  DualSolutionProto* const dual_solution = result.add_dual_solutions();
748  ASSIGN_OR_RETURN(const double obj_val, GetDoubleAttr(GRB_DBL_ATTR_OBJVAL));
749  dual_solution->set_objective_value(obj_val);
750  GurobiVectorToSparseDoubleVector(
751  grb_constraint_duals, linear_constraints_map_,
752  *dual_solution->mutable_dual_values(),
753  model_parameters.dual_linear_constraints_filter());
754 
755  std::vector<double> grb_reduced_cost_values(num_gurobi_variables_);
756  RETURN_IF_ERROR(GetDoubleAttrArray(
757  GRB_DBL_ATTR_RC, absl::MakeSpan(grb_reduced_cost_values)));
758  GurobiVectorToSparseDoubleVector(grb_reduced_cost_values, variables_map_,
759  *dual_solution->mutable_reduced_costs(),
760  model_parameters.dual_variables_filter());
761  }
762  if (is_lp && GRBisattravailable(gurobi_model_, GRB_DBL_ATTR_UNBDRAY)) {
763  std::vector<double> grb_ray_var_values(num_gurobi_variables_);
764  RETURN_IF_ERROR(GetDoubleAttrArray(GRB_DBL_ATTR_UNBDRAY,
765  absl::MakeSpan(grb_ray_var_values)));
766  PrimalRayProto* const primal_ray = result.add_primal_rays();
767  GurobiVectorToSparseDoubleVector(
768  grb_ray_var_values, variables_map_,
769  *primal_ray->mutable_variable_values(),
770  model_parameters.primal_variables_filter());
771  }
772  if (is_lp && GRBisattravailable(gurobi_model_, GRB_DBL_ATTR_FARKASDUAL)) {
773  ASSIGN_OR_RETURN(
774  DualRayProto dual_ray,
775  GetGurobiDualRay(model_parameters.dual_linear_constraints_filter(),
776  model_parameters.dual_variables_filter(),
777  is_maximize));
778  result.mutable_dual_rays()->Add(std::move(dual_ray));
779  }
780  if (is_lp && GRBisattravailable(gurobi_model_, GRB_INT_ATTR_VBASIS) &&
781  GRBisattravailable(gurobi_model_, GRB_INT_ATTR_CBASIS)) {
782  ASSIGN_OR_RETURN(BasisProto basis, GetGurobiBasis());
783  result.mutable_basis()->Add(std::move(basis));
784  }
785  return absl::OkStatus();
786 }

854 // Given a vector of pairs<LinearConstraintId,ConstraintData&> add a slack
855 // variable for each of the constraints in the underlying `gurobi_model_`
856 // using the referenced bounds.
857 absl::Status GurobiSolver::AddNewSlacks(
858  const std::vector<SlackInfo>& new_slacks) {
859  // Note that we are really adding the sub-matrix
860  // D * slack
861  // to the set of linear constraints, and the D matrix is stored in compressed
862  // sparse column (CSC) format. In our particular case, D is a diagonal matrix
863  // with -1.0 coefficients for each new slack in the row indicated in the
864  // row_indices vector.
865  const int num_slacks = new_slacks.size();
866  if (num_slacks == 0) {
867  return absl::OkStatus();
868  }
869  // Build the D matrix in CSC format.
870  std::vector<double> column_non_zeros(num_slacks, -1.0);
871  std::vector<double> lower_bounds;
872  std::vector<double> upper_bounds;
873  std::vector<GurobiLinearConstraintIndex> row_indices;
874  std::vector<int> column_non_zero_begin;
875  column_non_zero_begin.reserve(num_slacks);
876  row_indices.reserve(num_slacks);
877  lower_bounds.reserve(num_slacks);
878  upper_bounds.reserve(num_slacks);
879  for (int k = 0; k < num_slacks; ++k) {
880  auto& [id, constraint_data] = new_slacks[k];
881  gtl::InsertOrDie(&slack_map_, id, constraint_data);
882  row_indices.emplace_back(constraint_data.constraint_index);
883  lower_bounds.emplace_back(constraint_data.lower_bound);
884  upper_bounds.emplace_back(constraint_data.upper_bound);
885  column_non_zero_begin.emplace_back(k);
886  }
887  // Add variables to the underlying model.
888  RETURN_IF_GUROBI_ERROR(GRBaddvars(
889  /*model=*/gurobi_model_, /*numvars=*/num_slacks, /*numnz=*/num_slacks,
890  /*vbeg=*/column_non_zero_begin.data(), /*vind=*/row_indices.data(),
891  /*vval=*/column_non_zeros.data(), /*obj=*/nullptr,
892  /*lb=*/lower_bounds.data(), /*ub=*/upper_bounds.data(),
893  /*vtype=*/nullptr, /*varnames=*/nullptr));
894  num_gurobi_variables_ += num_slacks;
895  return absl::OkStatus();
896 }

898 absl::Status GurobiSolver::AddNewConstraints(
899  const LinearConstraintsProto& constraints) {
900  CHECK(gurobi_model_ != nullptr);
901  const int num_model_constraints = num_gurobi_constraints();
902  const int num_new_constraints = constraints.lower_bounds().size();
903 
904  const bool has_constraint_names = !constraints.names().empty();
905  std::vector<char*> constraint_names;
906  if (has_constraint_names) {
907  constraint_names.reserve(num_new_constraints);
908  }
909  // Constraints are translated into:
910  // 1. ax <= upper_bound (if lower bound <= -GRB_INFINITY, and upper_bound
911  // is finite and less than GRB_INFINITY)
912  // 2. ax >= lower_bound (if upper bound >= GRB_INFINITY, and lower_bound is
913  // finite and greater than -GRB_INFINITY)
914  // 3. ax == xxxxx_bound (if both bounds are finite, equal, and their
915  // absolute values less than GRB_INFINITY)
916  // 4. ax - slack = 0.0 (otherwise,
917  // slack bounds == [lower_bound, upper_bound])
918  std::vector<double> constraint_rhs;
919  std::vector<char> constraint_sense;
920  std::vector<SlackInfo> new_slacks;
921  constraint_rhs.reserve(num_new_constraints);
922  constraint_sense.reserve(num_new_constraints);
923  new_slacks.reserve(num_new_constraints);
924  for (int i = 0; i < num_new_constraints; ++i) {
925  const int64_t id = constraints.ids(i);
926  ConstraintData& constraint_data =
927  gtl::InsertKeyOrDie(&linear_constraints_map_, id);
928  constraint_data.lower_bound = constraints.lower_bounds(i);
929  constraint_data.upper_bound = constraints.upper_bounds(i);
930  constraint_data.constraint_index = i + num_model_constraints;
931  char sense = GRB_EQUAL;
932  double rhs = 0.0;
933  // Detect the type of constraint to add and store RHS and bounds.
934  if (constraint_data.lower_bound <= -GRB_INFINITY &&
935  constraint_data.upper_bound < GRB_INFINITY) {
936  rhs = constraint_data.upper_bound;
937  sense = GRB_LESS_EQUAL;
938  } else if (constraint_data.lower_bound > -GRB_INFINITY &&
939  constraint_data.upper_bound >= GRB_INFINITY) {
940  rhs = constraint_data.lower_bound;
941  sense = GRB_GREATER_EQUAL;
942  } else if (constraint_data.lower_bound == constraint_data.upper_bound) {
943  rhs = constraint_data.lower_bound;
944  sense = GRB_EQUAL;
945  } else {
946  // Note that constraints where the lower bound and the upper bound are
947  // -+infinity translate into a range constraint with an unbounded slack.
948  constraint_data.slack_index = new_slacks.size() + num_gurobi_variables_;
949  new_slacks.emplace_back(id, constraint_data);
950  }
951  constraint_rhs.emplace_back(rhs);
952  constraint_sense.emplace_back(sense);
953  if (has_constraint_names) {
954  constraint_names.emplace_back(
955  const_cast<char*>(constraints.names(i).c_str()));
956  }
957  }
958  // Add all constraints in one call.
959  RETURN_IF_GUROBI_ERROR(GRBaddconstrs(
960  gurobi_model_, /*numconstrs=*/num_new_constraints,
961  /*numnz=*/0, /*cbeg=*/nullptr, /*cind=*/nullptr, /*cval=*/nullptr,
962  /*sense=*/constraint_sense.data(), /*rhs=*/constraint_rhs.data(),
963  /*constrnames=*/
964  has_constraint_names ? constraint_names.data() : nullptr));
965 
966  // Add slacks for true ranged constraints (if needed)
967  if (!new_slacks.empty()) {
968  RETURN_IF_ERROR(AddNewSlacks(new_slacks));
969  }
970  return absl::OkStatus();
971 }

1031  CHECK(gurobi_model_ != nullptr);

1053 // Bound changes in constraints can induce new variables, and also remove
1054 // some slacks. We first add all new variables, and queue all deletions to be
1055 // dealt with later on.
1056 absl::Status GurobiSolver::UpdateLinearConstraints(
1057  const LinearConstraintUpdatesProto& constraints_update,
1058  std::vector<GurobiVariableIndex>& deleted_variables_index) {
1059  const SparseDoubleVectorProto& constraint_lower_bounds =
1060  constraints_update.lower_bounds();
1061  const SparseDoubleVectorProto& constraint_upper_bounds =
1062  constraints_update.upper_bounds();
1063 
1064  // If no update, just return.
1065  if (constraint_lower_bounds.ids().empty() &&
1066  constraint_upper_bounds.ids().empty()) {
1067  return absl::OkStatus();
1068  }
1069 
1070  // We want to avoid changing the right-hand-side, sense, or slacks of each
1071  // constraint more than once. Since we can refer to the same constraint ID
1072  // both in the `constraint_upper_bounds` and `constraint_lower_bounds` sparse
1073  // vectors, we collect all changes into a single structure:
1074  struct UpdateConstraintData {
1075  LinearConstraintId constraint_id;
1076  ConstraintData& source;
1077  double new_lower_bound;
1078  double new_upper_bound;
1079  UpdateConstraintData(const LinearConstraintId id, ConstraintData& reference)
1080  : constraint_id(id),
1081  source(reference),
1082  new_lower_bound(reference.lower_bound),
1083  new_upper_bound(reference.upper_bound) {}
1084  };
1085  const int upper_bounds_size = constraint_upper_bounds.ids().size();
1086  const int lower_bounds_size = constraint_lower_bounds.ids().size();
1087  std::vector<UpdateConstraintData> update_vector;
1088  update_vector.reserve(upper_bounds_size + lower_bounds_size);
1089  // We exploit the fact that IDs are sorted in increasing order to merge
1090  // changes into a vector of aggregated changes.
1091  for (int lower_index = 0, upper_index = 0;
1092  lower_index < lower_bounds_size || upper_index < upper_bounds_size;) {
1093  VariableId lower_id = std::numeric_limits<int64_t>::max();
1094  if (lower_index < lower_bounds_size) {
1095  lower_id = constraint_lower_bounds.ids(lower_index);
1096  }
1097  VariableId upper_id = std::numeric_limits<int64_t>::max();
1098  if (upper_index < upper_bounds_size) {
1099  upper_id = constraint_upper_bounds.ids(upper_index);
1100  }
1101  const VariableId id = std::min(lower_id, upper_id);
1102  DCHECK(id < std::numeric_limits<int64_t>::max());
1103  UpdateConstraintData update(id, linear_constraints_map_.at(id));
1104  if (lower_id == upper_id) {
1105  update.new_lower_bound = constraint_lower_bounds.values(lower_index++);
1106  update.new_upper_bound = constraint_upper_bounds.values(upper_index++);
1107  } else if (lower_id < upper_id) {
1108  update.new_lower_bound = constraint_lower_bounds.values(lower_index++);
1109  } else { /* upper_id < lower_id */
1110  update.new_upper_bound = constraint_upper_bounds.values(upper_index++);
1111  }
1112  update_vector.emplace_back(update);
1113  }
1114 
1115  // We have grouped all changes in update_vector, now generate changes in
1116  // slack bounds, rhs, senses, new slacks, and deleted_slacks (to be dealt
1117  // with later, outside this function).
1118  // These three vectors keep changes to right-hand-side and senses.
1119  std::vector<char> sense_data;
1120  std::vector<double> rhs_data;
1121  std::vector<GurobiLinearConstraintIndex> rhs_index;
1122  // These three vectors keep changes to bounds on existing slack.
1123  std::vector<double> lower_bound_data;
1124  std::vector<double> upper_bound_data;
1125  std::vector<GurobiVariableIndex> bound_index;
1126  // This vector keep newly introduced slacks.
1127  std::vector<SlackInfo> new_slacks;
1128  // Iterate on the changes, and populate the three possible changes.
1129  for (UpdateConstraintData& update_data : update_vector) {
1130  const bool same_lower_bound =
1131  (update_data.source.lower_bound == update_data.new_lower_bound) ||
1132  ((update_data.source.lower_bound <= -GRB_INFINITY) &&
1133  (update_data.new_lower_bound <= -GRB_INFINITY));
1134  const bool same_upper_bound =
1135  (update_data.source.upper_bound == update_data.new_upper_bound) ||
1136  ((update_data.source.upper_bound >= GRB_INFINITY) &&
1137  (update_data.new_upper_bound >= GRB_INFINITY));
1138  if (same_upper_bound && same_lower_bound) continue;
1139  // Save into linear_constraints_map_[id] the new bounds for the linear
1140  // constraint.
1141  update_data.source.lower_bound = update_data.new_lower_bound;
1142  update_data.source.upper_bound = update_data.new_upper_bound;
1143  bool delete_slack = false;
1144  // Detect the type of constraint to add and store RHS and bounds.
1145  if (update_data.new_lower_bound <= -GRB_INFINITY &&
1146  update_data.new_upper_bound < GRB_INFINITY) {
1147  delete_slack = true;
1148  rhs_index.emplace_back(update_data.source.constraint_index);
1149  rhs_data.emplace_back(update_data.new_upper_bound);
1150  sense_data.emplace_back(GRB_LESS_EQUAL);
1151  } else if (update_data.new_lower_bound > -GRB_INFINITY &&
1152  update_data.new_upper_bound >= GRB_INFINITY) {
1153  delete_slack = true;
1154  rhs_index.emplace_back(update_data.source.constraint_index);
1155  rhs_data.emplace_back(update_data.new_lower_bound);
1156  sense_data.emplace_back(GRB_GREATER_EQUAL);
1157  } else if (update_data.new_lower_bound == update_data.new_upper_bound) {
1158  delete_slack = true;
1159  rhs_index.emplace_back(update_data.source.constraint_index);
1160  rhs_data.emplace_back(update_data.new_lower_bound);
1161  sense_data.emplace_back(GRB_EQUAL);
1162  } else {
1163  // Note that constraints where the lower bound and the upper bound are
1164  // -+infinity translated into a range constraint with an unbounded
1165  // slack.
1166  if (update_data.source.slack_index != kUnspecifiedIndex) {
1167  bound_index.emplace_back(update_data.source.slack_index);
1168  lower_bound_data.emplace_back(update_data.new_lower_bound);
1169  upper_bound_data.emplace_back(update_data.new_upper_bound);
1170  } else {
1171  // Note that if we add a new slack, we must both reset the sense and
1172  // right hand side for the inequality.
1173  rhs_index.emplace_back(update_data.source.constraint_index);
1174  rhs_data.emplace_back(0.0);
1175  sense_data.emplace_back(GRB_EQUAL);
1176  // Update the slack_index in the linear_constraints_map_[id]
1177  update_data.source.slack_index =
1178  new_slacks.size() + num_gurobi_variables_;
1179  // Save the data needed to add the new slack.
1180  new_slacks.emplace_back(update_data.constraint_id, update_data.source);
1181  }
1182  }
1183  // If the constraint had a slack, and now is marked for deletion, we reset
1184  // the stored slack_index in linear_constraints_map_[id] and save the index
1185  // in the list of variables to be deleted later on.
1186  if (delete_slack && update_data.source.slack_index != kUnspecifiedIndex) {
1187  deleted_variables_index.emplace_back(update_data.source.slack_index);
1188  update_data.source.slack_index = kUnspecifiedIndex;
1189  }
1190  }
1191 
1192  // Pass down changes to Gurobi.
1193  if (!rhs_index.empty()) {
1194  RETURN_IF_GUROBI_ERROR(GRBsetdblattrlist(
1195  /*model=*/gurobi_model_, /*attrname=*/GRB_DBL_ATTR_RHS,
1196  /*len=*/rhs_index.size(), /*ind=*/rhs_index.data(),
1197  /*newvalues=*/rhs_data.data()));
1198  RETURN_IF_GUROBI_ERROR(GRBsetcharattrlist(
1199  /*model=*/gurobi_model_, /*attrname=*/GRB_CHAR_ATTR_SENSE,
1200  /*len=*/rhs_index.size(), /*ind=*/rhs_index.data(),
1201  /*newvalues=*/sense_data.data()));
1202  } // rhs changes
1203  if (!bound_index.empty()) {
1204  RETURN_IF_GUROBI_ERROR(GRBsetdblattrlist(
1205  /*model=*/gurobi_model_, /*attrname=*/GRB_DBL_ATTR_LB,
1206  /*len=*/bound_index.size(), /*ind=*/bound_index.data(),
1207  /*newvalues=*/lower_bound_data.data()));
1208  RETURN_IF_GUROBI_ERROR(GRBsetdblattrlist(
1209  /*model=*/gurobi_model_, /*attrname=*/GRB_DBL_ATTR_UB,
1210  /*len=*/bound_index.size(), /*ind=*/bound_index.data(),
1211  /*newvalues=*/upper_bound_data.data()));
1212  } // Slack bound changes.
1213 
1214  if (!new_slacks.empty()) {
1215  RETURN_IF_ERROR(AddNewSlacks(new_slacks));
1216  }
1217  return absl::OkStatus();
1218 }

1220 // This function re-assign indices for variables and constraints after
1221 // deletion. The updated indices are computed from the previous indices, sorted
1222 // in incremental form, but re-assigned so that all indices are contiguous
1223 // between [0, num_variables-1] and [0, num_linear_constraints-1].
1224 // This implementation exploit the fact that gtl::linked_hash_map preserves the
1225 // insertion order of whatever elements remain in the hash tables.
1226 absl::Status GurobiSolver::UpdateGurobiIndices() {
1227  { // Recover index of variables.
1228  GurobiVariableIndex next_index = 0;
1229  GurobiVariableIndex prev_index = kUnspecifiedIndex;
1230  auto variable_it = variables_map_.begin();
1231  auto slack_it = slack_map_.begin();
1232  while (variable_it != variables_map_.end() ||
1233  slack_it != slack_map_.end()) {
1234  GurobiVariableIndex variable_index = std::numeric_limits<int32_t>::max();
1235  if (variable_it != variables_map_.end()) {
1236  variable_index = variable_it->second;
1237  }
1238  GurobiVariableIndex slack_index = std::numeric_limits<int32_t>::max();
1239  if (slack_it != slack_map_.end()) {
1240  slack_index = slack_it->second.slack_index;
1241  }
1242  DCHECK_LT(prev_index, variable_index);
1243  DCHECK_LT(prev_index, slack_index);
1244  DCHECK_NE(variable_index, slack_index);
1245  if (slack_index < variable_index) {
1246  prev_index = slack_index;
1247  slack_it->second.slack_index = next_index++;
1248  ++slack_it;
1249  } else {
1250  prev_index = variable_index;
1251  variable_it->second = next_index++;
1252  ++variable_it;
1253  }
1254  }
1255  DCHECK_EQ(next_index, num_gurobi_variables_);
1256  }
1257  { // Recover index of constraints.
1258  GurobiLinearConstraintIndex next_constraint = 0;
1259  GurobiLinearConstraintIndex prev_constraint = kUnspecifiedConstraint;
1260  for (auto& constraint_iterator : linear_constraints_map_) {
1261  DCHECK_LT(prev_constraint, constraint_iterator.second.constraint_index);
1262  prev_constraint = constraint_iterator.second.constraint_index;
1263  constraint_iterator.second.constraint_index = next_constraint++;
1264  }
1265  DCHECK_EQ(next_constraint, num_gurobi_constraints());
1266  }
1267  return absl::OkStatus();
1268 }
