40 // Computes the 1-norm of the matrix.
41 // The 1-norm |A| is defined as max_j sum_i |a_ij| or
42 // max_col sum_row |a(row,col)|.
43 template <typename Matrix>
44 Fractional ComputeOneNormTemplate(const Matrix& matrix) {
45  Fractional norm(0.0);
46  const ColIndex num_cols(matrix.num_cols());
47  for (ColIndex col(0); col < num_cols; ++col) {
48  Fractional column_norm(0);
49  for (const SparseColumn::Entry e : matrix.column(col)) {
50  // Compute sum_i |a_ij|.
51  column_norm += fabs(e.coefficient());
52  }
53  // Compute max_j sum_i |a_ij|
54  norm = std::max(norm, column_norm);
55  }
56  return norm;
57 }
58 
59 // Computes the oo-norm (infinity-norm) of the matrix.
60 // The oo-norm |A| is defined as max_i sum_j |a_ij| or
61 // max_row sum_col |a(row,col)|.
62 template <typename Matrix>
63 Fractional ComputeInfinityNormTemplate(const Matrix& matrix) {
64  DenseColumn row_sum(matrix.num_rows(), 0.0);
65  const ColIndex num_cols(matrix.num_cols());
66  for (ColIndex col(0); col < num_cols; ++col) {
67  for (const SparseColumn::Entry e : matrix.column(col)) {
68  // Compute sum_j |a_ij|.
69  row_sum[e.row()] += fabs(e.coefficient());
70  }
71  }
72 
73  // Compute max_i sum_j |a_ij|
74  Fractional norm = 0.0;
75  const RowIndex num_rows(matrix.num_rows());
76  for (RowIndex row(0); row < num_rows; ++row) {
77  norm = std::max(norm, row_sum[row]);
78  }
79  return norm;
80 }

180 template <typename Matrix>
181 void SparseMatrix::PopulateFromTranspose(const Matrix& input) {
182  Reset(RowToColIndex(input.num_rows()), ColToRowIndex(input.num_cols()));
183 
184  // We do a first pass on the input matrix to resize the new columns properly.
185  StrictITIVector<RowIndex, EntryIndex> row_degree(input.num_rows(),
186  EntryIndex(0));
187  for (ColIndex col(0); col < input.num_cols(); ++col) {
188  for (const SparseColumn::Entry e : input.column(col)) {
189  ++row_degree[e.row()];
190  }
191  }
192  for (RowIndex row(0); row < input.num_rows(); ++row) {
193  columns_[RowToColIndex(row)].Reserve(row_degree[row]);
194  }
195 
196  for (ColIndex col(0); col < input.num_cols(); ++col) {
197  const RowIndex transposed_row = ColToRowIndex(col);
198  for (const SparseColumn::Entry e : input.column(col)) {
199  const ColIndex transposed_col = RowToColIndex(e.row());
200  columns_[transposed_col].SetCoefficient(transposed_row, e.coefficient());
201  }
202  }
203  DCHECK(IsCleanedUp());
204 }

226  const SparseMatrix& a,
227  Fractional beta,
228  const SparseMatrix& b) {
229  DCHECK_EQ(a.num_cols(), b.num_cols());
230  DCHECK_EQ(a.num_rows(), b.num_rows());
231 
232  const ColIndex num_cols = a.num_cols();
233  Reset(num_cols, a.num_rows());
234 
235  const RowIndex num_rows = a.num_rows();
236  RandomAccessSparseColumn dense_column(num_rows);
237  for (ColIndex col(0); col < num_cols; ++col) {
238  for (const SparseColumn::Entry e : a.columns_[col]) {
239  dense_column.AddToCoefficient(e.row(), alpha * e.coefficient());
240  }
241  for (const SparseColumn::Entry e : b.columns_[col]) {
242  dense_column.AddToCoefficient(e.row(), beta * e.coefficient());
243  }
244  dense_column.PopulateSparseColumn(&columns_[col]);
245  columns_[col].CleanUp();
246  dense_column.Clear();

250 void SparseMatrix::PopulateFromProduct(const SparseMatrix& a,
251  const SparseMatrix& b) {
252  const ColIndex num_cols = b.num_cols();
253  const RowIndex num_rows = a.num_rows();
254  Reset(num_cols, num_rows);
255 
256  RandomAccessSparseColumn tmp_column(num_rows);
257  for (ColIndex col_b(0); col_b < num_cols; ++col_b) {
258  for (const SparseColumn::Entry eb : b.columns_[col_b]) {
259  if (eb.coefficient() == 0.0) {
260  continue;
261  }
262  const ColIndex col_a = RowToColIndex(eb.row());
263  for (const SparseColumn::Entry ea : a.columns_[col_a]) {
264  const Fractional value = ea.coefficient() * eb.coefficient();
265  tmp_column.AddToCoefficient(ea.row(), value);
266  }
267  }
268 
269  // Populate column col_b.
270  tmp_column.PopulateSparseColumn(&columns_[col_b]);
271  columns_[col_b].CleanUp();
272  tmp_column.Clear();
273  }

327 bool SparseMatrix::Equals(const SparseMatrix& a, Fractional tolerance) const {
328  if (num_cols() != a.num_cols() || num_rows() != a.num_rows()) {
329  return false;
330  }
331 
332  RandomAccessSparseColumn dense_column(num_rows());
333  RandomAccessSparseColumn dense_column_a(num_rows());
334  const ColIndex num_cols = a.num_cols();
335  for (ColIndex col(0); col < num_cols; ++col) {
336  // Store all entries of current matrix in a dense column.
337  for (const SparseColumn::Entry e : columns_[col]) {
338  dense_column.AddToCoefficient(e.row(), e.coefficient());
339  }
340 
341  // Check all entries of a are those stored in the dense column.
342  for (const SparseColumn::Entry e : a.columns_[col]) {
343  if (fabs(e.coefficient() - dense_column.GetCoefficient(e.row())) >
344  tolerance) {
345  return false;
346  }
347  }
348 
349  // Store all entries of matrix a in a dense column.
350  for (const SparseColumn::Entry e : a.columns_[col]) {
351  dense_column_a.AddToCoefficient(e.row(), e.coefficient());
352  }
353 
354  // Check all entries are those stored in the dense column a.
355  for (const SparseColumn::Entry e : columns_[col]) {
356  if (fabs(e.coefficient() - dense_column_a.GetCoefficient(e.row())) >
357  tolerance) {
358  return false;
359  }
360  }
361 
362  dense_column.Clear();
363  dense_column_a.Clear();
364  }
365 
366  return true;
367 }

369 void SparseMatrix::ComputeMinAndMaxMagnitudes(Fractional* min_magnitude,
370  Fractional* max_magnitude) const {
371  RETURN_IF_NULL(min_magnitude);
372  RETURN_IF_NULL(max_magnitude);
373  *min_magnitude = kInfinity;
374  *max_magnitude = 0.0;
375  for (ColIndex col(0); col < num_cols(); ++col) {
376  for (const SparseColumn::Entry e : columns_[col]) {
377  const Fractional magnitude = fabs(e.coefficient());
378  if (magnitude != 0.0) {
379  *min_magnitude = std::min(*min_magnitude, magnitude);
380  *max_magnitude = std::max(*max_magnitude, magnitude);
381  }
382  }
383  }
384  if (*max_magnitude == 0.0) {
385  *min_magnitude = 0.0;
386  }
387 }

437 void CompactSparseMatrix::PopulateFromMatrixView(const MatrixView& input) {
438  num_cols_ = input.num_cols();
439  num_rows_ = input.num_rows();
440  const EntryIndex num_entries = input.num_entries();
441  starts_.assign(num_cols_ + 1, EntryIndex(0));

446  starts_[col] = index;
447  for (const SparseColumn::Entry e : input.column(col)) {

450  ++index;
451  }
452  }

456 void CompactSparseMatrix::PopulateFromTranspose(
457  const CompactSparseMatrix& input) {
458  num_cols_ = RowToColIndex(input.num_rows());
459  num_rows_ = ColToRowIndex(input.num_cols());
460 
461  // Fill the starts_ vector by computing the number of entries of each rows and
462  // then doing a cummulative sum. After this step starts_[col + 1] will be the
463  // actual start of the column col when we are done.
464  starts_.assign(num_cols_ + 2, EntryIndex(0));
465  for (const RowIndex row : input.rows_) {
466  ++starts_[RowToColIndex(row) + 2];
467  }
468  for (ColIndex col(2); col < starts_.size(); ++col) {
469  starts_[col] += starts_[col - 1];
470  }
471  coefficients_.resize(starts_.back(), 0.0);
472  rows_.resize(starts_.back(), kInvalidRow);
473  starts_.pop_back();
474 
475  // Use starts_ to fill the matrix. Note that starts_ is modified so that at
476  // the end it has its final values.
477  for (ColIndex col(0); col < input.num_cols(); ++col) {
478  const RowIndex transposed_row = ColToRowIndex(col);
479  for (const EntryIndex i : input.Column(col)) {
480  const ColIndex transposed_col = RowToColIndex(input.EntryRow(i));
481  const EntryIndex index = starts_[transposed_col + 1]++;
482  coefficients_[index] = input.EntryCoefficient(i);
483  rows_[index] = transposed_row;
484  }
485  }
486 
487  DCHECK_EQ(starts_.front(), 0);
488  DCHECK_EQ(starts_.back(), rows_.size());
489 }

491 void TriangularMatrix::PopulateFromTranspose(const TriangularMatrix& input) {
492  CompactSparseMatrix::PopulateFromTranspose(input);
493 
494  // This takes care of the triangular special case.
495  diagonal_coefficients_ = input.diagonal_coefficients_;
496  all_diagonal_coefficients_are_one_ = input.all_diagonal_coefficients_are_one_;
497 
498  // The elimination structure of the transpose is not the same.
499  pruned_ends_.resize(num_cols_, EntryIndex(0));
500  for (ColIndex col(0); col < num_cols_; ++col) {
501  pruned_ends_[col] = starts_[col + 1];
502  }
503 
504  // Compute first_non_identity_column_. Note that this is not necessarily the
505  // same as input.first_non_identity_column_ for an upper triangular matrix.
506  first_non_identity_column_ = 0;
507  const ColIndex end = diagonal_coefficients_.size();
508  while (first_non_identity_column_ < end &&
509  ColumnNumEntries(first_non_identity_column_) == 0 &&
510  diagonal_coefficients_[first_non_identity_column_] == 1.0) {
511  ++first_non_identity_column_;
512  }

611 // Internal function used to finish adding one column to a triangular matrix.
612 // This sets the diagonal coefficient to the given value, and prepares the
613 // matrix for the next column addition.
614 void TriangularMatrix::CloseCurrentColumn(Fractional diagonal_value) {
615  DCHECK_NE(diagonal_value, 0.0);
616  // The vectors diagonal_coefficients, pruned_ends, and starts_ should have all
617  // been preallocated by a call to SetTotalNumberOfColumns().
618  DCHECK_LT(num_cols_, diagonal_coefficients_.size());
619  diagonal_coefficients_[num_cols_] = diagonal_value;
620 
621  // TODO(user): This is currently not used by all matrices. It will be good
622  // to fill it only when needed.

999  const SparseColumn& rhs, const RowPermutation& row_perm,
1000  const RowMapping& partial_inverse_row_perm, SparseColumn* lower,
1001  SparseColumn* upper) const {
1002  DCHECK(all_diagonal_coefficients_are_one_);

1004  RETURN_IF_NULL(upper);
1005 
1006  initially_all_zero_scratchpad_.resize(num_rows_, 0.0);
1007  for (const SparseColumn::Entry e : rhs) {
1008  initially_all_zero_scratchpad_[e.row()] = e.coefficient();
1009  }
1010

1012  for (RowIndex row(ColToRowIndex(first_non_identity_column_)); row < end_row;

1016  if (pivot == 0.0) continue;
1017  for (EntryIndex i : Column(RowToColIndex(row))) {
1018  initially_all_zero_scratchpad_[EntryRow(i)] -=
1019  EntryCoefficient(i) * pivot;
1020  }
1021  }
1022

1024  const RowIndex num_rows = num_rows_;
1025  for (RowIndex row(0); row < num_rows; ++row) {
1026  if (initially_all_zero_scratchpad_[row] != 0.0) {
1027  if (row_perm[row] < 0) {
1028  lower->SetCoefficient(row, initially_all_zero_scratchpad_[row]);
1029  } else {
1030  upper->SetCoefficient(row, initially_all_zero_scratchpad_[row]);
1031  }
1032  initially_all_zero_scratchpad_[row] = 0.0;
1033  }
1034  }

1038 void TriangularMatrix::PermutedLowerSparseSolve(const ColumnView& rhs,
1039  const RowPermutation& row_perm,
1040  SparseColumn* lower_column,
1041  SparseColumn* upper_column) {
1042  DCHECK(all_diagonal_coefficients_are_one_);
1043  RETURN_IF_NULL(lower_column);
1044  RETURN_IF_NULL(upper_column);
1045 
1046  // Compute the set of rows that will be non zero in the result (lower_column,
1047  // upper_column).
1048  PermutedComputeRowsToConsider(rhs, row_perm, &lower_column_rows_,
1049  &upper_column_rows_);
1050 
1051  // Copy rhs into initially_all_zero_scratchpad_.
1052  initially_all_zero_scratchpad_.resize(num_rows_, 0.0);
1053  for (const auto e : rhs) {
1054  initially_all_zero_scratchpad_[e.row()] = e.coefficient();
1055  }
1056 
1057  // We clear lower_column first in case upper_column and lower_column point to
1058  // the same underlying SparseColumn.
1059  num_fp_operations_ = 0;
1060  lower_column->Clear();
1061 
1062  // rows_to_consider_ contains the row to process in reverse order. Note in
1063  // particular that each "permuted_row" will never be touched again and so its
1064  // value is final. We copy the result in (lower_column, upper_column) and
1065  // clear initially_all_zero_scratchpad_ at the same time.
1066  upper_column->Reserve(upper_column->num_entries() +
1067  EntryIndex(upper_column_rows_.size()));
1068  for (const RowIndex permuted_row : Reverse(upper_column_rows_)) {
1069  const Fractional pivot = initially_all_zero_scratchpad_[permuted_row];
1070  if (pivot == 0.0) continue;
1071  // Note that permuted_row will not appear in the loop below so we
1072  // already know the value of the solution at this position.
1073  initially_all_zero_scratchpad_[permuted_row] = 0.0;
1074  const ColIndex row_as_col = RowToColIndex(row_perm[permuted_row]);
1075  DCHECK_GE(row_as_col, 0);
1076  upper_column->SetCoefficient(permuted_row, pivot);
1077  DCHECK_EQ(diagonal_coefficients_[row_as_col], 1.0);
1078  num_fp_operations_ += 1 + ColumnNumEntries(row_as_col).value();
1079  for (const auto e : column(row_as_col)) {
1080  initially_all_zero_scratchpad_[e.row()] -= e.coefficient() * pivot;
1081  }
1082  }
1083 
1084  // TODO(user): The size of lower is exact, so we could be slighly faster here.
1085  lower_column->Reserve(EntryIndex(lower_column_rows_.size()));
1086  for (const RowIndex permuted_row : lower_column_rows_) {
1087  const Fractional pivot = initially_all_zero_scratchpad_[permuted_row];
1088  initially_all_zero_scratchpad_[permuted_row] = 0.0;
1089  lower_column->SetCoefficient(permuted_row, pivot);
1090  }
1091  DCHECK(lower_column->CheckNoDuplicates());
1092  DCHECK(upper_column->CheckNoDuplicates());
1093 }

1095 // The goal is to find which rows of the working column we will need to look
1096 // at in PermutedLowerSparseSolve() when solving P^{-1}.L.P.x = rhs, 'P' being a
1097 // row permutation, 'L' a lower triangular matrix and 'this' being 'P^{-1}.L'.
1098 // Note that the columns of L that are identity columns (this is the case for
1099 // the ones corresponding to a kNonPivotal in P) can be skipped since they will
1100 // leave the working column unchanged.
1101 //
1102 // Let G denote the graph G = (V,E) of the column-to-row adjacency of A:
1103 // - 'V' is the set of nodes, one node i corresponds to a both a row
1104 // and a column (the matrix is square).
1105 // - 'E' is the set of arcs. There is an arc from node i to node j iff the
1106 // coefficient of i-th column, j-th row of A = P^{-1}.L.P is non zero.
1107 //
1108 // Let S denote the set of nodes i such that rhs_i != 0.
1109 // Let R denote the set of all accessible nodes from S in G.
1110 // x_k is possibly non-zero iff k is in R, i.e. if k is not in R then x_k = 0
1111 // for sure, and there is no need to look a the row k during the solve.
1112 //
1113 // So, to solve P^{-1}.L.P.x = rhs, only rows corresponding to P.R have to be
1114 // considered (ignoring the one that map to identity column of L). A topological
1115 // sort of P.R is used to decide in which order one should iterate on them. This
1116 // will be given by upper_column_rows_ and it will be populated in reverse
1117 // order.
1118 void TriangularMatrix::PermutedComputeRowsToConsider(
1119  const ColumnView& rhs, const RowPermutation& row_perm,
1120  RowIndexVector* lower_column_rows, RowIndexVector* upper_column_rows) {
1121  stored_.resize(num_rows_, false);
1122  marked_.resize(num_rows_, false);
1123  lower_column_rows->clear();
1124  upper_column_rows->clear();
1125  nodes_to_explore_.clear();
1126 
1127  for (SparseColumn::Entry e : rhs) {
1128  const ColIndex col = RowToColIndex(row_perm[e.row()]);
1129  if (col < 0) {
1130  stored_[e.row()] = true;
1131  lower_column_rows->push_back(e.row());
1132  } else {
1133  nodes_to_explore_.push_back(e.row());
1134  }
1135  }
1136 
1137  // Topological sort based on Depth-First-Search.
1138  // A few notes:
1139  // - By construction, if the matrix can be permuted into a lower triangular
1140  // form, there is no cycle. This code does nothing to test for cycles, but
1141  // there is a DCHECK() to detect them during debugging.
1142  // - This version uses sentinels (kInvalidRow) on nodes_to_explore_ to know
1143  // when a node has been explored (i.e. when the recursive dfs goes back in
1144  // the call stack). This is faster than an alternate implementation that
1145  // uses another Boolean array to detect when we go back in the
1146  // depth-first search.
1147  while (!nodes_to_explore_.empty()) {
1148  const RowIndex row = nodes_to_explore_.back();
1149 
1150  // If the depth-first search from the current node is finished (i.e. there
1151  // is a sentinel on the stack), we store the node (which is just before on
1152  // the stack). This will store the nodes in reverse topological order.
1153  if (row < 0) {
1154  nodes_to_explore_.pop_back();
1155  const RowIndex explored_row = nodes_to_explore_.back();
1156  nodes_to_explore_.pop_back();
1157  DCHECK(!stored_[explored_row]);
1158  stored_[explored_row] = true;
1159  upper_column_rows->push_back(explored_row);
1160 
1161  // Unmark and prune the nodes that are already unmarked. See the header
1162  // comment on marked_ for the algorithm description.
1163  //
1164  // Complexity note: The only difference with the "normal" DFS doing no
1165  // pruning is this extra loop here and the marked_[entry_row] = true in
1166  // the loop later in this function. On an already pruned graph, this is
1167  // probably between 1 and 2 times slower than the "normal" DFS.
1168  const ColIndex col = RowToColIndex(row_perm[explored_row]);
1169  EntryIndex i = starts_[col];
1170  EntryIndex end = pruned_ends_[col];
1171  while (i < end) {
1172  const RowIndex entry_row = EntryRow(i);
1173  if (!marked_[entry_row]) {
1174  --end;
1175

1231 void TriangularMatrix::ComputeRowsToConsiderWithDfs(
1232  RowIndexVector* non_zero_rows) const {
1233  if (non_zero_rows->empty()) return;
1234 
1235  // We don't start the DFS if the initial number of non-zeros is under the
1236  // sparsity_threshold. During the DFS, we abort it if the number of floating
1237  // points operations get larger than the num_ops_threshold.
1238  //
1239  // In both cases, we make sure to clear non_zero_rows so that the solving part
1240  // will use the non-hypersparse version of the code.
1241  //
1242  // TODO(user): Investigate the best thresholds.
1243  const int sparsity_threshold =
1244  static_cast<int>(0.025 * static_cast<double>(num_rows_.value()));
1245  const int num_ops_threshold =
1246  static_cast<int>(0.05 * static_cast<double>(num_rows_.value()));
1247  int num_ops = non_zero_rows->size();
1248  if (num_ops > sparsity_threshold) {
1249  non_zero_rows->clear();
1250  return;
1251  }
1252 
1253  // Initialize using the non-zero positions of the input.
1254  stored_.resize(num_rows_, false);
1255  nodes_to_explore_.clear();
1256  nodes_to_explore_.swap(*non_zero_rows);
1257 
1258  // Topological sort based on Depth-First-Search.
1259  // Same remarks as the version implemented in PermutedComputeRowsToConsider().
1260  while (!nodes_to_explore_.empty()) {
1261  const RowIndex row = nodes_to_explore_.back();
1262 
1263  // If the depth-first search from the current node is finished, we store the
1264  // node. This will store the node in reverse topological order.
1265  if (row < 0) {
1266  nodes_to_explore_.pop_back();
1267  const RowIndex explored_row = -row - 1;
1268  stored_[explored_row] = true;
1269  non_zero_rows->push_back(explored_row);
1270  continue;
1271  }
1272 
1273  // If the node is already stored, skip.
1274  if (stored_[row]) {
1275  nodes_to_explore_.pop_back();
1276  continue;
1277  }
1278 
1279  // Go one level forward in the depth-first search, and store the 'adjacent'
1280  // node on nodes_to_explore_ for further processing.
1281  //
1282  // We reverse the sign of nodes_to_explore_.back() to detect when the
1283  // DFS will be back on this node.
1284  nodes_to_explore_.back() = -row - 1;
1285  for (const EntryIndex i : Column(RowToColIndex(row))) {
1286  ++num_ops;
1287  const RowIndex entry_row = EntryRow(i);
1288  if (!stored_[entry_row]) {
1289  nodes_to_explore_.push_back(entry_row);
1290  }
1291  }
1292 
1293  // Abort if the number of operations is not negligible compared to the
1294  // number of rows. Note that this test also prevents the code from cycling
1295  // in case the matrix is actually not triangular.
1296  if (num_ops > num_ops_threshold) break;
1297  }
1298 
1299  // Clear stored_.
1300  for (const RowIndex row : *non_zero_rows) {
1301  stored_[row] = false;
1302  }
1303 
1304  // If we aborted, clear the result.
1305  if (num_ops > num_ops_threshold) non_zero_rows->clear();
1306 }

1355 // A known upper bound for the infinity norm of T^{-1} is the

1361  // Identity matrix
1362  return 1.0;
1363  }

1367  const int num_cols = num_cols_.value();

1370  const ColIndex col(is_upper ? num_cols - 1 - i : i);

1373  std::abs(diagonal_coefficients_[col]);

1376  for (const EntryIndex i : Column(col)) {
1377  row_norm_estimate[EntryRow(i)] += coeff * std::abs(EntryCoefficient(i));
1378  }

1384 Fractional TriangularMatrix::ComputeInverseInfinityNorm() const {
1385  const bool is_upper = IsUpperTriangular();

1387  DenseColumn row_sum(num_rows_, 0.0);
1388  DenseColumn right_hand_side;
1389  for (ColIndex col(0); col < num_cols_; ++col) {
1390  right_hand_side.assign(num_rows_, 0);
1391  right_hand_side[ColToRowIndex(col)] = 1.0;
1392

1394  if (is_upper) {
1395  UpperSolve(&right_hand_side);
1396  } else {
1397  LowerSolve(&right_hand_side);

1399

1401  for (RowIndex row(0); row < num_rows_; ++row) {
1402  row_sum[row] += std::abs(right_hand_side[row]);
1403  }
1404  }
1405  // Compute max_i sum_j |inverse_ij|.
1406  Fractional norm = 0.0;
1407  for (RowIndex row(0); row < num_rows_; ++row) {
1408  norm = std::max(norm, row_sum[row]);
1409  }
1410 
1411  return norm;
