37 bool IntervalsAreSortedAndNonAdjacent(
38  absl::Span<const ClosedInterval> intervals) {
39  for (int i = 1; i < intervals.size(); ++i) {
40  if (intervals[i - 1].start > intervals[i - 1].end) return false;
41  // First test make sure that intervals[i - 1].end + 1 will not overflow.
42  if (intervals[i - 1].end >= intervals[i].start ||
43  intervals[i - 1].end + 1 >= intervals[i].start) {
44  return false;
45  }
46  }
47  return intervals.empty() ? true
48  : intervals.back().start <= intervals.back().end;
49 }

63 // Transforms a sorted list of intervals in a sorted DISJOINT list for which
64 // IntervalsAreSortedAndNonAdjacent() would return true.
65 void UnionOfSortedIntervals(absl::InlinedVector<ClosedInterval, 1>* intervals) {
66  DCHECK(std::is_sorted(intervals->begin(), intervals->end()));
67  int new_size = 0;
68  for (const ClosedInterval& i : *intervals) {
69  if (new_size > 0 && i.start <= CapAdd((*intervals)[new_size - 1].end, 1)) {
70  (*intervals)[new_size - 1].end =
71  std::max(i.end, (*intervals)[new_size - 1].end);
72  } else {
73  (*intervals)[new_size++] = i;
74  }
75  }
76  intervals->resize(new_size);
77 
78  // This is important for InlinedVector in the case the result is a single
79  // intervals.
80  intervals->shrink_to_fit();
81  DCHECK(IntervalsAreSortedAndNonAdjacent(*intervals));
82 }

87 int64_t CeilRatio(int64_t value, int64_t positive_coeff) {
88  DCHECK_GT(positive_coeff, 0);
89  const int64_t result = value / positive_coeff;
90  const int64_t adjust = static_cast<int64_t>(result * positive_coeff < value);
91  return result + adjust;
92 }

94 int64_t FloorRatio(int64_t value, int64_t positive_coeff) {
95  DCHECK_GT(positive_coeff, 0);
96  const int64_t result = value / positive_coeff;
97  const int64_t adjust = static_cast<int64_t>(result * positive_coeff > value);
98  return result - adjust;
99 }

223 bool Domain::Contains(int64_t value) const {
224  // Because we only compare by start and there is no duplicate starts, this
225  // should be the next interval after the one that has a chance to contains
226  // value.
227  auto it = std::upper_bound(intervals_.begin(), intervals_.end(),
228  ClosedInterval(value, value));
229  if (it == intervals_.begin()) return false;
230  --it;
231  return value <= it->end;
232 }

235  int i = 0;
236  const auto& others = domain.intervals_;

244  return true;

247 Domain Domain::Complement() const {
248  Domain result;
249  int64_t next_start = kint64min;
250  result.intervals_.reserve(intervals_.size() + 1);
251  for (const ClosedInterval& interval : intervals_) {
252  if (interval.start != kint64min) {
253  result.intervals_.push_back({next_start, interval.start - 1});
254  }
255  if (interval.end == kint64max) return result;
256  next_start = interval.end + 1;
257  }
258  result.intervals_.push_back({next_start, kint64max});
259  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
260  return result;
261 }

269 void Domain::NegateInPlace() {
270  if (intervals_.empty()) return;
271  std::reverse(intervals_.begin(), intervals_.end());
272  if (intervals_.back().end == kint64min) {
273  // corner-case
274  intervals_.pop_back();
275  }
276  for (ClosedInterval& ref : intervals_) {
277  std::swap(ref.start, ref.end);
278  ref.start = ref.start == kint64min ? kint64max : -ref.start;
279  ref.end = ref.end == kint64min ? kint64max : -ref.end;
280  }
281  DCHECK(IntervalsAreSortedAndNonAdjacent(intervals_));
282 }

284 Domain Domain::IntersectionWith(const Domain& domain) const {
285  Domain result;
286  const auto& a = intervals_;
287  const auto& b = domain.intervals_;
288  for (int i = 0, j = 0; i < a.size() && j < b.size();) {
289  if (a[i].start <= b[j].start) {
290  if (a[i].end < b[j].start) {
291  // Empty intersection. We advance past the first interval.
292  ++i;
293  } else { // a[i].end >= b[j].start
294  // Non-empty intersection: push back the intersection of these two, and
295  // advance past the first interval to finish.
296  if (a[i].end <= b[j].end) {
297  result.intervals_.push_back({b[j].start, a[i].end});
298  ++i;
299  } else { // a[i].end > b[j].end.
300  result.intervals_.push_back({b[j].start, b[j].end});
301  ++j;
302  }
303  }
304  } else { // a[i].start > b[i].start.
305  // We do the exact same thing as above, but swapping a and b.
306  if (b[j].end < a[i].start) {
307  ++j;
308  } else { // b[j].end >= a[i].start
309  if (b[j].end <= a[i].end) {
310  result.intervals_.push_back({a[i].start, b[j].end});
311  ++j;
312  } else { // a[i].end > b[j].end.
313  result.intervals_.push_back({a[i].start, a[i].end});
314  ++i;
315  }
316  }
317  }
318  }
319  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
320  return result;
321 }

363 Domain Domain::MultiplicationBy(int64_t coeff, bool* exact) const {
364  if (exact != nullptr) *exact = true;
365  if (intervals_.empty()) return {};
366  if (coeff == 0) return Domain(0);
367 
368  const int64_t abs_coeff = std::abs(coeff);
369  const int64_t size_if_non_trivial = abs_coeff > 1 ? Size() : 0;
370  if (size_if_non_trivial > kDomainComplexityLimit) {
371  if (exact != nullptr) *exact = false;
372  return ContinuousMultiplicationBy(coeff);
373  }
374 
375  Domain result;
376  if (abs_coeff > 1) {
377  const int64_t max_value = kint64max / abs_coeff;
378  const int64_t min_value = kint64min / abs_coeff;
379  result.intervals_.reserve(size_if_non_trivial);
380  for (const ClosedInterval& i : intervals_) {
381  for (int64_t v = i.start;; ++v) {
382  // We ignore anything that overflow.
383  if (v >= min_value && v <= max_value) {
384  // Because abs_coeff > 1, all new values are disjoint.
385  const int64_t new_value = v * abs_coeff;
386  result.intervals_.push_back({new_value, new_value});
387  }
388 
389  // This is to avoid doing ++v when v is kint64max!
390  if (v == i.end) break;
391  }
392  }
393  } else {
394  result = *this;
395  }
396  if (coeff < 0) result.NegateInPlace();
397  return result;
398 }

400 Domain Domain::ContinuousMultiplicationBy(int64_t coeff) const {
401  Domain result = *this;
402  const int64_t abs_coeff = std::abs(coeff);
403  for (ClosedInterval& i : result.intervals_) {
404  i.start = CapProd(i.start, abs_coeff);
405  i.end = CapProd(i.end, abs_coeff);
406  }
407  UnionOfSortedIntervals(&result.intervals_);
408  if (coeff < 0) result.NegateInPlace();
409  return result;
410 }

412 Domain Domain::ContinuousMultiplicationBy(const Domain& domain) const {
413  Domain result;
414  for (const ClosedInterval& i : this->intervals_) {
415  for (const ClosedInterval& j : domain.intervals_) {
416  ClosedInterval new_interval;
417  const int64_t a = CapProd(i.start, j.start);
418  const int64_t b = CapProd(i.end, j.end);
419  const int64_t c = CapProd(i.start, j.end);
420  const int64_t d = CapProd(i.end, j.start);
421  new_interval.start = std::min({a, b, c, d});
422  new_interval.end = std::max({a, b, c, d});
423  result.intervals_.push_back(new_interval);
424  }
425  }
426  std::sort(result.intervals_.begin(), result.intervals_.end());
427  UnionOfSortedIntervals(&result.intervals_);
428  return result;
429 }

444 Domain Domain::InverseMultiplicationBy(const int64_t coeff) const {
445  if (coeff == 0) {
446  return Contains(0) ? Domain::AllValues() : Domain();
447  }
448  Domain result = *this;
449  int new_size = 0;
450  const int64_t abs_coeff = std::abs(coeff);
451  for (const ClosedInterval& i : result.intervals_) {
452  const int64_t start = CeilRatio(i.start, abs_coeff);
453  const int64_t end = FloorRatio(i.end, abs_coeff);
454  if (start > end) continue;
455  if (new_size > 0 && start == result.intervals_[new_size - 1].end + 1) {
456  result.intervals_[new_size - 1].end = end;
457  } else {
458  result.intervals_[new_size++] = {start, end};
459  }
460  }
461  result.intervals_.resize(new_size);
462  result.intervals_.shrink_to_fit();
463  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
464  if (coeff < 0) result.NegateInPlace();
465  return result;
466 }

468 // It is a bit difficult to see, but this code is doing the same thing as
469 // for all interval in this.UnionWith(implied_domain.Complement())):
470 // - Take the two extreme points (min and max) in interval \inter implied.
471 // - Append to result [min, max] if these points exists.
472 Domain Domain::SimplifyUsingImpliedDomain(const Domain& implied_domain) const {
473  Domain result;
474  if (implied_domain.IsEmpty()) return result;
475 
476  int i = 0;
477  int64_t min_point;
478  int64_t max_point;
479  bool started = false;
480  for (const ClosedInterval interval : intervals_) {
481  // We only "close" the new result interval if it cannot be extended by
482  // implied_domain.Complement(). The only extension possible look like:
483  // interval_: ...] [....
484  // implied : ...] [... i ...]
485  if (started && implied_domain.intervals_[i].start < interval.start) {
486  result.intervals_.push_back({min_point, max_point});
487  started = false;
488  }
489 
490  // Find the two extreme points in interval \inter implied_domain.
491  // Always stop the loop at the first interval with and end strictly greater
492  // that interval.end.
493  for (; i < implied_domain.intervals_.size(); ++i) {
494  const ClosedInterval current = implied_domain.intervals_[i];
495  if (current.end >= interval.start && current.start <= interval.end) {
496  // Current and interval have a non-empty intersection.
497  const int64_t inter_max = std::min(interval.end, current.end);
498  if (!started) {
499  started = true;
500  min_point = std::max(interval.start, current.start);
501  max_point = inter_max;
502  } else {
503  // No need to update the min_point here, and the new inter_max must
504  // necessarily be > old one.
505  DCHECK_GE(inter_max, max_point);
506  max_point = inter_max;
507  }
508  }
509  if (current.end > interval.end) break;
510  }
511  if (i == implied_domain.intervals_.size()) break;
512  }
513  if (started) {
514  result.intervals_.push_back({min_point, max_point});
515  }
516  DCHECK(IntervalsAreSortedAndNonAdjacent(result.intervals_));
517  return result;
518 }

529 bool Domain::operator<(const Domain& other) const {
530  const auto& d1 = intervals_;
531  const auto& d2 = other.intervals_;
532  const int common_size = std::min(d1.size(), d2.size());
533  for (int i = 0; i < common_size; ++i) {
534  const ClosedInterval& i1 = d1[i];
535  const ClosedInterval& i2 = d2[i];
536  if (i1.start < i2.start) return true;
537  if (i1.start > i2.start) return false;
538  if (i1.end < i2.end) return true;
539  if (i1.end > i2.end) return false;
540  }
541  return d1.size() < d2.size();
542 }

546 int64_t SumOfKMinValueInDomain(const Domain& domain, int k) {
547  int64_t current_sum = 0.0;
548  int current_index = 0;
549  for (const ClosedInterval interval : domain) {
550  if (current_index >= k) break;
551  for (int v(interval.start); v <= interval.end; ++v) {
552  if (current_index >= k) break;
553  current_index++;
554  current_sum += v;
555  }
556  }
557  return current_sum;
558 }

583 SortedDisjointIntervalList
584 SortedDisjointIntervalList::BuildComplementOnInterval(int64_t start,
585  int64_t end) {
586  SortedDisjointIntervalList interval_list;
587  int64_t next_start = start;
588  for (auto it = FirstIntervalGreaterOrEqual(start); it != this->end(); ++it) {
589  const ClosedInterval& interval = *it;
590  const int64_t next_end = CapSub(interval.start, 1);
591  if (next_end > end) break;
592  if (next_start <= next_end) {
593  interval_list.InsertInterval(next_start, next_end);
594  }
595  next_start = CapAdd(interval.end, 1);
596  }
597  if (next_start <= end) {
598  interval_list.InsertInterval(next_start, end);
599  }
600  return interval_list;
601 }

603 SortedDisjointIntervalList::Iterator SortedDisjointIntervalList::InsertInterval(
604  int64_t start, int64_t end) {
605  // start > end could mean an empty interval, but we prefer to LOG(DFATAL)
606  // anyway. Really, the user should never give us that.
607  if (start > end) {
608  LOG(DFATAL) << "Invalid interval: " << ClosedInterval({start, end});
609  return intervals_.end();
610  }
611 
612  auto result = intervals_.insert({start, end});
613  if (!result.second) return result.first; // Duplicate: exit immediately.
614 
615  // TODO(user): tune the algorithm below if it proves to be a bottleneck.
616  // For example, one could try to avoid an insertion if it's not needed
617  // (when the interval merges with a single existing interval or is fully
618  // contained by one).
619 
620  // Iterate over the previous iterators whose end is after (or almost at) our
621  // start. After this, "it1" will point to the first interval that needs to be
622  // merged with the current interval (possibly pointing to the current interval
623  // itself, if no "earlier" interval should be merged).
624  auto it1 = result.first;
625  if (start == kint64min) { // Catch underflows
626  it1 = intervals_.begin();
627  } else {
628  const int64_t before_start = start - 1;
629  while (it1 != intervals_.begin()) {
630  auto prev_it = it1;
631  --prev_it;
632  if (prev_it->end < before_start) break;
633  it1 = prev_it;
634  }
635  }
636 
637  // Ditto, on the other side: "it2" will point to the interval *after* the last
638  // one that should be merged with the current interval.
639  auto it2 = result.first;
640  if (end == kint64max) {
641  it2 = intervals_.end();
642  } else {
643  const int64_t after_end = end + 1;
644  do {
645  ++it2;
646  } while (it2 != intervals_.end() && it2->start <= after_end);
647  }
648 
649  // [it1..it2) is the range (inclusive on it1, exclusive on it2) of intervals
650  // that should be merged together. We'll set it3 = it2-1 and erase [it1..it3)
651  // and set *it3 to the merged interval.
652  auto it3 = it2;
653  it3--;
654  if (it1 == it3) return it3; // Nothing was merged.
655  const int64_t new_start = std::min(it1->start, start);
656  const int64_t new_end = std::max(it3->end, end);
657  auto it = intervals_.erase(it1, it3);
658  // HACK(user): set iterators point to *const* values. Which is expected,
659  // because if one alters a set element's value, then it collapses the set
660  // property! But in this very special case, we know that we can just overwrite
661  // it->start, so we do it.
662  const_cast<ClosedInterval*>(&(*it))->start = new_start;
663  const_cast<ClosedInterval*>(&(*it))->end = new_end;
664  return it;
665 }

667 SortedDisjointIntervalList::Iterator SortedDisjointIntervalList::GrowRightByOne(
668  int64_t value, int64_t* newly_covered) {
669  auto it = intervals_.upper_bound({value, kint64max});
670  auto it_prev = it;
671 
672  // No interval containing or adjacent to "value" on the left (i.e. below).
673  if (it != begin()) {
674  --it_prev;
675  }
676  if (it == begin() || ((value != kint64min) && it_prev->end < value - 1)) {
677  *newly_covered = value;
678  if (it == end() || it->start != value + 1) {
679  // No interval adjacent to "value" on the right: insert a singleton.
680  return intervals_.insert(it, {value, value});
681  } else {
682  // There is an interval adjacent to "value" on the right. Extend it by
683  // one. Note that we already know that there won't be a merge with another
684  // interval on the left, since there were no interval adjacent to "value"
685  // on the left.
686  DCHECK_EQ(it->start, value + 1);
687  const_cast<ClosedInterval*>(&(*it))->start = value;
688  return it;
689  }
690  }
691 
692  // At this point, "it_prev" points to an interval containing or adjacent to
693  // "value" on the left: grow it by one, and if it now touches the next
694  // interval, merge with it.
695  CHECK_NE(kint64max, it_prev->end) << "Cannot grow right by one: the interval "
696  "that would grow already ends at "
697  "kint64max";
698  *newly_covered = it_prev->end + 1;
699  if (it != end() && it_prev->end + 2 == it->start) {
700  // We need to merge it_prev with 'it'.
701  const_cast<ClosedInterval*>(&(*it_prev))->end = it->end;
702  intervals_.erase(it);
703  } else {
704  const_cast<ClosedInterval*>(&(*it_prev))->end = it_prev->end + 1;
705  }
706  return it_prev;
707 }
