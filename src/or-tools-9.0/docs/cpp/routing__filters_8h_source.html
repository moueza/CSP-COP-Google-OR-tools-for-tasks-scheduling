96  protected:

99  int64_t GetNext(int64_t node) const {
100  return (new_nexts_[node] == kUnassigned)
101  ? (IsVarSynced(node) ? Value(node) : kUnassigned)
102  : new_nexts_[node];
103  }
104  int NumPaths() const { return starts_.size(); }

110  return touched_paths_.PositionsSetAtLeastOnce();

113  return new_synchronized_unperformed_nodes_.PositionsSetAtLeastOnce();
114  }

121  virtual void OnAfterSynchronizePaths() {}

123  virtual void InitializeAcceptPath() {}

125  int64_t chain_end) = 0;
126  virtual bool FinalizeAcceptPath(const Assignment* delta,
