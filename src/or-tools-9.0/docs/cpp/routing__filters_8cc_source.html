66 class MaxActiveVehiclesFilter : public IntVarLocalSearchFilter {
67  public:
68  explicit MaxActiveVehiclesFilter(const RoutingModel& routing_model)
69  : IntVarLocalSearchFilter(routing_model.Nexts()),
70  routing_model_(routing_model),
71  is_active_(routing_model.vehicles(), false),
72  active_vehicles_(0) {}
73  bool Accept(const Assignment* delta, const Assignment* deltadelta,
74  int64_t objective_min, int64_t objective_max) override {
75  const int64_t kUnassigned = -1;
76  const Assignment::IntContainer& container = delta->IntVarContainer();
77  const int delta_size = container.Size();
78  int current_active_vehicles = active_vehicles_;
79  for (int i = 0; i < delta_size; ++i) {
80  const IntVarElement& new_element = container.Element(i);
81  IntVar* const var = new_element.Var();
82  int64_t index = kUnassigned;
83  if (FindIndex(var, &index) && routing_model_.IsStart(index)) {
84  if (new_element.Min() != new_element.Max()) {
85  // LNS detected.
86  return true;
87  }
88  const int vehicle = routing_model_.VehicleIndex(index);
89  const bool is_active =
90  (new_element.Min() != routing_model_.End(vehicle));
91  if (is_active && !is_active_[vehicle]) {
92  ++current_active_vehicles;
93  } else if (!is_active && is_active_[vehicle]) {
94  --current_active_vehicles;
95  }
96  }
97  }
98  return current_active_vehicles <=
99  routing_model_.GetMaximumNumberOfActiveVehicles();
100  }
101 
102  private:
103  void OnSynchronize(const Assignment* delta) override {
104  active_vehicles_ = 0;
105  for (int i = 0; i < routing_model_.vehicles(); ++i) {
106  const int index = routing_model_.Start(i);
107  if (IsVarSynced(index) && Value(index) != routing_model_.End(i)) {
108  is_active_[i] = true;
109  ++active_vehicles_;
110  } else {
111  is_active_[i] = false;
112  }
113  }
114  }
115 
116  const RoutingModel& routing_model_;
117  std::vector<bool> is_active_;
118  int active_vehicles_;
119 };

125  new MaxActiveVehiclesFilter(routing_model));

132 class NodeDisjunctionFilter : public IntVarLocalSearchFilter {
133  public:
134  explicit NodeDisjunctionFilter(const RoutingModel& routing_model)
135  : IntVarLocalSearchFilter(routing_model.Nexts()),
136  routing_model_(routing_model),
137  active_per_disjunction_(routing_model.GetNumberOfDisjunctions(), 0),
138  inactive_per_disjunction_(routing_model.GetNumberOfDisjunctions(), 0),
139  synchronized_objective_value_(std::numeric_limits<int64_t>::min()),
140  accepted_objective_value_(std::numeric_limits<int64_t>::min()) {}
141 
142  bool Accept(const Assignment* delta, const Assignment* deltadelta,
143  int64_t objective_min, int64_t objective_max) override {
144  const int64_t kUnassigned = -1;
145  const Assignment::IntContainer& container = delta->IntVarContainer();
146  const int delta_size = container.Size();
147  gtl::small_map<std::map<RoutingModel::DisjunctionIndex, int>>
148  disjunction_active_deltas;
149  gtl::small_map<std::map<RoutingModel::DisjunctionIndex, int>>
150  disjunction_inactive_deltas;
151  bool lns_detected = false;
152  // Update active/inactive count per disjunction for each element of delta.
153  for (int i = 0; i < delta_size; ++i) {
154  const IntVarElement& new_element = container.Element(i);
155  IntVar* const var = new_element.Var();
156  int64_t index = kUnassigned;
157  if (FindIndex(var, &index)) {
158  const bool is_inactive =
159  (new_element.Min() <= index && new_element.Max() >= index);
160  if (new_element.Min() != new_element.Max()) {
161  lns_detected = true;
162  }
163  for (const RoutingModel::DisjunctionIndex disjunction_index :
164  routing_model_.GetDisjunctionIndices(index)) {
165  const bool active_state_changed =
166  !IsVarSynced(index) || (Value(index) == index) != is_inactive;
167  if (active_state_changed) {
168  if (!is_inactive) {
169  ++gtl::LookupOrInsert(&disjunction_active_deltas,
170  disjunction_index, 0);
171  if (IsVarSynced(index)) {
172  --gtl::LookupOrInsert(&disjunction_inactive_deltas,
173  disjunction_index, 0);
174  }
175  } else {
176  ++gtl::LookupOrInsert(&disjunction_inactive_deltas,
177  disjunction_index, 0);
178  if (IsVarSynced(index)) {
179  --gtl::LookupOrInsert(&disjunction_active_deltas,
180  disjunction_index, 0);
181  }
182  }
183  }
184  }
185  }
186  }
187  // Check if any disjunction has too many active nodes.
188  for (const std::pair<RoutingModel::DisjunctionIndex, int>
189  disjunction_active_delta : disjunction_active_deltas) {
190  const int current_active_nodes =
191  active_per_disjunction_[disjunction_active_delta.first];
192  const int active_nodes =
193  current_active_nodes + disjunction_active_delta.second;
194  const int max_cardinality = routing_model_.GetDisjunctionMaxCardinality(
195  disjunction_active_delta.first);
196  // Too many active nodes.
197  if (active_nodes > max_cardinality) {
198  return false;
199  }
200  }
201  // Update penalty costs for disjunctions.
202  accepted_objective_value_ = synchronized_objective_value_;
203  for (const std::pair<RoutingModel::DisjunctionIndex, int>
204  disjunction_inactive_delta : disjunction_inactive_deltas) {
205  const int64_t penalty = routing_model_.GetDisjunctionPenalty(
206  disjunction_inactive_delta.first);
207  if (penalty != 0 && !lns_detected) {
208  const RoutingModel::DisjunctionIndex disjunction_index =
209  disjunction_inactive_delta.first;
210  const int current_inactive_nodes =
211  inactive_per_disjunction_[disjunction_index];
212  const int inactive_nodes =
213  current_inactive_nodes + disjunction_inactive_delta.second;
214  const int max_inactive_cardinality =
215  routing_model_.GetDisjunctionIndices(disjunction_index).size() -
216  routing_model_.GetDisjunctionMaxCardinality(disjunction_index);
217  // Too many inactive nodes.
218  if (inactive_nodes > max_inactive_cardinality) {
219  if (penalty < 0) {
220  // Nodes are mandatory, i.e. exactly max_cardinality nodes must be
221  // performed, so the move is not acceptable.
222  return false;
223  } else if (current_inactive_nodes <= max_inactive_cardinality) {
224  // Add penalty if there were not too many inactive nodes before the
225  // move.
226  accepted_objective_value_ =
227  CapAdd(accepted_objective_value_, penalty);
228  }
229  } else if (current_inactive_nodes > max_inactive_cardinality) {
230  // Remove penalty if there were too many inactive nodes before the
231  // move and there are not too many after the move.
232  accepted_objective_value_ =
233  CapSub(accepted_objective_value_, penalty);
234  }
235  }
236  }
237  if (lns_detected) {
238  accepted_objective_value_ = 0;
239  return true;
240  } else {
241  // Only compare to max as a cost lower bound is computed.
242  return accepted_objective_value_ <= objective_max;
243  }
244  }
245  std::string DebugString() const override { return "NodeDisjunctionFilter"; }
246  int64_t GetSynchronizedObjectiveValue() const override {
247  return synchronized_objective_value_;
248  }
249  int64_t GetAcceptedObjectiveValue() const override {
250  return accepted_objective_value_;
251  }
252 
253  private:
254  void OnSynchronize(const Assignment* delta) override {
255  synchronized_objective_value_ = 0;
256  for (RoutingModel::DisjunctionIndex i(0);
257  i < active_per_disjunction_.size(); ++i) {
258  active_per_disjunction_[i] = 0;
259  inactive_per_disjunction_[i] = 0;
260  const std::vector<int64_t>& disjunction_indices =
261  routing_model_.GetDisjunctionIndices(i);
262  for (const int64_t index : disjunction_indices) {
263  const bool index_synced = IsVarSynced(index);
264  if (index_synced) {
265  if (Value(index) != index) {
266  ++active_per_disjunction_[i];
267  } else {
268  ++inactive_per_disjunction_[i];
269  }
270  }
271  }
272  const int64_t penalty = routing_model_.GetDisjunctionPenalty(i);
273  const int max_cardinality =
274  routing_model_.GetDisjunctionMaxCardinality(i);
275  if (inactive_per_disjunction_[i] >
276  disjunction_indices.size() - max_cardinality &&
277  penalty > 0) {
278  synchronized_objective_value_ =
279  CapAdd(synchronized_objective_value_, penalty);
280  }
281  }
282  }
283 
284  const RoutingModel& routing_model_;
285 
286  absl::StrongVector<RoutingModel::DisjunctionIndex, int>
287  active_per_disjunction_;
288  absl::StrongVector<RoutingModel::DisjunctionIndex, int>
289  inactive_per_disjunction_;
290  int64_t synchronized_objective_value_;
291  int64_t accepted_objective_value_;
292 };

315 bool BasePathFilter::Accept(const Assignment* delta,
316  const Assignment* deltadelta, int64_t objective_min,
317  int64_t objective_max) {
318  if (IsDisabled()) return true;
319  for (const int touched : delta_touched_) {
320  new_nexts_[touched] = kUnassigned;
321  }
322  delta_touched_.clear();
323  const Assignment::IntContainer& container = delta->IntVarContainer();
324  const int delta_size = container.Size();
325  delta_touched_.reserve(delta_size);
326  // Determining touched paths and their touched chain start and ends (a node is
327  // touched if it corresponds to an element of delta or that an element of
328  // delta points to it).
329  // The start and end of a touched path subchain will have remained on the same
330  // path and will correspond to the min and max ranks of touched nodes in the
331  // current assignment.
332  for (int64_t touched_path : touched_paths_.PositionsSetAtLeastOnce()) {
333  touched_path_chain_start_ends_[touched_path] = {kUnassigned, kUnassigned};
334  }
335  touched_paths_.SparseClearAll();
336 
337  const auto update_touched_path_chain_start_end = [this](int64_t index) {
338  const int64_t start = node_path_starts_[index];
339  if (start == kUnassigned) return;
340  touched_paths_.Set(start);
341 
342  int64_t& chain_start = touched_path_chain_start_ends_[start].first;
343  if (chain_start == kUnassigned || ranks_[index] < ranks_[chain_start]) {
344  chain_start = index;
345  }
346 
347  int64_t& chain_end = touched_path_chain_start_ends_[start].second;
348  if (chain_end == kUnassigned || ranks_[index] > ranks_[chain_end]) {
349  chain_end = index;
350  }
351  };
352 
353  for (int i = 0; i < delta_size; ++i) {
354  const IntVarElement& new_element = container.Element(i);
355  IntVar* const var = new_element.Var();
356  int64_t index = kUnassigned;
357  if (FindIndex(var, &index)) {
358  if (!new_element.Bound()) {
359  // LNS detected
360  return true;
361  }
362  new_nexts_[index] = new_element.Value();
363  delta_touched_.push_back(index);
364  update_touched_path_chain_start_end(index);
365  update_touched_path_chain_start_end(new_nexts_[index]);
366  }
367  }
368  // Checking feasibility of touched paths.
369  InitializeAcceptPath();
370  bool accept = true;
371  for (const int64_t touched_start : touched_paths_.PositionsSetAtLeastOnce()) {
372  const std::pair<int64_t, int64_t> start_end =
373  touched_path_chain_start_ends_[touched_start];
374  if (!AcceptPath(touched_start, start_end.first, start_end.second)) {
375  accept = false;
376  break;
377  }
378  }
379  // NOTE: FinalizeAcceptPath() is only called if all paths are accepted.
380  return accept && FinalizeAcceptPath(delta, objective_min, objective_max);
381 }

383 void BasePathFilter::ComputePathStarts(std::vector<int64_t>* path_starts,
384  std::vector<int>* index_to_path) {
385  path_starts->clear();
386  const int nexts_size = Size();
387  index_to_path->assign(nexts_size, kUnassigned);
388  Bitset64<> has_prevs(nexts_size);
389  for (int i = 0; i < nexts_size; ++i) {
390  if (!IsVarSynced(i)) {
391  has_prevs.Set(i);
392  } else {
393  const int next = Value(i);
394  if (next < nexts_size) {
395  has_prevs.Set(next);
396  }
397  }
398  }
399  for (int i = 0; i < nexts_size; ++i) {
400  if (!has_prevs[i]) {
401  (*index_to_path)[i] = path_starts->size();
402  path_starts->push_back(i);
403  }
404  }
405 }

410  ComputePathStarts(&path_starts, &index_to_path);

415  if (path_starts[i] != starts_[i]) {

427 void BasePathFilter::SynchronizeFullAssignment() {
428  // Subclasses of BasePathFilter might not propagate injected objective values
429  // so making sure it is done here (can be done again by the subclass if
430  // needed).
431  ComputePathStarts(&starts_, &paths_);
432  for (int64_t index = 0; index < Size(); index++) {
433  if (IsVarSynced(index) && Value(index) == index &&
434  node_path_starts_[index] != kUnassigned) {
435  // index was performed before and is now unperformed.
436  new_synchronized_unperformed_nodes_.Set(index);
437  }
438  }
439  // Marking unactive nodes (which are not on a path).
440  node_path_starts_.assign(node_path_starts_.size(), kUnassigned);
441  // Marking nodes on a path and storing next values.
442  const int nexts_size = Size();
443  for (const int64_t start : starts_) {
444  int node = start;
445  node_path_starts_[node] = start;
446  DCHECK(IsVarSynced(node));
447  int next = Value(node);
448  while (next < nexts_size) {
449  node = next;
450  node_path_starts_[node] = start;
451  DCHECK(IsVarSynced(node));
452  next = Value(node);
453  }
454  node_path_starts_[next] = start;
455  }
456  OnBeforeSynchronizePaths();
457  UpdateAllRanks();
458  OnAfterSynchronizePaths();
459 }

461 void BasePathFilter::OnSynchronize(const Assignment* delta) {
462  if (status_ == BasePathFilter::UNKNOWN) {
463  status_ =
464  DisableFiltering() ? BasePathFilter::DISABLED : BasePathFilter::ENABLED;
465  }
466  if (IsDisabled()) return;
467  new_synchronized_unperformed_nodes_.ClearAll();
468  if (delta == nullptr || delta->Empty() || starts_.empty()) {
469  SynchronizeFullAssignment();
470  return;
471  }
472  // Subclasses of BasePathFilter might not propagate injected objective values
473  // so making sure it is done here (can be done again by the subclass if
474  // needed).
475  // This code supposes that path starts didn't change.
476  DCHECK(!absl::GetFlag(FLAGS_routing_strong_debug_checks) ||
477  !HavePathsChanged());
478  const Assignment::IntContainer& container = delta->IntVarContainer();
479  touched_paths_.SparseClearAll();
480  for (int i = 0; i < container.Size(); ++i) {
481  const IntVarElement& new_element = container.Element(i);
482  int64_t index = kUnassigned;
483  if (FindIndex(new_element.Var(), &index)) {
484  const int64_t start = node_path_starts_[index];
485  if (start != kUnassigned) {
486  touched_paths_.Set(start);
487  if (Value(index) == index) {
488  // New unperformed node (its previous start isn't unassigned).
489  DCHECK_LT(index, new_nexts_.size());
490  new_synchronized_unperformed_nodes_.Set(index);
491  node_path_starts_[index] = kUnassigned;
492  }
493  }
494  }
495  }
496  OnBeforeSynchronizePaths();
497  for (const int64_t touched_start : touched_paths_.PositionsSetAtLeastOnce()) {
498  int64_t node = touched_start;
499  while (node < Size()) {
500  node_path_starts_[node] = touched_start;
501  node = Value(node);
502  }
503  node_path_starts_[node] = touched_start;
504  UpdatePathRanksFromStart(touched_start);
505  OnSynchronizePathFromStart(touched_start);
506  }
507  OnAfterSynchronizePaths();
508 }

510 void BasePathFilter::UpdateAllRanks() {
511  for (int i = 0; i < ranks_.size(); ++i) {
512  ranks_[i] = kUnassigned;
513  }
514  for (int r = 0; r < NumPaths(); ++r) {
515  UpdatePathRanksFromStart(Start(r));
516  OnSynchronizePathFromStart(Start(r));
517  }
518 }

520 void BasePathFilter::UpdatePathRanksFromStart(int start) {
521  int rank = 0;
522  int64_t node = start;
523  while (node < Size()) {
524  ranks_[node] = rank;
525  rank++;
526  node = Value(node);
527  }
528  ranks_[node] = rank;

596 void VehicleAmortizedCostFilter::OnAfterSynchronizePaths() {
597  current_vehicle_cost_ = 0;
598  for (int vehicle = 0; vehicle < vehicle_to_start_.size(); vehicle++) {

624 bool VehicleAmortizedCostFilter::AcceptPath(int64_t path_start,
625  int64_t chain_start,
626  int64_t chain_end) {
627  // Number of nodes previously between chain_start and chain_end
628  const int previous_chain_nodes = Rank(chain_end) - 1 - Rank(chain_start);
629  CHECK_GE(previous_chain_nodes, 0);
630  int new_chain_nodes = 0;
631  int64_t node = GetNext(chain_start);
632  while (node != chain_end) {
633  new_chain_nodes++;
634  node = GetNext(node);
635  }
636 
637  const int previous_route_length = current_route_lengths_[path_start];
638  CHECK_GE(previous_route_length, 0);
639  const int new_route_length =
640  previous_route_length - previous_chain_nodes + new_chain_nodes;
641 
642  const int vehicle = start_to_vehicle_[path_start];
643  CHECK_GE(vehicle, 0);
644  DCHECK_EQ(path_start, vehicle_to_start_[vehicle]);
645 
646  // Update the cost related to used vehicles.
647  // TODO(user): Handle possible overflows.
648  if (previous_route_length == 0) {
649  // The route was empty before, it is no longer the case (changed path).
650  CHECK_GT(new_route_length, 0);
651  delta_vehicle_cost_ =
652  CapAdd(delta_vehicle_cost_, linear_cost_factor_of_vehicle_[vehicle]);
653  } else if (new_route_length == 0) {
654  // The route is now empty.
655  delta_vehicle_cost_ =
656  CapSub(delta_vehicle_cost_, linear_cost_factor_of_vehicle_[vehicle]);
657  }
658 
659  // Update the cost related to the sum of the squares of the route lengths.
660  const int64_t quadratic_cost_factor =
661  quadratic_cost_factor_of_vehicle_[vehicle];
662  delta_vehicle_cost_ =
663  CapAdd(delta_vehicle_cost_,
664  CapProd(quadratic_cost_factor,
665  previous_route_length * previous_route_length));
666  delta_vehicle_cost_ = CapSub(
667  delta_vehicle_cost_,
668  CapProd(quadratic_cost_factor, new_route_length * new_route_length));
669 
670  return true;
671 }

760 bool TypeRegulationsFilter::HardIncompatibilitiesRespected(int vehicle,
761  int64_t chain_start,
762  int64_t chain_end) {
763  if (!routing_model_.HasHardTypeIncompatibilities()) return true;
764 
765  const std::vector<int>& previous_type_counts =
766  hard_incompatibility_type_counts_per_vehicle_[vehicle];
767 
768  absl::flat_hash_map</*type*/ int, /*new_count*/ int> new_type_counts;
769  absl::flat_hash_set<int> types_to_check;
770 
771  // Go through the new nodes on the path and increment their type counts.
772  int64_t node = GetNext(chain_start);
773  while (node != chain_end) {
774  const int type = routing_model_.GetVisitType(node);
775  if (type >= 0 && routing_model_.GetVisitTypePolicy(node) !=
776  RoutingModel::ADDED_TYPE_REMOVED_FROM_VEHICLE) {
777  DCHECK_LT(type, previous_type_counts.size());
778  int& type_count = gtl::LookupOrInsert(&new_type_counts, type,
779  previous_type_counts[type]);
780  if (type_count++ == 0) {
781  // New type on the route, mark to check its incompatibilities.
782  types_to_check.insert(type);
783  }
784  }
785  node = GetNext(node);
786  }
787 
788  // Update new_type_counts by decrementing the occurrence of the types of the
789  // nodes no longer on the route.
790  node = Value(chain_start);
791  while (node != chain_end) {
792  const int type = routing_model_.GetVisitType(node);
793  if (type >= 0 && routing_model_.GetVisitTypePolicy(node) !=
794  RoutingModel::ADDED_TYPE_REMOVED_FROM_VEHICLE) {
795  DCHECK_LT(type, previous_type_counts.size());
796  int& type_count = gtl::LookupOrInsert(&new_type_counts, type,
797  previous_type_counts[type]);
798  CHECK_GE(type_count, 1);
799  type_count--;
800  }
801  node = Value(node);
802  }
803 
804  // Check the incompatibilities for types in types_to_check.
805  for (int type : types_to_check) {
806  for (int incompatible_type :
807  routing_model_.GetHardTypeIncompatibilitiesOfType(type)) {
808  if (gtl::FindWithDefault(new_type_counts, incompatible_type,
809  previous_type_counts[incompatible_type]) > 0) {
810  return false;
811  }
812  }
813  }
814  return true;
815 }

891 // On synchronization, maintain "propagated" cumul mins and max level of cumul
892 // from each node to the end of the path; to be used by AcceptPath to
893 // incrementally check feasibility.
894 void ChainCumulFilter::OnSynchronizePathFromStart(int64_t start) {
895  const int vehicle = start_to_vehicle_[start];
896  std::vector<int64_t> path_nodes;
897  int64_t node = start;
898  int64_t cumul = cumuls_[node]->Min();
899  while (node < Size()) {
900  path_nodes.push_back(node);
901  current_path_cumul_mins_[node] = cumul;
902  const int64_t next = Value(node);
903  if (next != old_nexts_[node] || vehicle != old_vehicles_[node]) {
904  old_nexts_[node] = next;
905  old_vehicles_[node] = vehicle;
906  current_transits_[node] = (*evaluators_[vehicle])(node, next);
907  }
908  cumul = CapAdd(cumul, current_transits_[node]);
909  cumul = std::max(cumuls_[next]->Min(), cumul);
910  node = next;
911  }
912  path_nodes.push_back(node);
913  current_path_cumul_mins_[node] = cumul;
914  int64_t max_cumuls = cumul;
915  for (int i = path_nodes.size() - 1; i >= 0; --i) {
916  const int64_t node = path_nodes[i];
917  max_cumuls = std::max(max_cumuls, current_path_cumul_mins_[node]);
918  current_max_of_path_end_cumul_mins_[node] = max_cumuls;
919  }
920 }

922 // The complexity of the method is O(size of chain (chain_start...chain_end).
923 bool ChainCumulFilter::AcceptPath(int64_t path_start, int64_t chain_start,
924  int64_t chain_end) {
925  const int vehicle = start_to_vehicle_[path_start];
926  const int64_t capacity = vehicle_capacities_[vehicle];
927  int64_t node = chain_start;
928  int64_t cumul = current_path_cumul_mins_[node];
929  while (node != chain_end) {
930  const int64_t next = GetNext(node);
931  if (IsVarSynced(node) && next == Value(node) &&
932  vehicle == old_vehicles_[node]) {
933  cumul = CapAdd(cumul, current_transits_[node]);
934  } else {
935  cumul = CapAdd(cumul, (*evaluators_[vehicle])(node, next));
936  }
937  cumul = std::max(cumuls_[next]->Min(), cumul);
938  if (cumul > capacity) return false;
939  node = next;
940  }
941  const int64_t end = start_to_end_[path_start];
942  const int64_t end_cumul_delta =
943  CapSub(current_path_cumul_mins_[end], current_path_cumul_mins_[node]);
944  const int64_t after_chain_cumul_delta =
945  CapSub(current_max_of_path_end_cumul_mins_[node],
946  current_path_cumul_mins_[node]);
947  return CapAdd(cumul, after_chain_cumul_delta) <= capacity &&
948  CapAdd(cumul, end_cumul_delta) <= cumuls_[end]->Max();
949 }

1441 void PathCumulFilter::OnBeforeSynchronizePaths() {
1442  total_current_cumul_cost_value_ = 0;
1443  cumul_cost_delta_ = 0;
1444  current_cumul_cost_values_.clear();
1445  if (NumPaths() > 0 &&
1446  (FilterSpanCost() || FilterCumulSoftBounds() || FilterSlackCost() ||
1447  FilterCumulSoftLowerBounds() || FilterCumulPiecewiseLinearCosts() ||
1448  FilterPrecedences() || FilterSoftSpanCost() ||
1449  FilterSoftSpanQuadraticCost())) {
1450  InitializeSupportedPathCumul(&current_min_start_,
1451  std::numeric_limits<int64_t>::max());
1452  InitializeSupportedPathCumul(&current_max_end_,
1453  std::numeric_limits<int64_t>::min());
1454  current_path_transits_.Clear();
1455  current_path_transits_.AddPaths(NumPaths());
1456  // For each path, compute the minimum end cumul and store the max of these.
1457  for (int r = 0; r < NumPaths(); ++r) {
1458  int64_t node = Start(r);
1459  const int vehicle = start_to_vehicle_[Start(r)];
1460  // First pass: evaluating route length to reserve memory to store route
1461  // information.
1462  int number_of_route_arcs = 0;
1463  while (node < Size()) {
1464  ++number_of_route_arcs;
1465  node = Value(node);
1466  }
1467  current_path_transits_.ReserveTransits(r, number_of_route_arcs);
1468  // Second pass: update cumul, transit and cost values.
1469  node = Start(r);
1470  int64_t cumul = cumuls_[node]->Min();
1471  std::vector<int64_t> min_path_cumuls;
1472  min_path_cumuls.reserve(number_of_route_arcs + 1);
1473  min_path_cumuls.push_back(cumul);
1474 
1475  int64_t current_cumul_cost_value = GetCumulSoftCost(node, cumul);
1476  current_cumul_cost_value = CapAdd(
1477  current_cumul_cost_value, GetCumulPiecewiseLinearCost(node, cumul));
1478 
1479  int64_t total_transit = 0;
1480  while (node < Size()) {
1481  const int64_t next = Value(node);
1482  const int64_t transit = (*evaluators_[vehicle])(node, next);
1483  total_transit = CapAdd(total_transit, transit);
1484  const int64_t transit_slack = CapAdd(transit, slacks_[node]->Min());
1485  current_path_transits_.PushTransit(r, node, next, transit_slack);
1486  cumul = CapAdd(cumul, transit_slack);
1487  cumul =
1488  dimension_.GetFirstPossibleGreaterOrEqualValueForNode(next, cumul);
1489  cumul = std::max(cumuls_[next]->Min(), cumul);
1490  min_path_cumuls.push_back(cumul);
1491  node = next;
1492  current_cumul_cost_value =
1493  CapAdd(current_cumul_cost_value, GetCumulSoftCost(node, cumul));
1494  current_cumul_cost_value = CapAdd(
1495  current_cumul_cost_value, GetCumulPiecewiseLinearCost(node, cumul));
1496  }
1497  if (FilterPrecedences()) {
1498  StoreMinMaxCumulOfNodesOnPath(/*path=*/r, min_path_cumuls,
1499  /*is_delta=*/false);
1500  }
1501  if (number_of_route_arcs == 1 &&
1502  !routing_model_.AreEmptyRouteCostsConsideredForVehicle(vehicle)) {
1503  // This is an empty route (single start->end arc) which we don't take
1504  // into account for costs.
1505  current_cumul_cost_values_[Start(r)] = 0;
1506  current_path_transits_.ClearPath(r);
1507  continue;
1508  }
1509  if (FilterSlackCost() || FilterSoftSpanCost() ||
1510  FilterSoftSpanQuadraticCost()) {
1511  const int64_t start = ComputePathMaxStartFromEndCumul(
1512  current_path_transits_, r, Start(r), cumul);
1513  const int64_t span_lower_bound = CapSub(cumul, start);
1514  if (FilterSlackCost()) {
1515  current_cumul_cost_value =
1516  CapAdd(current_cumul_cost_value,
1517  CapProd(vehicle_span_cost_coefficients_[vehicle],
1518  CapSub(span_lower_bound, total_transit)));
1519  }
1520  if (FilterSoftSpanCost()) {
1521  const SimpleBoundCosts::BoundCost bound_cost =
1522  dimension_.GetSoftSpanUpperBoundForVehicle(vehicle);
1523  if (bound_cost.bound < span_lower_bound) {
1524  const int64_t violation =
1525  CapSub(span_lower_bound, bound_cost.bound);
1526  current_cumul_cost_value = CapAdd(
1527  current_cumul_cost_value, CapProd(bound_cost.cost, violation));
1528  }
1529  }
1530  if (FilterSoftSpanQuadraticCost()) {
1531  const SimpleBoundCosts::BoundCost bound_cost =
1532  dimension_.GetQuadraticCostSoftSpanUpperBoundForVehicle(vehicle);
1533  if (bound_cost.bound < span_lower_bound) {
1534  const int64_t violation =
1535  CapSub(span_lower_bound, bound_cost.bound);
1536  current_cumul_cost_value =
1537  CapAdd(current_cumul_cost_value,
1538  CapProd(bound_cost.cost, CapProd(violation, violation)));
1539  }
1540  }
1541  }
1542  if (FilterCumulSoftLowerBounds()) {
1543  current_cumul_cost_value =
1544  CapAdd(current_cumul_cost_value,
1545  GetPathCumulSoftLowerBoundCost(current_path_transits_, r));
1546  }
1547  if (FilterWithDimensionCumulOptimizerForVehicle(vehicle)) {
1548  // TODO(user): Return a status from the optimizer to detect failures
1549  // The only admissible failures here are because of LP timeout.
1550  int64_t lp_cumul_cost_value = 0;
1551  LocalDimensionCumulOptimizer* const optimizer =
1552  FilterBreakCost(vehicle) ? mp_optimizer_ : optimizer_;
1553  DCHECK(optimizer != nullptr);
1554  const DimensionSchedulingStatus status =
1555  optimizer->ComputeRouteCumulCostWithoutFixedTransits(
1556  vehicle, [this](int64_t node) { return Value(node); },
1557  &lp_cumul_cost_value);
1558  switch (status) {
1559  case DimensionSchedulingStatus::INFEASIBLE:
1560  lp_cumul_cost_value = 0;
1561  break;
1562  case DimensionSchedulingStatus::RELAXED_OPTIMAL_ONLY:
1563  DCHECK(mp_optimizer_ != nullptr);
1564  if (mp_optimizer_->ComputeRouteCumulCostWithoutFixedTransits(
1565  vehicle, [this](int64_t node) { return Value(node); },
1566  &lp_cumul_cost_value) ==
1567  DimensionSchedulingStatus::INFEASIBLE) {
1568  lp_cumul_cost_value = 0;
1569  }
1570  break;
1571  default:
1572  DCHECK(status == DimensionSchedulingStatus::OPTIMAL);
1573  }
1574  current_cumul_cost_value =
1575  std::max(current_cumul_cost_value, lp_cumul_cost_value);
1576  }
1577  current_cumul_cost_values_[Start(r)] = current_cumul_cost_value;
1578  current_max_end_.path_values[r] = cumul;
1579  if (current_max_end_.cumul_value < cumul) {
1580  current_max_end_.cumul_value = cumul;
1581  current_max_end_.cumul_value_support = r;
1582  }
1583  total_current_cumul_cost_value_ =
1584  CapAdd(total_current_cumul_cost_value_, current_cumul_cost_value);
1585  }
1586  if (FilterPrecedences()) {
1587  // Update the min/max node cumuls of new unperformed nodes.
1588  for (int64_t node : GetNewSynchronizedUnperformedNodes()) {
1589  current_min_max_node_cumuls_[node] = {-1, -1};
1590  }
1591  }
1592  // Use the max of the path end cumul mins to compute the corresponding
1593  // maximum start cumul of each path; store the minimum of these.
1594  for (int r = 0; r < NumPaths(); ++r) {
1595  const int64_t start = ComputePathMaxStartFromEndCumul(
1596  current_path_transits_, r, Start(r), current_max_end_.cumul_value);
1597  current_min_start_.path_values[r] = start;
1598  if (current_min_start_.cumul_value > start) {
1599  current_min_start_.cumul_value = start;
1600  current_min_start_.cumul_value_support = r;
1601  }
1602  }
1603  }
1604  // Initialize this before considering any deltas (neighbor).
1605  delta_max_end_cumul_ = std::numeric_limits<int64_t>::min();
1606  lns_detected_ = false;
1607 
1608  DCHECK_GE(current_max_end_.cumul_value, current_min_start_.cumul_value);
1609  synchronized_objective_value_ =
1610  CapAdd(total_current_cumul_cost_value_,
1611  CapProd(global_span_cost_coefficient_,
1612  CapSub(current_max_end_.cumul_value,
1613  current_min_start_.cumul_value)));
1614 }

1638  if (next == kUnassigned) {
1639  // LNS detected, return true since other paths were ok up to now.
1640  lns_detected_ = true;
1641  return true;
1642  }
1643  ++number_of_route_arcs;
1644  node = next;
1645  }
1646  delta_path_transits_.ReserveTransits(path, number_of_route_arcs);
1647  std::vector<int64_t> min_path_cumuls;
1648  min_path_cumuls.reserve(number_of_route_arcs + 1);
1649  min_path_cumuls.push_back(cumul);
1650  // Check that the path is feasible with regards to cumul bounds, scanning
1651  // the paths from start to end (caching path node sequences and transits
1652  // for further span cost filtering).
1653  node = path_start;
1654  while (node < Size()) {
1655  const int64_t next = GetNext(node);
1656  const int64_t transit = (*evaluators_[vehicle])(node, next);
1657  total_transit = CapAdd(total_transit, transit);
1658  const int64_t transit_slack = CapAdd(transit, slacks_[node]->Min());
1659  delta_path_transits_.PushTransit(path, node, next, transit_slack);
1660  cumul = CapAdd(cumul, transit_slack);
1661  cumul = dimension_.GetFirstPossibleGreaterOrEqualValueForNode(next, cumul);
1662  if (cumul > std::min(capacity, cumuls_[next]->Max())) {
1663  return false;
1664  }
1665  cumul = std::max(cumuls_[next]->Min(), cumul);
1666  min_path_cumuls.push_back(cumul);
1667  node = next;
1668  if (filter_vehicle_costs) {
1669  cumul_cost_delta =
1670  CapAdd(cumul_cost_delta, GetCumulSoftCost(node, cumul));
1671  cumul_cost_delta =
1672  CapAdd(cumul_cost_delta, GetCumulPiecewiseLinearCost(node, cumul));
1673  }
1674  }
1675  const int64_t min_end = cumul;
1676 
1677  if (!PickupToDeliveryLimitsRespected(delta_path_transits_, path,
1678  min_path_cumuls)) {
1679  return false;
1680  }
1681  if (FilterSlackCost() || FilterBreakCost(vehicle) ||
1682  FilterSoftSpanCost(vehicle) || FilterSoftSpanQuadraticCost(vehicle)) {
1683  int64_t slack_max = std::numeric_limits<int64_t>::max();
1684  if (vehicle_span_upper_bounds_[vehicle] <
1685  std::numeric_limits<int64_t>::max()) {
1686  const int64_t span_max = vehicle_span_upper_bounds_[vehicle];
1687  slack_max = std::min(slack_max, CapSub(span_max, total_transit));
1688  }
1689  const int64_t max_start_from_min_end = ComputePathMaxStartFromEndCumul(
1690  delta_path_transits_, path, path_start, min_end);
1691  const int64_t span_lb = CapSub(min_end, max_start_from_min_end);
1692  int64_t min_total_slack = CapSub(span_lb, total_transit);
1693  if (min_total_slack > slack_max) return false;
1694 
1695  if (dimension_.HasBreakConstraints()) {
1696  for (const auto [limit, min_break_duration] :
1697  dimension_.GetBreakDistanceDurationOfVehicle(vehicle)) {
1698  // Minimal number of breaks depends on total transit:
1699  // 0 breaks for 0 <= total transit <= limit,
1700  // 1 break for limit + 1 <= total transit <= 2 * limit,
1701  // i breaks for i * limit + 1 <= total transit <= (i+1) * limit, ...
1702  if (limit == 0 || total_transit == 0) continue;
1703  const int num_breaks_lb = (total_transit - 1) / limit;
1704  const int64_t slack_lb = CapProd(num_breaks_lb, min_break_duration);
1705  if (slack_lb > slack_max) return false;
1706  min_total_slack = std::max(min_total_slack, slack_lb);
1707  }
1708  // Compute a lower bound of the amount of break that must be made inside
1709  // the route. We compute a mandatory interval (might be empty)
1710  // [max_start, min_end[ during which the route will have to happen,
1711  // then the duration of break that must happen during this interval.
1712  int64_t min_total_break = 0;
1713  int64_t max_path_end = cumuls_[routing_model_.End(vehicle)]->Max();
1714  const int64_t max_start = ComputePathMaxStartFromEndCumul(
1715  delta_path_transits_, path, path_start, max_path_end);
1716  for (const IntervalVar* br :
1717  dimension_.GetBreakIntervalsOfVehicle(vehicle)) {
1718  if (!br->MustBePerformed()) continue;
1719  if (max_start < br->EndMin() && br->StartMax() < min_end) {
1720  min_total_break = CapAdd(min_total_break, br->DurationMin());
1721  }
1722  }
1723  if (min_total_break > slack_max) return false;
1724  min_total_slack = std::max(min_total_slack, min_total_break);
1725  }
1726  if (filter_vehicle_costs) {
1727  cumul_cost_delta = CapAdd(
1728  cumul_cost_delta,
1729  CapProd(vehicle_span_cost_coefficients_[vehicle], min_total_slack));
1730  const int64_t span_lower_bound = CapAdd(total_transit, min_total_slack);
1731  if (FilterSoftSpanCost()) {
1732  const SimpleBoundCosts::BoundCost bound_cost =
1733  dimension_.GetSoftSpanUpperBoundForVehicle(vehicle);
1734  if (bound_cost.bound < span_lower_bound) {
1735  const int64_t violation = CapSub(span_lower_bound, bound_cost.bound);
1736  cumul_cost_delta =
1737  CapAdd(cumul_cost_delta, CapProd(bound_cost.cost, violation));
1738  }
1739  }
1740  if (FilterSoftSpanQuadraticCost()) {
1741  const SimpleBoundCosts::BoundCost bound_cost =
1742  dimension_.GetQuadraticCostSoftSpanUpperBoundForVehicle(vehicle);
1743  if (bound_cost.bound < span_lower_bound) {
1744  const int64_t violation = CapSub(span_lower_bound, bound_cost.bound);
1745  cumul_cost_delta =
1746  CapAdd(cumul_cost_delta,
1747  CapProd(bound_cost.cost, CapProd(violation, violation)));
1748  }
1749  }
1750  }
1751  if (CapAdd(total_transit, min_total_slack) >
1752  vehicle_span_upper_bounds_[vehicle]) {
1753  return false;
1754  }
1755  }
1756  if (FilterCumulSoftLowerBounds() && filter_vehicle_costs) {
1757  cumul_cost_delta =
1758  CapAdd(cumul_cost_delta,
1759  GetPathCumulSoftLowerBoundCost(delta_path_transits_, path));
1760  }
1761  if (FilterPrecedences()) {
1762  StoreMinMaxCumulOfNodesOnPath(path, min_path_cumuls, /*is_delta=*/true);
1763  }
1764  if (!filter_vehicle_costs) {
1765  // If this route's costs should't be taken into account, reset the
1766  // cumul_cost_delta and delta_path_transits_ for this path.
1767  cumul_cost_delta = 0;
1768  delta_path_transits_.ClearPath(path);
1769  }
1770  if (FilterSpanCost() || FilterCumulSoftBounds() || FilterSlackCost() ||
1771  FilterCumulSoftLowerBounds() || FilterCumulPiecewiseLinearCosts() ||
1772  FilterSoftSpanCost(vehicle) || FilterSoftSpanQuadraticCost(vehicle)) {
1773  delta_paths_.insert(GetPath(path_start));
1774  delta_path_cumul_cost_values_[vehicle] = cumul_cost_delta;
1775  cumul_cost_delta =
1776  CapSub(cumul_cost_delta, current_cumul_cost_values_[path_start]);
1777  if (filter_vehicle_costs) {
1778  delta_max_end_cumul_ = std::max(delta_max_end_cumul_, min_end);
1779  }
1780  }
1781  cumul_cost_delta_ = CapAdd(cumul_cost_delta_, cumul_cost_delta);
1782  return true;
1783 }

1785 bool PathCumulFilter::FinalizeAcceptPath(const Assignment* delta,
1786  int64_t objective_min,
1787  int64_t objective_max) {
1788  if ((!FilterSpanCost() && !FilterCumulSoftBounds() && !FilterSlackCost() &&
1789  !FilterCumulSoftLowerBounds() && !FilterCumulPiecewiseLinearCosts() &&
1790  !FilterPrecedences() && !FilterSoftSpanCost() &&
1791  !FilterSoftSpanQuadraticCost()) ||
1792  lns_detected_) {
1793  return true;
1794  }
1795  if (FilterPrecedences()) {
1796  for (int64_t node : delta_nodes_with_precedences_and_changed_cumul_
1797  .PositionsSetAtLeastOnce()) {
1798  const std::pair<int64_t, int64_t> node_min_max_cumul_in_delta =
1799  gtl::FindWithDefault(node_with_precedence_to_delta_min_max_cumuls_,
1800  node, {-1, -1});
1801  // NOTE: This node was seen in delta, so its delta min/max cumul should be
1802  // stored in the map.
1803  DCHECK(node_min_max_cumul_in_delta.first >= 0 &&
1804  node_min_max_cumul_in_delta.second >= 0);
1805  for (const RoutingDimension::NodePrecedence& precedence :
1806  node_index_to_precedences_[node]) {
1807  const bool node_is_first = (precedence.first_node == node);
1808  const int64_t other_node =
1809  node_is_first ? precedence.second_node : precedence.first_node;
1810  if (GetNext(other_node) == kUnassigned ||
1811  GetNext(other_node) == other_node) {
1812  // The other node is unperformed, so the precedence constraint is
1813  // inactive.
1814  continue;
1815  }
1816  // max_cumul[second_node] should be greater or equal than
1817  // min_cumul[first_node] + offset.
1818  const std::pair<int64_t, int64_t>& other_min_max_cumul_in_delta =
1819  gtl::FindWithDefault(node_with_precedence_to_delta_min_max_cumuls_,
1820  other_node,
1821  current_min_max_node_cumuls_[other_node]);
1822 
1823  const int64_t first_min_cumul =
1824  node_is_first ? node_min_max_cumul_in_delta.first
1825  : other_min_max_cumul_in_delta.first;
1826  const int64_t second_max_cumul =
1827  node_is_first ? other_min_max_cumul_in_delta.second
1828  : node_min_max_cumul_in_delta.second;
1829 
1830  if (second_max_cumul < first_min_cumul + precedence.offset) {
1831  return false;
1832  }
1833  }
1834  }
1835  }
1836  int64_t new_max_end = delta_max_end_cumul_;
1837  int64_t new_min_start = std::numeric_limits<int64_t>::max();
1838  if (FilterSpanCost()) {
1839  if (new_max_end < current_max_end_.cumul_value) {
1840  // Delta max end is lower than the current solution one.
1841  // If the path supporting the current max end has been modified, we need
1842  // to check all paths to find the largest max end.
1843  if (!gtl::ContainsKey(delta_paths_,
1844  current_max_end_.cumul_value_support)) {
1845  new_max_end = current_max_end_.cumul_value;
1846  } else {
1847  for (int i = 0; i < current_max_end_.path_values.size(); ++i) {
1848  if (current_max_end_.path_values[i] > new_max_end &&
1849  !gtl::ContainsKey(delta_paths_, i)) {
1850  new_max_end = current_max_end_.path_values[i];
1851  }
1852  }
1853  }
1854  }
1855  // Now that the max end cumul has been found, compute the corresponding
1856  // min start cumul, first from the delta, then if the max end cumul has
1857  // changed, from the unchanged paths as well.
1858  for (int r = 0; r < delta_path_transits_.NumPaths(); ++r) {
1859  new_min_start =
1860  std::min(ComputePathMaxStartFromEndCumul(delta_path_transits_, r,
1861  Start(r), new_max_end),
1862  new_min_start);
1863  }
1864  if (new_max_end != current_max_end_.cumul_value) {
1865  for (int r = 0; r < NumPaths(); ++r) {
1866  if (gtl::ContainsKey(delta_paths_, r)) {
1867  continue;
1868  }
1869  new_min_start = std::min(new_min_start, ComputePathMaxStartFromEndCumul(
1870  current_path_transits_, r,
1871  Start(r), new_max_end));
1872  }
1873  } else if (new_min_start > current_min_start_.cumul_value) {
1874  // Delta min start is greater than the current solution one.
1875  // If the path supporting the current min start has been modified, we need
1876  // to check all paths to find the smallest min start.
1877  if (!gtl::ContainsKey(delta_paths_,
1878  current_min_start_.cumul_value_support)) {
1879  new_min_start = current_min_start_.cumul_value;
1880  } else {
1881  for (int i = 0; i < current_min_start_.path_values.size(); ++i) {
1882  if (current_min_start_.path_values[i] < new_min_start &&
1883  !gtl::ContainsKey(delta_paths_, i)) {
1884  new_min_start = current_min_start_.path_values[i];
1885  }
1886  }
1887  }
1888  }
1889  }
1890 
1891  // Filtering on objective value, calling LPs and MIPs if needed..
1892  accepted_objective_value_ =
1893  CapAdd(cumul_cost_delta_, CapProd(global_span_cost_coefficient_,
1894  CapSub(new_max_end, new_min_start)));
1895 
1896  if (can_use_lp_ && optimizer_ != nullptr &&
1897  accepted_objective_value_ <= objective_max) {
1898  const size_t num_touched_paths = GetTouchedPathStarts().size();
1899  std::vector<int64_t> path_delta_cost_values(num_touched_paths, 0);
1900  std::vector<bool> requires_mp(num_touched_paths, false);
1901  for (int i = 0; i < num_touched_paths; ++i) {
1902  const int64_t start = GetTouchedPathStarts()[i];
1903  const int vehicle = start_to_vehicle_[start];
1904  if (!FilterWithDimensionCumulOptimizerForVehicle(vehicle)) {
1905  continue;
1906  }
1907  int64_t path_delta_cost_with_lp = 0;
1908  const DimensionSchedulingStatus status =
1909  optimizer_->ComputeRouteCumulCostWithoutFixedTransits(
1910  vehicle, [this](int64_t node) { return GetNext(node); },
1911  &path_delta_cost_with_lp);
1912  if (status == DimensionSchedulingStatus::INFEASIBLE) {
1913  return false;
1914  }
1915  DCHECK(gtl::ContainsKey(delta_paths_, GetPath(start)));
1916  const int64_t path_cost_diff_with_lp = CapSub(
1917  path_delta_cost_with_lp, delta_path_cumul_cost_values_[vehicle]);
1918  if (path_cost_diff_with_lp > 0) {
1919  path_delta_cost_values[i] = path_delta_cost_with_lp;
1920  accepted_objective_value_ =
1921  CapAdd(accepted_objective_value_, path_cost_diff_with_lp);
1922  if (accepted_objective_value_ > objective_max) {
1923  return false;
1924  }
1925  } else {
1926  path_delta_cost_values[i] = delta_path_cumul_cost_values_[vehicle];
1927  }
1928  if (mp_optimizer_ != nullptr) {
1929  requires_mp[i] =
1930  FilterBreakCost(vehicle) ||
1931  (status == DimensionSchedulingStatus::RELAXED_OPTIMAL_ONLY);
1932  }
1933  }
1934  if (mp_optimizer_ == nullptr) {
1935  return accepted_objective_value_ <= objective_max;
1936  }
1937  for (int i = 0; i < num_touched_paths; ++i) {
1938  if (!requires_mp[i]) {
1939  continue;
1940  }
1941  const int64_t start = GetTouchedPathStarts()[i];
1942  const int vehicle = start_to_vehicle_[start];
1943  int64_t path_delta_cost_with_mp = 0;
1944  if (mp_optimizer_->ComputeRouteCumulCostWithoutFixedTransits(
1945  vehicle, [this](int64_t node) { return GetNext(node); },
1946  &path_delta_cost_with_mp) ==
1947  DimensionSchedulingStatus::INFEASIBLE) {
1948  return false;
1949  }
1950  DCHECK(gtl::ContainsKey(delta_paths_, GetPath(start)));
1951  const int64_t path_cost_diff_with_mp =
1952  CapSub(path_delta_cost_with_mp, path_delta_cost_values[i]);
1953  if (path_cost_diff_with_mp > 0) {
1954  accepted_objective_value_ =
1955  CapAdd(accepted_objective_value_, path_cost_diff_with_mp);
1956  if (accepted_objective_value_ > objective_max) {
1957  return false;
1958  }
1959  }
1960  }
1961  }
1962 
1963  return accepted_objective_value_ <= objective_max;
1964 }

1973 bool PathCumulFilter::PickupToDeliveryLimitsRespected(
1974  const PathTransits& path_transits, int path,
1975  const std::vector<int64_t>& min_path_cumuls) const {
1976  if (!dimension_.HasPickupToDeliveryLimits()) {
1977  return true;
1978  }
1979  const int num_pairs = routing_model_.GetPickupAndDeliveryPairs().size();
1980  DCHECK_GT(num_pairs, 0);
1981  std::vector<std::pair<int, int64_t>> visited_delivery_and_min_cumul_per_pair(
1982  num_pairs, {-1, -1});
1983 
1984  const int path_size = path_transits.PathSize(path);
1985  CHECK_EQ(min_path_cumuls.size(), path_size);
1986 
1987  int64_t max_cumul = min_path_cumuls.back();
1988  for (int i = path_transits.PathSize(path) - 2; i >= 0; i--) {
1989  const int node_index = path_transits.Node(path, i);
1990  max_cumul = CapSub(max_cumul, path_transits.Transit(path, i));
1991  max_cumul = std::min(cumuls_[node_index]->Max(), max_cumul);
1992 
1993  const std::vector<std::pair<int, int>>& pickup_index_pairs =
1994  routing_model_.GetPickupIndexPairs(node_index);
1995  const std::vector<std::pair<int, int>>& delivery_index_pairs =
1996  routing_model_.GetDeliveryIndexPairs(node_index);
1997  if (!pickup_index_pairs.empty()) {
1998  // The node is a pickup. Check that it is not a delivery and that it
1999  // appears in a single pickup/delivery pair (as required when limits are
2000  // set on dimension cumuls for pickup and deliveries).
2001  DCHECK(delivery_index_pairs.empty());
2002  DCHECK_EQ(pickup_index_pairs.size(), 1);
2003  const int pair_index = pickup_index_pairs[0].first;
2004  // Get the delivery visited for this pair.
2005  const int delivery_index =
2006  visited_delivery_and_min_cumul_per_pair[pair_index].first;
2007  if (delivery_index < 0) {
2008  // No delivery visited after this pickup for this pickup/delivery pair.
2009  continue;
2010  }
2011  const int64_t cumul_diff_limit =
2012  dimension_.GetPickupToDeliveryLimitForPair(
2013  pair_index, pickup_index_pairs[0].second, delivery_index);
2014  if (CapSub(visited_delivery_and_min_cumul_per_pair[pair_index].second,
2015  max_cumul) > cumul_diff_limit) {
2016  return false;
2017  }
2018  }
2019  if (!delivery_index_pairs.empty()) {
2020  // The node is a delivery. Check that it's not a pickup and it belongs to
2021  // a single pair.
2022  DCHECK(pickup_index_pairs.empty());
2023  DCHECK_EQ(delivery_index_pairs.size(), 1);
2024  const int pair_index = delivery_index_pairs[0].first;
2025  std::pair<int, int64_t>& delivery_index_and_cumul =
2026  visited_delivery_and_min_cumul_per_pair[pair_index];
2027  int& delivery_index = delivery_index_and_cumul.first;
2028  DCHECK_EQ(delivery_index, -1);
2029  delivery_index = delivery_index_pairs[0].second;
2030  delivery_index_and_cumul.second = min_path_cumuls[i];
2031  }
2032  }
2033  return true;
2034 }

2037  int path, const std::vector<int64_t>& min_path_cumuls, bool is_delta) {

2039  is_delta ? delta_path_transits_ : current_path_transits_;

2042  DCHECK_EQ(min_path_cumuls.size(), path_size);
2043

2046  const int node_index = path_transits.Node(path, i);
2047

2049  max_cumul = CapSub(max_cumul, path_transits.Transit(path, i));

2059  is_delta ? node_with_precedence_to_delta_min_max_cumuls_[node_index]

2062  min_max_cumuls.second = max_cumul;

2065  (min_max_cumuls.first !=

2067  max_cumul != current_min_max_node_cumuls_[node_index].second)) {

2073 int64_t PathCumulFilter::ComputePathMaxStartFromEndCumul(
2074  const PathTransits& path_transits, int path, int64_t path_start,
2075  int64_t min_end_cumul) const {
2076  int64_t cumul_from_min_end = min_end_cumul;
2077  int64_t cumul_from_max_end =
2078  cumuls_[routing_model_.End(start_to_vehicle_[path_start])]->Max();

2081  const int64_t node = path_transits.Node(path, i);
2082  cumul_from_min_end =
2083  std::min(cumuls_[node]->Max(), CapSub(cumul_from_min_end, transit));
2084  cumul_from_max_end = dimension_.GetLastPossibleLessOrEqualValueForNode(
2085  node, CapSub(cumul_from_max_end, transit));
2086  }
2087  return std::min(cumul_from_min_end, cumul_from_max_end);
